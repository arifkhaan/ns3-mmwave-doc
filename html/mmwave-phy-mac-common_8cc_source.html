112 void
113 MmWavePhyMacCommon::SetNrNumerology (uint8_t index)
114 {
115  NS_LOG_FUNCTION (this << index);
116 
117  NS_ASSERT_MSG ( (index == 2) || (index == 3), "Numerology index is not valid."); // Only 2 and 3 are supported in NR for FR2.
118 
119  double subCarriersPerRB = 12; // TS 38.211 Sec 4.4.4.1
120  double subcarrierSpacing = 15 * std::pow (2, index) * 1000; // Subcarrier spacing, only 60KHz and 120KHz are supported in NR for FR2.
121 
122  m_symbolsPerSlot = 14; // TS 38.211 Sec 4.3.2: each slot must have 14 symbols
123  m_slotsPerSubframe = std::pow (2, index);// Number of slots per subframe, see TS 38.211 Sec 4.3.2
124  m_symbolsPerSubframe = m_symbolsPerSlot * m_slotsPerSubframe;
125  m_subframesPerFrame = 10;
126  m_subframePeriod = Time (MilliSeconds (1)); // TS 38.211 Section 4.3.1: the subframe duration is 1ms
127  m_symbolPeriod = Time (m_subframePeriod / m_symbolsPerSlot / m_slotsPerSubframe); // Duration of an OFDM symbol
128  m_numSubCarriersPerChunk = subCarriersPerRB;
129  m_chunkWidth = subCarriersPerRB * subcarrierSpacing;
130 }

133 MmWavePhyMacCommon::SetBandwidth (double bw)
134 {
135  NS_LOG_FUNCTION (this << bw);
136 
137  m_numChunks = bw/m_chunkWidth; // The amount of RBs is fixed to 1 as only TDMA is supported
138  m_numRefScPerSym = 864*bw/1e9; // TODO: check whether 864 is the proper value for 1 GHz of bandwidth
139 }
