352 void
353 MmWaveFlexTtiPfMacScheduler::DoSchedUlMacCtrlInfoReq (const struct MmWaveMacSchedSapProvider::SchedUlMacCtrlInfoReqParameters& params)
354 {
355  NS_LOG_FUNCTION (this);
356 
357  std::map <uint16_t,uint32_t>::iterator it;
358 
359  for (unsigned int i = 0; i < params.m_macCeList.size (); i++)
360  {
361  if ( params.m_macCeList.at (i).m_macCeType == MacCeElement::BSR )
362  {
363  // buffer status report
364  // note that this scheduler does not differentiate the
365  // allocation according to which LCGs have more/less bytes
366  // to send.
367  // Hence the BSR of different LCGs are just summed up to get
368  // a total queue size that is used for allocation purposes.
369  uint16_t rnti = params.m_macCeList.at (i).m_rnti;
370  std::map <uint16_t, struct UeSchedInfo>::iterator itUe = m_ueSchedInfoMap.find (rnti);
371 
372  uint32_t buffer = 0;
373  for (uint8_t lcg = 1; lcg <= 3; ++lcg)
374  {
375  uint8_t bsrId = params.m_macCeList.at (i).m_macCeValue.m_bufferStatus.at (lcg - 1);
376  uint32_t bufSize = BsrId2BufferSize (bsrId);
377  if (bufSize > 0)
378  {
379  buffer += bufSize;
380 
381  if (itUe == m_ueSchedInfoMap.end ())
382  {
383  NS_LOG_ERROR ("UE entry not found in sched info map");
384  }
385  else
386  {
387  int diff = bufSize - (itUe->second.m_flowStatsUl[lcg].m_totalBufSize + itUe->second.m_flowStatsUl[lcg].m_totalSchedSize);
388  if (diff > 0)
389  { // estimate additional packet sizes
390  itUe->second.m_flowStatsUl[lcg].m_totalBufSize += diff;
391  itUe->second.m_flowStatsUl[lcg].m_txPacketSizes.push_back (diff);
392  // since we expect the BSR to be generated following a packet arrival and sent at least by the end of the prev. subframe,
393  // the maximum delay is one SF (in microseconds)
394  itUe->second.m_flowStatsUl[lcg].m_txPacketDelays.push_back (m_phyMacConfig->GetSlotPeriod ().GetMicroSeconds());
395  if (itUe->second.m_flowStatsUl[lcg].m_txQueueHolDelay == 0)
396  {
397  itUe->second.m_flowStatsUl[lcg].m_txQueueHolDelay = m_phyMacConfig->GetSlotPeriod ().GetMicroSeconds();
398  }
399  }
400  }
401  }
402  }
403  }
404  }
405 
406  return;

454 MmWaveFlexTtiPfMacScheduler::DoSchedUlCqiInfoReq (const struct MmWaveMacSchedSapProvider::SchedUlCqiInfoReqParameters& params)
455 {
456  NS_LOG_FUNCTION (this);
457 
458  uint16_t frameNum = params.m_sfnSf.m_frameNum;
459  uint8_t subframeNum = params.m_sfnSf.m_sfNum;
460  uint8_t slotNum = params.m_sfnSf.m_slotNum;
461  uint8_t symNum = params.m_sfnSf.m_symStart;
462 
463  switch (params.m_ulCqi.m_type)
464  {
465  case UlCqiInfo::PUSCH:
466  {
467  std::map <uint32_t, struct AllocMapElem>::iterator itMap;
468  std::map <uint16_t, struct UlCqiMapElem>::iterator itCqi;
469  itMap = m_ulAllocationMap.find (params.m_sfnSf.Encode ());
470  if (itMap == m_ulAllocationMap.end ())
471  {
472  NS_LOG_INFO (this << " Does not find info on allocation, size : " << m_ulAllocationMap.size ());
473  return;
474  }
475  NS_ASSERT_MSG (itMap->second.m_rntiPerChunk.size () == m_phyMacConfig->GetNumChunks (), "SINR chunk map must cover full BW in TDMA mode");
476  for (unsigned i = 0; i < itMap->second.m_rntiPerChunk.size (); i++)
477  {
478  // convert from fixed point notation Sxxxxxxxxxxx.xxx to double
479  //double sinr = LteFfConverter::fpS11dot3toDouble (params.m_ulCqi.m_sinr.at (i));
480  itCqi = m_ueUlCqi.find (itMap->second.m_rntiPerChunk.at (i));
481  if (itCqi == m_ueUlCqi.end ())
482  {
483  // create a new entry
484  std::vector <double> newCqi;
485  for (uint32_t j = 0; j < m_phyMacConfig->GetNumChunks (); j++)
486  {
487  unsigned chunkInd = i;
488  if (chunkInd == j)
489  {
490  newCqi.push_back (params.m_ulCqi.m_sinr.at (i));
491  NS_LOG_INFO ("UL CQI report for RNTI " << itMap->second.m_rntiPerChunk.at (i) << " chunk " << i << " SINR " << params.m_ulCqi.m_sinr.at (i) << \
492  " frame " << frameNum << " subframe " << (unsigned)subframeNum << " slot " << (unsigned)slotNum << " startSym " << (unsigned)symNum);
493  }
494  else
495  {
496  // initialize with NO_SINR value.
497  newCqi.push_back (30.0);
498  }
499  }
500  m_ueUlCqi.insert (std::pair <uint16_t, struct UlCqiMapElem> (itMap->second.m_rntiPerChunk.at (i),
501  UlCqiMapElem (newCqi, itMap->second.m_numSym, itMap->second.m_tbSize)) );
502  // generate correspondent timer
503  m_ueCqiTimers.insert (std::pair <uint16_t, uint32_t > (itMap->second.m_rntiPerChunk.at (i), m_cqiTimersThreshold));
504  }
505  else
506  {
507  // update the value
508  (*itCqi).second.m_ueUlCqi.at (i) = params.m_ulCqi.m_sinr.at (i);
509  (*itCqi).second.m_numSym = itMap->second.m_numSym;
510  (*itCqi).second.m_tbSize = itMap->second.m_tbSize;
511  // update correspondent timer
512  std::map <uint16_t, uint32_t>::iterator itTimers;
513  itTimers = m_ueCqiTimers.find (itMap->second.m_rntiPerChunk.at (i));
514  (*itTimers).second = m_cqiTimersThreshold;
515 
516  NS_LOG_INFO ("UL CQI report for RNTI " << itMap->second.m_rntiPerChunk.at (i) << " chunk " << i << " SINR " << params.m_ulCqi.m_sinr.at (i) << \
517  " frame " << frameNum << " subframe " << (unsigned)subframeNum << " slot " << (unsigned)slotNum << " startSym " << (unsigned)symNum);
518 
519  }
520 
521  }
522  // remove obsolete info on allocation
523  m_ulAllocationMap.erase (itMap);
524  }
525  break;
526  default:
527  NS_FATAL_ERROR ("Unknown type of UL-CQI");
528  }
529  return;

704 unsigned MmWaveFlexTtiPfMacScheduler::CalcMinTbSizeNumSym (unsigned mcs, unsigned bufSize, unsigned &tbSize)
705 {
706  // Bisection line search is used to find the minimum number of slots (OFDM symbols)
707  // needed to encode entire buffer.
708  MmWaveMacPduHeader dummyMacHeader;
709  //unsigned macHdrSize = 10; //dummyMacHeader.GetSerializedSize ();
710  int numSymLow = 0;
711  int numSymHigh = m_phyMacConfig->GetSymbPerSlot();
712 
713  int diff = 0;
714  tbSize = (m_amc->GetTbSizeFromMcsSymbols (mcs, numSymHigh) / 8); // start with max value, in number of bytes
715  while ((unsigned)tbSize > bufSize)
716  {
717  diff = abs(numSymHigh-numSymLow)/2;
718  if (diff == 0)
719  {
720  tbSize = (m_amc->GetTbSizeFromMcsSymbols (mcs, numSymHigh) / 8);
721  return numSymHigh;
722  }
723  tbSize = (m_amc->GetTbSizeFromMcsSymbols (mcs, numSymHigh - diff) / 8);
724  if ((unsigned)tbSize >= bufSize)
725  {
726  numSymHigh -= diff;
727  }
728  if ((unsigned) tbSize == bufSize)
729  {
730  return numSymHigh;
731  }
732  while ((unsigned)tbSize < bufSize)
733  {
734  diff = abs(numSymHigh-numSymLow)/2;
735  if (diff == 0)
736  {
737  tbSize = (m_amc->GetTbSizeFromMcsSymbols (mcs, numSymHigh) / 8);
738  return numSymHigh;
739  }
740  tbSize = (m_amc->GetTbSizeFromMcsSymbols (mcs, numSymLow + diff) / 8);
741  if ((unsigned)tbSize <= bufSize)
742  {
743  numSymLow += diff;
744  }
745  if ((unsigned) tbSize == bufSize)
746  {
747  return numSymLow;
748  }
749  }
750  }
751  tbSize = (m_amc->GetTbSizeFromMcsSymbols (mcs, numSymHigh) / 8);
752  return (unsigned)numSymHigh;
