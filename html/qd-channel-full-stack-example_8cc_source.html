43 NS_LOG_COMPONENT_DEFINE ("QdChannelModelExample");
44 
45 using namespace ns3;
46 using namespace mmwave;

48 int
49 main (int argc, char *argv[])
50 {
51  std::string qdFilesPath = "src/spectrum/model/QD/"; // The path of the folder with the QD scenarios
52  std::string scenario = "Indoor1"; // The name of the scenario
53  uint32_t interPacketInterval = 1e3; // App inter packet arrival [us]
54  double txPower = 30.0; // Transmitted power for both eNB and UE [dBm]
55  double noiseFigure = 9.0; // Noise figure for both eNB and UE [dB]
56  uint16_t enbAntennaNum = 64; // The number of antenna elements for the gNBs antenna arrays, assuming a square architecture
57  uint16_t ueAntennaNum = 16; // The number of antenna elements for the UE antenna arrays, assuming a square architecture
58  uint32_t appPacketSize = 1460; // Application packet size [B]
59 
60  CommandLine cmd;
61  cmd.AddValue ("qdFilesPath", "The path of the folder with the QD scenarios", qdFilesPath);
62  cmd.AddValue ("scenario", "The name of the scenario", scenario);
63  cmd.AddValue ("ipi", "App inter packet arrival [us]", interPacketInterval);
64  cmd.AddValue ("txPower", "Transmitted power for both eNB and UE [dBm]", txPower);
65  cmd.AddValue ("noiseFigure", "Noise figure for both eNB and UE [dB]", noiseFigure);
66  cmd.AddValue ("enbAntennaNum", "The number of antenna elements for the gNBs antenna arrays, assuming a square architecture", enbAntennaNum);
67  cmd.AddValue ("ueAntennaNum", "The number of antenna elements for the UE antenna arrays, assuming a square architecture", ueAntennaNum);
68  cmd.AddValue ("appPacketSize", "Application packet size [B]", appPacketSize);
69  cmd.Parse (argc, argv);
70 
71  // Setup
72  LogComponentEnableAll (LOG_PREFIX_ALL);
73 
74  bool harqEnabled = true;
75  bool rlcAmEnabled = true;
76 
77  Config::SetDefault ("ns3::MmWaveHelper::RlcAmEnabled", BooleanValue (rlcAmEnabled));
78  Config::SetDefault ("ns3::MmWaveHelper::HarqEnabled", BooleanValue (harqEnabled));
79  Config::SetDefault ("ns3::MmWaveFlexTtiMacScheduler::HarqEnabled", BooleanValue (harqEnabled));
80 
81  // Create the tx and rx nodes
82  NodeContainer ueNodes;
83  NodeContainer enbNodes;
84  enbNodes.Create (1);
85  ueNodes.Create (1);
86 
87  // Create the tx and rx mobility models, set the positions to be equal to the
88  // initial positions of the nodes in the ray tracer
89  Ptr<MobilityModel> ueRefMob = CreateObject<ConstantPositionMobilityModel> ();
90  ueRefMob->SetPosition (Vector (5, 0.1, 1.5));
91  Ptr<MobilityModel> enb1Mob = CreateObject<ConstantPositionMobilityModel> ();
92  enb1Mob->SetPosition (Vector (5, 0.1, 2.9));
93 
94  // Assign the mobility models to the nodes
95  enbNodes.Get (0)->AggregateObject (enb1Mob);
96  ueNodes.Get (0)->AggregateObject (ueRefMob);
97 
98  // Configure the channel
99  Config::SetDefault ("ns3::MmWaveHelper::PathlossModel", StringValue (""));
100  Config::SetDefault ("ns3::MmWaveHelper::ChannelModel", StringValue ("ns3::ThreeGppSpectrumPropagationLossModel"));
101  Ptr<QdChannelModel> qdModel = CreateObject<QdChannelModel> (qdFilesPath, scenario);
102  Time simTime = qdModel->GetQdSimTime ();
103  Config::SetDefault ("ns3::ThreeGppSpectrumPropagationLossModel::ChannelModel", PointerValue (qdModel));
104 
105  // Bandwidth = ChunkPerRB * ChunkWidth (about 400 MHz)
106  Config::SetDefault ("ns3::MmWavePhyMacCommon::ChunkPerRB", UintegerValue (29));
107  Config::SetDefault ("ns3::MmWavePhyMacCommon::ChunkWidth", DoubleValue (13.889e6));
108 
109  // Set power and noise figure
110  Config::SetDefault ("ns3::MmWaveEnbPhy::TxPower", DoubleValue (txPower));
111  Config::SetDefault ("ns3::MmWaveEnbPhy::NoiseFigure", DoubleValue (noiseFigure));
112  Config::SetDefault ("ns3::MmWaveUePhy::TxPower", DoubleValue (txPower));
113  Config::SetDefault ("ns3::MmWaveUePhy::NoiseFigure", DoubleValue (noiseFigure));
114 
115  // Setup antenna configuration
116  Config::SetDefault ("ns3::ThreeGppAntennaArrayModel::IsotropicElements", BooleanValue (true));
117 
118  // Create the MmWave helper
119  Ptr<MmWaveHelper> mmwaveHelper = CreateObject<MmWaveHelper> ();
120  mmwaveHelper->SetMmWaveEnbNetDeviceAttribute("AntennaNum", UintegerValue (enbAntennaNum));
121  mmwaveHelper->SetMmWaveUeNetDeviceAttribute("AntennaNum", UintegerValue (ueAntennaNum));
122 
123  mmwaveHelper->SetSchedulerType ("ns3::MmWaveFlexTtiMacScheduler");
124  Ptr<MmWavePointToPointEpcHelper> epcHelper = CreateObject<MmWavePointToPointEpcHelper> ();
125  mmwaveHelper->SetEpcHelper (epcHelper);
126  mmwaveHelper->SetHarqEnabled (harqEnabled);
127 
128  // Create a single RemoteHost
129  Ptr<Node> pgw = epcHelper->GetPgwNode ();
130  NodeContainer remoteHostContainer;
131  remoteHostContainer.Create (1);
132  Ptr<Node> remoteHost = remoteHostContainer.Get (0);
133  InternetStackHelper internet;
134  internet.Install (remoteHostContainer);
135 
136  // Create the Internet
137  PointToPointHelper p2ph;
138  p2ph.SetDeviceAttribute ("DataRate", DataRateValue (DataRate ("100Gb/s")));
139  p2ph.SetDeviceAttribute ("Mtu", UintegerValue (1500));
140  p2ph.SetChannelAttribute ("Delay", TimeValue (Seconds (0.010)));
141  NetDeviceContainer internetDevices = p2ph.Install (pgw, remoteHost);
142  Ipv4AddressHelper ipv4h;
143  ipv4h.SetBase ("1.0.0.0", "255.0.0.0");
144  Ipv4InterfaceContainer internetIpIfaces = ipv4h.Assign (internetDevices);
145  // Interface 0 is localhost, 1 is the p2p device
146  Ipv4StaticRoutingHelper ipv4RoutingHelper;
147  Ptr<Ipv4StaticRouting> remoteHostStaticRouting = ipv4RoutingHelper.GetStaticRouting (remoteHost->GetObject<Ipv4> ());
148  remoteHostStaticRouting->AddNetworkRouteTo (Ipv4Address ("7.0.0.0"), Ipv4Mask ("255.0.0.0"), 1);
149 
150  // Create the tx and rx devices
151  NetDeviceContainer enbMmWaveDevs = mmwaveHelper->InstallEnbDevice (enbNodes);
152  NetDeviceContainer ueMmWaveDevs = mmwaveHelper->InstallUeDevice (ueNodes);
153 
154  // Install the IP stack on the UEs
155  internet.Install (ueNodes);
156  Ipv4InterfaceContainer ueIpIface;
157  ueIpIface = epcHelper->AssignUeIpv4Address (NetDeviceContainer (ueMmWaveDevs));
158  // Assign IP address to UEs, and install applications
159  Ptr<Node> ueNode = ueNodes.Get (0);
160  // Set the default gateway for the UE
161  Ptr<Ipv4StaticRouting> ueStaticRouting = ipv4RoutingHelper.GetStaticRouting (ueNode->GetObject<Ipv4> ());
162  ueStaticRouting->SetDefaultRoute (epcHelper->GetUeDefaultGatewayAddress (), 1);
163 
164  // This performs the attachment of each UE to a specific eNB
165  mmwaveHelper->AttachToEnbWithIndex (ueMmWaveDevs.Get (0), enbMmWaveDevs, 0);
166 
167  // Add apps
168  uint16_t dlPort = 1234;
169  uint16_t ulPort = 2000;
170  uint16_t otherPort = 3000;
171  ApplicationContainer clientApps;
172  ApplicationContainer serverApps;
173  ++ulPort;
174  ++otherPort;
175  PacketSinkHelper dlPacketSinkHelper ("ns3::UdpSocketFactory", InetSocketAddress (Ipv4Address::GetAny (), dlPort));
176  serverApps.Add (dlPacketSinkHelper.Install (ueNodes.Get (0)));
177 
178  UdpClientHelper dlClient (ueIpIface.GetAddress (0), dlPort);
179  dlClient.SetAttribute ("Interval", TimeValue (MicroSeconds (interPacketInterval)));
180  dlClient.SetAttribute ("MaxPackets", UintegerValue (1000000));
181  dlClient.SetAttribute ("PacketSize", UintegerValue (appPacketSize));
182 
183  clientApps.Add (dlClient.Install (remoteHost));
184 
185  serverApps.Start (Seconds (0.01));
186  clientApps.Start (Seconds (0.01));
187  mmwaveHelper->EnableTraces ();
188 
189  Simulator::Stop (simTime);
190  Simulator::Run ();
191  Simulator::Destroy ();
192  return 0;
193 }
