310 void
311 MmWaveUeMac::DoTransmitPdu (LteMacSapProvider::TransmitPduParameters params)
312 {
313  // TB UID passed back along with RLC data as HARQ process ID
314  std::map<uint32_t, struct MacPduInfo>::iterator it = m_macPduMap.find (params.harqProcessId);
315  if (it == m_macPduMap.end ())
316  {
317  NS_FATAL_ERROR ("No MAC PDU storage element found for this TB UID/RNTI");
318  }
319  else
320  {
321  MmWaveMacPduTag tag;
322  it->second.m_pdu->PeekPacketTag (tag);
323  if (tag.GetSfn ().m_frameNum < m_frameNum)
324  {
325  return;
326  }
327  if (it->second.m_pdu == 0)
328  {
329  it->second.m_pdu = params.pdu;
330  }
331  else
332  {
333  it->second.m_pdu->AddAtEnd (params.pdu); // append to MAC PDU
334  }
335 
336  MacSubheader subheader (params.lcid, params.pdu->GetSize ());
337  it->second.m_macHeader.AddSubheader (subheader); // add RLC PDU sub-header into MAC header
338  m_miUlHarqProcessesPacket.at (params.harqProcessId).m_lcidList.push_back (params.lcid);
339  if (it->second.m_size <
340  (params.pdu->GetSize () + it->second.m_macHeader.GetSerializedSize ()))
341  {
342  NS_FATAL_ERROR ("Maximum TB size exceeded");
343  }
344 
345  if (it->second.m_numRlcPdu <= 1)
346  {
347  // wait for all RLC PDUs to be received
348  it->second.m_pdu->AddHeader (it->second.m_macHeader);
349 
350  MmWaveMacPduHeader headerTst;
351  it->second.m_pdu->PeekHeader (headerTst);
352  if (headerTst.GetSubheaders ().size () > 0)
353  {
354  //NS_LOG_UNCOND(headerTst.GetSubheaders ().at (0).m_size);
355  }
356 
357  LteRadioBearerTag bearerTag (params.rnti, 0, 0);
358  it->second.m_pdu->AddPacketTag (bearerTag);
359  m_miUlHarqProcessesPacket.at (params.harqProcessId).m_pktBurst->AddPacket (it->second.m_pdu);
360  m_miUlHarqProcessesPacketTimer.at (params.harqProcessId) = m_phyMacConfig->GetHarqTimeout ();
361  //m_harqProcessId = (m_harqProcessId + 1) % m_phyMacConfig->GetHarqTimeout();
362 
363  m_txMacPacketTraceUe (params.rnti, m_componentCarrierId, it->second.m_pdu->GetSize ());
364 
365  m_phySapProvider->SendMacPdu (it->second.m_pdu);
366  m_macPduMap.erase (it); // delete map entry
367  }
368  else
369  {
370  it->second.m_numRlcPdu--; // decrement count of remaining RLC requests
371  }
372  }

405 void
406 MmWaveUeMac::SendReportBufferStatus (void)
407 {
408  NS_LOG_FUNCTION (this);
409 
410  if (m_rnti == 0)
411  {
412  NS_LOG_INFO ("MAC not initialized, BSR deferred");
413  return;
414  }
415 
416  if (m_ulBsrReceived.size () == 0)
417  {
418  NS_LOG_INFO ("No BSR report to transmit");
419  return;
420  }
421  MacCeElement bsr {
422  m_rnti, MacCeElement::BSR
423  };
424 
425  bool send = true;
426 
427  // BSR is reported for each LCG
428  std::map <uint8_t, LteMacSapProvider::ReportBufferStatusParameters>::iterator it;
429  std::vector<uint32_t> queue (4, 0); // one value per each of the 4 LCGs, initialized to 0
430  for (it = m_ulBsrReceived.begin (); it != m_ulBsrReceived.end (); it++)
431  {
432  uint8_t lcid = it->first;
433  NS_LOG_INFO ("MmWave lcid of BSR " << (uint16_t)lcid << " queue size " << (*it).second.txQueueSize);
434 
435  std::map <uint8_t, LcInfo>::iterator lcInfoMapIt;
436  lcInfoMapIt = m_lcInfoMap.find (lcid);
437  if (lcInfoMapIt != m_lcInfoMap.end ())
438  {
439  NS_ASSERT_MSG ((lcid != 0) || (((*it).second.txQueueSize == 0)
440  && ((*it).second.retxQueueSize == 0)
441  && ((*it).second.statusPduSize == 0)),
442  "BSR should not be used for LCID 0");
443  uint8_t lcg = lcInfoMapIt->second.lcConfig.logicalChannelGroup;
444  queue.at (lcg) += ((*it).second.txQueueSize + (*it).second.retxQueueSize + (*it).second.statusPduSize);
445  }
446  else
447  {
448  NS_LOG_INFO ("lcid not found");
449  send = false;
450  }
451  }
452 
453  // FF API says that all 4 LCGs are always present
454  bsr.m_macCeValue.m_bufferStatus.push_back (BufferSizeLevelBsr::BufferSize2BsrId (queue.at (0)));
455  bsr.m_macCeValue.m_bufferStatus.push_back (BufferSizeLevelBsr::BufferSize2BsrId (queue.at (1)));
456  bsr.m_macCeValue.m_bufferStatus.push_back (BufferSizeLevelBsr::BufferSize2BsrId (queue.at (2)));
457  bsr.m_macCeValue.m_bufferStatus.push_back (BufferSizeLevelBsr::BufferSize2BsrId (queue.at (3)));
458 
459  // create the feedback to eNB
460  Ptr<MmWaveBsrMessage> msg = Create<MmWaveBsrMessage> ();
461  msg->SetBsr (bsr);
462  if (send)
463  {
464  m_phySapProvider->SendControlMessage (msg);
465  }

526 void
527 MmWaveUeMac::DoReceivePhyPdu (Ptr<Packet> p)
528 {
529  NS_LOG_FUNCTION (this);
530  LteRadioBearerTag tag;
531  p->RemovePacketTag (tag);
532  MmWaveMacPduHeader macHeader;
533  p->RemoveHeader (macHeader);
534  NS_LOG_INFO ("ReceivePdu for rnti " << tag.GetRnti ());
535  if (tag.GetRnti () == m_rnti) // packet is for the current user
536  {
537  std::vector<MacSubheader> macSubheaders = macHeader.GetSubheaders ();
538  //NS_LOG_UNCOND("MmWaveUeMac receive PHY pdu with " << macSubheaders.size() << " pdu");
539  uint32_t currPos = 0;
540  for (unsigned ipdu = 0; ipdu < macSubheaders.size (); ipdu++)
541  {
542  if (macSubheaders[ipdu].m_size == 0)
543  {
544  continue;
545  }
546  NS_LOG_INFO ("It is for lcid " << (uint16_t)macSubheaders[ipdu].m_lcid);
547  std::map <uint8_t, LcInfo>::const_iterator it = m_lcInfoMap.find (macSubheaders[ipdu].m_lcid);
548  //NS_ASSERT_MSG (it != m_lcInfoMap.end (), "received packet with unknown lcid " << (uint16_t)macSubheaders[ipdu].m_lcid);
549  if (it == m_lcInfoMap.end ())
550  {
551  NS_LOG_WARN ("received packet with unknown lcid " << (uint16_t)macSubheaders[ipdu].m_lcid);
552  }
553 
554  if (it != m_lcInfoMap.end ())
555  {
556  Ptr<Packet> rlcPdu;
557  if ((p->GetSize () - currPos) < (uint32_t)macSubheaders[ipdu].m_size)
558  {
559  NS_LOG_ERROR ("Packet size less than specified in MAC header (actual= " \
560  << p->GetSize () << " header= " << (uint32_t)macSubheaders[ipdu].m_size << ")" );
561  }
562  else if ((p->GetSize () - currPos) > (uint32_t)macSubheaders[ipdu].m_size)
563  {
564  NS_LOG_DEBUG ("Fragmenting MAC PDU (packet size greater than specified in MAC header (actual= " \
565  << p->GetSize () << " header= " << (uint32_t)macSubheaders[ipdu].m_size << ")" );
566  rlcPdu = p->CreateFragment (currPos, (uint32_t)macSubheaders[ipdu].m_size);
567  currPos += (uint32_t)macSubheaders[ipdu].m_size;
568 
569  LteMacSapUser::ReceivePduParameters rxPduParams;
570  rxPduParams.p = rlcPdu;
571  rxPduParams.rnti = m_rnti;
572  rxPduParams.lcid = macSubheaders[ipdu].m_lcid;
573  it->second.macSapUser->ReceivePdu (rxPduParams);
574  }
575  else
576  {
577  rlcPdu = p->CreateFragment (currPos, p->GetSize () - currPos);
578  currPos = p->GetSize ();
579  LteRlcSpecificLteMacSapUser* user = (LteRlcSpecificLteMacSapUser*)it->second.macSapUser;
580 
581  LteMacSapUser::ReceivePduParameters rxPduParams;
582  rxPduParams.p = rlcPdu;
583  rxPduParams.rnti = m_rnti;
584  rxPduParams.lcid = macSubheaders[ipdu].m_lcid;
585  user->ReceivePdu (rxPduParams);
586  }
587  }
588  }
589  }
