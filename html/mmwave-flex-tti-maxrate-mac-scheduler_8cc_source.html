349 MmWaveFlexTtiMaxRateMacScheduler::DoSchedUlMacCtrlInfoReq (const struct MmWaveMacSchedSapProvider::SchedUlMacCtrlInfoReqParameters& params)
350 {
351  NS_LOG_FUNCTION (this);
352 
353  std::map <uint16_t,uint32_t>::iterator it;
354 
355  for (unsigned int i = 0; i < params.m_macCeList.size (); i++)
356  {
357  if ( params.m_macCeList.at (i).m_macCeType == MacCeElement::BSR )
358  {
359  // buffer status report
360  // note that this scheduler does not differentiate the
361  // allocation according to which LCGs have more/less bytes
362  // to send.
363  // Hence the BSR of different LCGs are just summed up to get
364  // a total queue size that is used for allocation purposes.
365  uint16_t rnti = params.m_macCeList.at (i).m_rnti;
366  std::map <uint16_t, struct UeSchedInfo>::iterator itUe = m_ueSchedInfoMap.find (rnti);
367 
368  uint32_t buffer = 0;
369  for (uint8_t lcg = 1; lcg <= 3; ++lcg)
370  {
371  uint8_t bsrId = params.m_macCeList.at (i).m_macCeValue.m_bufferStatus.at (lcg - 1);
372  uint32_t bufSize = BsrId2BufferSize (bsrId);
373  if (bufSize > 0)
374  {
375  buffer += bufSize;
376 
377  if (itUe == m_ueSchedInfoMap.end ())
378  {
379  NS_LOG_ERROR ("UE entry not found in sched info map");
380  }
381  else
382  {
383  int diff = bufSize - (itUe->second.m_flowStatsUl[lcg].m_totalBufSize + itUe->second.m_flowStatsUl[lcg].m_totalSchedSize);
384  if (diff > 0)
385  { // estimate additional packet sizes
386  itUe->second.m_flowStatsUl[lcg].m_totalBufSize += diff;
387  itUe->second.m_flowStatsUl[lcg].m_txPacketSizes.push_back (diff);
388  // since we expect the BSR to be generated following a packet arrival and sent at least by the end of the prev. subframe,
389  // the maximum delay is one SF (in microseconds)
390  itUe->second.m_flowStatsUl[lcg].m_txPacketDelays.push_back (m_phyMacConfig->GetSlotPeriod ().GetMicroSeconds());
391  if (itUe->second.m_flowStatsUl[lcg].m_txQueueHolDelay == 0)
392  {
393  itUe->second.m_flowStatsUl[lcg].m_txQueueHolDelay = m_phyMacConfig->GetSlotPeriod ().GetMicroSeconds();
394  }
395  }
396  }
397  }
398  }
399  }
400  }
401 
402  return;

450 MmWaveFlexTtiMaxRateMacScheduler::DoSchedUlCqiInfoReq (const struct MmWaveMacSchedSapProvider::SchedUlCqiInfoReqParameters& params)
451 {
452  NS_LOG_FUNCTION (this);
453 
454  unsigned frameNum = params.m_sfnSf.m_frameNum;
455  unsigned subframeNum = params.m_sfnSf.m_sfNum;
456  unsigned startSymIdx = params.m_sfnSf.m_slotNum;
457 
458  switch (params.m_ulCqi.m_type)
459  {
460  case UlCqiInfo::PUSCH:
461  {
462  std::map <uint32_t, struct AllocMapElem>::iterator itMap;
463  std::map <uint16_t, struct UlCqiMapElem>::iterator itCqi;
464  itMap = m_ulAllocationMap.find (params.m_sfnSf.Encode ());
465  if (itMap == m_ulAllocationMap.end ())
466  {
467  NS_LOG_INFO (this << " Does not find info on allocation, size : " << m_ulAllocationMap.size ());
468  return;
469  }
470  NS_ASSERT_MSG (itMap->second.m_rntiPerChunk.size () == m_phyMacConfig->GetNumChunks (), "SINR chunk map must cover full BW in TDMA mode");
471  for (unsigned i = 0; i < itMap->second.m_rntiPerChunk.size (); i++)
472  {
473  // convert from fixed point notation Sxxxxxxxxxxx.xxx to double
474  //double sinr = LteFfConverter::fpS11dot3toDouble (params.m_ulCqi.m_sinr.at (i));
475  itCqi = m_ueUlCqi.find (itMap->second.m_rntiPerChunk.at (i));
476  if (itCqi == m_ueUlCqi.end ())
477  {
478  // create a new entry
479  std::vector <double> newCqi;
480  for (uint32_t j = 0; j < m_phyMacConfig->GetNumChunks (); j++)
481  {
482  unsigned chunkInd = i;
483  if (chunkInd == j)
484  {
485  newCqi.push_back (params.m_ulCqi.m_sinr.at (i));
486  NS_LOG_INFO ("UL CQI report for RNTI " << itMap->second.m_rntiPerChunk.at (i) << " chunk " << i << " SINR " << params.m_ulCqi.m_sinr.at (i) << \
487  " frame " << frameNum << " subframe " << subframeNum << " startSym " << startSymIdx);
488  }
489  else
490  {
491  // initialize with NO_SINR value.
492  newCqi.push_back (30.0);
493  }
494  }
495  m_ueUlCqi.insert (std::pair <uint16_t, struct UlCqiMapElem> (itMap->second.m_rntiPerChunk.at (i),
496  UlCqiMapElem (newCqi, itMap->second.m_numSym, itMap->second.m_tbSize)) );
497  // generate correspondent timer
498  m_ueCqiTimers.insert (std::pair <uint16_t, uint32_t > (itMap->second.m_rntiPerChunk.at (i), m_cqiTimersThreshold));
499  }
500  else
501  {
502  // update the value
503  (*itCqi).second.m_ueUlCqi.at (i) = params.m_ulCqi.m_sinr.at (i);
504  (*itCqi).second.m_numSym = itMap->second.m_numSym;
505  (*itCqi).second.m_tbSize = itMap->second.m_tbSize;
506  // update correspondent timer
507  std::map <uint16_t, uint32_t>::iterator itTimers;
508  itTimers = m_ueCqiTimers.find (itMap->second.m_rntiPerChunk.at (i));
509  (*itTimers).second = m_cqiTimersThreshold;
510 
511  NS_LOG_INFO ("UL CQI report for RNTI " << itMap->second.m_rntiPerChunk.at (i) << " chunk " << i << " SINR " << params.m_ulCqi.m_sinr.at (i) << \
512  " frame " << frameNum << " subframe " << subframeNum << " startSym " << startSymIdx);
513 
514  }
515 
516  }
517  // remove obsolete info on allocation
518  m_ulAllocationMap.erase (itMap);
519  }
520  break;
521  default:
522  NS_FATAL_ERROR ("Unknown type of UL-CQI");
523  }

699 unsigned MmWaveFlexTtiMaxRateMacScheduler::CalcMinTbSizeNumSym (unsigned mcs, unsigned bufSize, unsigned &tbSize)
700 {
701  // Bisection line search is used to find the minimum number of slots (OFDM symbols)
702  // needed to encode entire buffer.
703  MmWaveMacPduHeader dummyMacHeader;
704  //unsigned macHdrSize = 10; //dummyMacHeader.GetSerializedSize ();
705  int numSymLow = 0;
706  int numSymHigh = m_phyMacConfig->GetSymbPerSlot();
707 
708  int diff = 0;
709  tbSize = (m_amc->GetTbSizeFromMcsSymbols (mcs, numSymHigh) / 8); // start with max value, in number of bytes
710  while ((unsigned)tbSize > bufSize)
711  {
712  diff = abs(numSymHigh-numSymLow)/2;
713  if (diff == 0)
714  {
715  tbSize = (m_amc->GetTbSizeFromMcsSymbols (mcs, numSymHigh) / 8);
716  return numSymHigh;
717  }
718  tbSize = (m_amc->GetTbSizeFromMcsSymbols (mcs, numSymHigh - diff) / 8);
719  if ((unsigned)tbSize >= bufSize)
720  {
721  numSymHigh -= diff;
722  }
723  if ((unsigned) tbSize == bufSize)
724  {
725  return numSymHigh;
726  }
727  while ((unsigned)tbSize < bufSize)
728  {
729  diff = abs(numSymHigh-numSymLow)/2;
730  if (diff == 0)
731  {
732  tbSize = (m_amc->GetTbSizeFromMcsSymbols (mcs, numSymHigh) / 8);
733  return numSymHigh;
734  }
735  tbSize = (m_amc->GetTbSizeFromMcsSymbols (mcs, numSymLow + diff) / 8);
736  if ((unsigned)tbSize <= bufSize)
737  {
738  numSymLow += diff;
739  }
740  if ((unsigned) tbSize == bufSize)
741  {
742  return numSymLow;
743  }
744  }
745  }
746  tbSize = (m_amc->GetTbSizeFromMcsSymbols (mcs, numSymHigh) / 8);
747  return (unsigned)numSymHigh;
