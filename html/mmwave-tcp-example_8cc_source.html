131 void
132 MyApp::SendPacket (void)
133 {
134  static int send_num = 1;
135  Ptr<Packet> packet = Create<Packet> (m_packetSize);
136  m_socket->Send (packet);
137  NS_LOG_DEBUG ("Sending: " << send_num++ << "\t" << Simulator::Now ().GetSeconds ());
138 
139  if (++m_packetsSent < m_nPackets)
140  {
141  ScheduleTx ();
142  }
143 }

151 void
152 MyApp::ScheduleTx (void)
153 {
154  if (m_running)
155  {
156  Time tNext (Seconds (m_packetSize * 8 / static_cast<double> (m_dataRate.GetBitRate ())));
157  m_sendEvent = Simulator::Schedule (tNext, &MyApp::SendPacket, this);
158  }

171  Config::SetDefault ("ns3::LteRlcUm::MaxTxBufferSize", UintegerValue (10 * 1024 * 1024));
172  Config::SetDefault ("ns3::LteRlcAm::MaxTxBufferSize", UintegerValue (10 * 1024 * 1024));
173  Config::SetDefault ("ns3::LteRlcUmLowLat::MaxTxBufferSize", UintegerValue (10 * 1024 * 1024));
174 
175  // TCP settings
176  Config::SetDefault ("ns3::TcpL4Protocol::SocketType", TypeIdValue (TcpCubic::GetTypeId ()));
177  Config::SetDefault ("ns3::TcpSocketBase::MinRto", TimeValue (MilliSeconds (200)));
178  Config::SetDefault ("ns3::Ipv4L3Protocol::FragmentExpirationTimeout", TimeValue (Seconds (0.2)));
179  Config::SetDefault ("ns3::TcpSocket::SegmentSize", UintegerValue (1400));
180  Config::SetDefault ("ns3::TcpSocket::DelAckCount", UintegerValue (1));
181  Config::SetDefault ("ns3::TcpSocket::SndBufSize", UintegerValue (131072*400));
182  Config::SetDefault ("ns3::TcpSocket::RcvBufSize", UintegerValue (131072*400));
183 
184  // set to false to use the 3GPP radiation pattern (proper configuration of the bearing and downtilt angles is needed)
185  Config::SetDefault ("ns3::ThreeGppAntennaArrayModel::IsotropicElements", BooleanValue (true));
186 
187  double stopTime = 5.9;
188  double simStopTime = 7.00;
189  Ipv4Address remoteHostAddr;
190 
191  // Command line arguments
192  CommandLine cmd;
193  cmd.Parse (argc, argv);
194 
195  Ptr<MmWaveHelper> mmwaveHelper = CreateObject<MmWaveHelper> ();
196  mmwaveHelper->SetSchedulerType ("ns3::MmWaveFlexTtiMacScheduler");
197  Ptr<MmWavePointToPointEpcHelper> epcHelper = CreateObject<MmWavePointToPointEpcHelper> ();
198  mmwaveHelper->SetEpcHelper (epcHelper);
199 
200  ConfigStore inputConfig;
201  inputConfig.ConfigureDefaults ();
202 
203  // parse again so you can override default values from the command line
204  cmd.Parse (argc, argv);
205 
206  Ptr<Node> pgw = epcHelper->GetPgwNode ();
207 
208  // Create a single RemoteHost
209  NodeContainer remoteHostContainer;
210  remoteHostContainer.Create (1);
211  Ptr<Node> remoteHost = remoteHostContainer.Get (0);
212  InternetStackHelper internet;
213  internet.Install (remoteHostContainer);
214 
215  // Create the Internet
216  PointToPointHelper p2ph;
217  p2ph.SetDeviceAttribute ("DataRate", DataRateValue (DataRate ("100Gb/s")));
218  p2ph.SetDeviceAttribute ("Mtu", UintegerValue (1500));
219  p2ph.SetChannelAttribute ("Delay", TimeValue (MicroSeconds (1)));
220  NetDeviceContainer internetDevices = p2ph.Install (pgw, remoteHost);
221  Ipv4AddressHelper ipv4h;
222  ipv4h.SetBase ("1.0.0.0", "255.0.0.0");
223  Ipv4InterfaceContainer internetIpIfaces = ipv4h.Assign (internetDevices);
224  // interface 0 is localhost, 1 is the p2p device
225  remoteHostAddr = internetIpIfaces.GetAddress (1);
226 
227  Ipv4StaticRoutingHelper ipv4RoutingHelper;
228  Ptr<Ipv4StaticRouting> remoteHostStaticRouting = ipv4RoutingHelper.GetStaticRouting (remoteHost->GetObject<Ipv4> ());
229  remoteHostStaticRouting->AddNetworkRouteTo (Ipv4Address ("7.0.0.0"), Ipv4Mask ("255.0.0.0"), 1);
230 
231  NodeContainer ueNodes;
232  NodeContainer enbNodes;
233  enbNodes.Create (1);
234  ueNodes.Create (1);
235 
236  // Install Mobility Model
237  MobilityHelper enbmobility;
238  Ptr<ListPositionAllocator> enbPositionAlloc = CreateObject<ListPositionAllocator> ();
239  enbPositionAlloc->Add (Vector (0.0, 0.0, 0.0));
240  enbmobility.SetMobilityModel ("ns3::ConstantPositionMobilityModel");
241  enbmobility.SetPositionAllocator (enbPositionAlloc);
242  enbmobility.Install (enbNodes);
243 
244  MobilityHelper uemobility;
245  Ptr<ListPositionAllocator> uePositionAlloc = CreateObject<ListPositionAllocator> ();
246  uePositionAlloc->Add (Vector (30.0, 0.0, 0.0));
247  uemobility.SetMobilityModel ("ns3::ConstantPositionMobilityModel");
248  uemobility.SetPositionAllocator (uePositionAlloc);
249  uemobility.Install (ueNodes);
250 
251  // Install LTE Devices to the nodes
252  NetDeviceContainer enbDevs = mmwaveHelper->InstallEnbDevice (enbNodes);
253  NetDeviceContainer ueDevs = mmwaveHelper->InstallUeDevice (ueNodes);
254 
255  // Install the IP stack on the UEs
256  // Assign IP address to UEs, and install applications
257  internet.Install (ueNodes);
258  Ipv4InterfaceContainer ueIpIface;
259  ueIpIface = epcHelper->AssignUeIpv4Address (NetDeviceContainer (ueDevs));
260 
261  mmwaveHelper->AttachToClosestEnb (ueDevs, enbDevs);
262  mmwaveHelper->EnableTraces ();
263 
264  // Set the default gateway for the UE
265  Ptr<Node> ueNode = ueNodes.Get (0);
266  Ptr<Ipv4StaticRouting> ueStaticRouting = ipv4RoutingHelper.GetStaticRouting (ueNode->GetObject<Ipv4> ());
267  ueStaticRouting->SetDefaultRoute (epcHelper->GetUeDefaultGatewayAddress (), 1);
268 
269  // Install and start applications on UEs and remote host
270  uint16_t sinkPort = 20000;
271 
272  Address sinkAddress (InetSocketAddress (ueIpIface.GetAddress (0), sinkPort));
273  PacketSinkHelper packetSinkHelper ("ns3::TcpSocketFactory", InetSocketAddress (Ipv4Address::GetAny (), sinkPort));
274  ApplicationContainer sinkApps = packetSinkHelper.Install (ueNodes.Get (0));
275  sinkApps.Start (Seconds (0.));
276  sinkApps.Stop (Seconds (simStopTime));
277 
278  Ptr<Socket> ns3TcpSocket = Socket::CreateSocket (remoteHostContainer.Get (0), TcpSocketFactory::GetTypeId ());
279 
280  Ptr<MyApp> app = CreateObject<MyApp> ();
281  app->Setup (ns3TcpSocket, sinkAddress, 1400, 5000000, DataRate ("1000Mb/s"));
282  remoteHostContainer.Get (0)->AddApplication (app);
283 
284  AsciiTraceHelper asciiTraceHelper;
285  Ptr<OutputStreamWrapper> stream1 = asciiTraceHelper.CreateFileStream ("mmWave-tcp-window.txt");
286  ns3TcpSocket->TraceConnectWithoutContext ("CongestionWindow", MakeBoundCallback (&CwndChange, stream1));
287 
288  Ptr<OutputStreamWrapper> stream2 = asciiTraceHelper.CreateFileStream ("mmWave-tcp-data.txt");
289  sinkApps.Get (0)->TraceConnectWithoutContext ("Rx",MakeBoundCallback (&Rx, stream2));
290 
291  app->SetStartTime (Seconds (0.1));
292  app->SetStopTime (Seconds (stopTime));
293 
294 
295  p2ph.EnablePcapAll ("mmwave-sgi-capture");
296 
297  Simulator::Stop (Seconds (simStopTime));
298  Simulator::Run ();
299 
300 
301  Simulator::Destroy ();
302 
303  return 0;
