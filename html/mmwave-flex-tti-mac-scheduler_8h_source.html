125  uint32_t
126  BsrId2BufferSize (uint8_t val)
127  {
128  NS_ABORT_MSG_UNLESS (val < 64, "val = " << val << " is out of range");
129  return BufferSizeLevelBsrTable[val];
130  }
131 
132  uint8_t
133  BufferSize2BsrId (uint32_t val)
134  {
135  int index = 0;
136  if (BufferSizeLevelBsrTable[63] < val)
137  {
138  index = 63;
139  }
140  else
141  {
142  while (BufferSizeLevelBsrTable[index] < val)
143  {
144  NS_ASSERT (index < 64);
145  index++;
146  }
147  }
148 
149  return (index);
150  }
