211 std::pair<Box, std::list<Box> >
212 GenerateBuildingBounds (double xArea, double yArea, double maxBuildSize, std::list<Box> m_previousBlocks )
213 {
214 
215  Ptr<UniformRandomVariable> xMinBuilding = CreateObject<UniformRandomVariable> ();
216  xMinBuilding->SetAttribute ("Min",DoubleValue (30));
217  xMinBuilding->SetAttribute ("Max",DoubleValue (xArea));
218 
219  NS_LOG_UNCOND ("min " << 0 << " max " << xArea);
220 
221  Ptr<UniformRandomVariable> yMinBuilding = CreateObject<UniformRandomVariable> ();
222  yMinBuilding->SetAttribute ("Min",DoubleValue (0));
223  yMinBuilding->SetAttribute ("Max",DoubleValue (yArea));
224 
225  NS_LOG_UNCOND ("min " << 0 << " max " << yArea);
226 
227  Box box;
228  uint32_t attempt = 0;
229  do
230  {
231  NS_ASSERT_MSG (attempt < 100, "Too many failed attempts to position non-overlapping buildings. Maybe area too small or too many buildings?");
232  box.xMin = xMinBuilding->GetValue ();
233 
234  Ptr<UniformRandomVariable> xMaxBuilding = CreateObject<UniformRandomVariable> ();
235  xMaxBuilding->SetAttribute ("Min",DoubleValue (box.xMin));
236  xMaxBuilding->SetAttribute ("Max",DoubleValue (box.xMin + maxBuildSize));
237  box.xMax = xMaxBuilding->GetValue ();
238 
239  box.yMin = yMinBuilding->GetValue ();
240 
241  Ptr<UniformRandomVariable> yMaxBuilding = CreateObject<UniformRandomVariable> ();
242  yMaxBuilding->SetAttribute ("Min",DoubleValue (box.yMin));
243  yMaxBuilding->SetAttribute ("Max",DoubleValue (box.yMin + maxBuildSize));
244  box.yMax = yMaxBuilding->GetValue ();
245 
246  ++attempt;
247  }
248  while (OverlapWithAnyPrevious (box, m_previousBlocks));
249 
250 
251  NS_LOG_UNCOND ("Building in coordinates (" << box.xMin << " , " << box.yMin << ") and (" << box.xMax << " , " << box.yMax <<
252  ") accepted after " << attempt << " attempts");
253  m_previousBlocks.push_back (box);
254  std::pair<Box, std::list<Box> > pairReturn = std::make_pair (box,m_previousBlocks);
255  return pairReturn;
