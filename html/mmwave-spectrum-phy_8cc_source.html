333 void
334 MmWaveSpectrumPhy::StartRx (Ptr<SpectrumSignalParameters> params)
335 {
336 
337  NS_LOG_FUNCTION (this);
338 
339  // TODO the code below avoids to receive a signal from devices of the
340  // same type.
341  // In principle, this should be allowed and the signal should be considered
342  // as interferer.
343  // CASE 1: BS is transmitting and another BS is receiving
344  // No need to do this check, the packet will be discarded since the cell ids
345  // are different
346  // CASE 2: UE is transmitting and another UE is receiving, both attached to
347  // the same cell
348  //
349 
350  Ptr<MmWaveEnbNetDevice> enbTx = DynamicCast<MmWaveEnbNetDevice> (params->txPhy->GetDevice ());
351  Ptr<MmWaveEnbNetDevice> enbRx = DynamicCast<MmWaveEnbNetDevice> (GetDevice ());
352  if ((enbTx != 0 && enbRx != 0) || (enbTx == 0 && enbRx == 0))
353  {
354  NS_LOG_INFO ("BS to BS or UE to UE transmission neglected.");
355  return;
356  }
357 
358  // check if the received signal is mmWave DATA or CTRL
359  Ptr<MmwaveSpectrumSignalParametersDataFrame> mmwaveDataRxParams = DynamicCast<MmwaveSpectrumSignalParametersDataFrame> (params);
360  Ptr<MmWaveSpectrumSignalParametersDlCtrlFrame> mmwaveDlCtrlRxParams = DynamicCast<MmWaveSpectrumSignalParametersDlCtrlFrame> (params);
361 
362  if (mmwaveDataRxParams != 0)
363  {
364  // mmWave DATA case
365 
366  // TODO in case our device is a UE, the code below checks if the UePhy is
367  // allowed to receive a signal.
368  // This should be done inside the UePhy class as a state machine
369 
370  bool isAllocated = true;
371  Ptr<MmWaveUeNetDevice> ueRx = DynamicCast<MmWaveUeNetDevice> (GetDevice ());
372  Ptr<McUeNetDevice> rxMcUe = DynamicCast<McUeNetDevice> (GetDevice ());
373 
374 
375  if ((ueRx != 0) && (ueRx->GetPhy (m_componentCarrierId)->IsReceptionEnabled () == false))
376  { // if the first cast is 0 (the device is MC) then this if will not be executed
377  isAllocated = false;
378  }
379  else if ((rxMcUe != 0) && (rxMcUe->GetMmWavePhy (m_componentCarrierId)->IsReceptionEnabled () == false))
380  { // this is executed if the device is MC and is transmitting
381  isAllocated = false;
382  }
383 
384  NS_LOG_DEBUG("Now: " << Simulator::Now().GetSeconds() << " enb? " << (enbRx != 0)
385  << " ue? " << (ueRx != 0) << " "
386  << isAllocated);
387 
388  if (isAllocated)
389  {
390 
391  // add the signal to the sum of all the signals received in this time instant
392  m_interferenceData->AddSignal (mmwaveDataRxParams->psd, mmwaveDataRxParams->duration);
393 
394  // check if the signal comes from a device connected to this cell
395  if (mmwaveDataRxParams->cellId == m_cellId)
396  {
397  StartRxData (mmwaveDataRxParams);
398  }
399  else
400  {
401  NS_LOG_LOGIC (this << " not in sync with this signal (cellId=" << mmwaveDataRxParams->cellId << ", m_cellId=" << m_cellId << ")");
402  }
403  }
404  }
405  else if (mmwaveDlCtrlRxParams != 0)
406  {
407  // for CTRL messages interference is not considered
408  StartRxCtrl (mmwaveDlCtrlRxParams);
409  }
410  else
411  {
412  // other type of signal
413  m_interferenceData->AddSignal (params->psd, params->duration);
414  }
415 }

417 void
418 MmWaveSpectrumPhy::StartRxData (Ptr<MmwaveSpectrumSignalParametersDataFrame> params)
419 {
420 
421  NS_LOG_FUNCTION (this);
422 
423  switch (m_state)
424  {
425  case TX:
426  NS_FATAL_ERROR ("Cannot receive while transmitting");
427  break;
428 
429  case RX_CTRL:
430  NS_FATAL_ERROR ("Cannot receive control in data period");
431  break;
432 
433  case RX_DATA:
434  case IDLE:
435  {
436  // this is a useful signal
437  m_interferenceData->StartRx (params->psd);
438 
439  if (m_rxPacketBurstList.empty ())
440  {
441  NS_ASSERT (m_state == IDLE);
442  // first transmission, i.e., we're IDLE and we start RX
443  m_firstRxStart = Simulator::Now ();
444  m_firstRxDuration = params->duration;
445  NS_LOG_LOGIC (this << " scheduling EndRx with delay " << params->duration.GetSeconds () << "s");
446 
447  m_endRxDataEvent = Simulator::Schedule (params->duration, &MmWaveSpectrumPhy::EndRxData, this);
448 
449  ChangeState (RX_DATA);
450  }
451  else
452  {
453  NS_ASSERT (m_state == RX_DATA);
454  // sanity check: if there are multiple RX events, they
455  // should occur at the same time and have the same
456  // duration, otherwise the interference calculation
457  // won't be correct
458  NS_ASSERT ((m_firstRxStart == Simulator::Now ()) && (m_firstRxDuration == params->duration));
459  }
460 
461 
462  if (params->packetBurst && !params->packetBurst->GetPackets ().empty ())
463  {
464  m_rxPacketBurstList.push_back (params->packetBurst);
465  }
466 
467  m_rxControlMessageList.insert (m_rxControlMessageList.end (), params->ctrlMsgList.begin (), params->ctrlMsgList.end ());
468 
469  NS_LOG_LOGIC (this << " numSimultaneousRxEvents = " << m_rxPacketBurstList.size ());
470  }
471  break;
472 
473  default:
474  NS_FATAL_ERROR ("Programming Error: Unknown State");
475  }
476 }

478 void
479 MmWaveSpectrumPhy::StartRxCtrl (Ptr<MmWaveSpectrumSignalParametersDlCtrlFrame> dlCtrlRxParams)
480 {
481  NS_LOG_FUNCTION (this);
482  // RDF: method currently supports Downlink control only! // TODO is this true??
483  switch (m_state)
484  {
485  case TX:
486  NS_FATAL_ERROR ("Cannot RX while TX: according to FDD channel access, the physical layer for transmission cannot be used for reception");
487  break;
488 
489  case RX_DATA:
490  NS_FATAL_ERROR ("Cannot RX data while receiving control");
491  break;
492 
493  case RX_CTRL:
494  {
495  // To check if we're synchronized to this signal, we check for the CellId
496  if (dlCtrlRxParams->cellId == m_cellId)
497  {
498  NS_LOG_LOGIC (this << " synchronized with this signal (cellId=" << m_cellId << ")");
499 
500  Ptr<MmWaveUeNetDevice> ueRx = DynamicCast<MmWaveUeNetDevice> (GetDevice ());
501  Ptr<McUeNetDevice> rxMcUe = DynamicCast<McUeNetDevice> (GetDevice ());
502  if (ueRx || rxMcUe)
503  {
504  NS_FATAL_ERROR ("UE already receiving control data from serving cell");
505  }
506  NS_ASSERT ((m_firstRxStart == Simulator::Now ()) && (m_firstRxDuration == dlCtrlRxParams->duration));
507 
508  m_rxControlMessageList.insert (m_rxControlMessageList.end (), dlCtrlRxParams->ctrlMsgList.begin (), dlCtrlRxParams->ctrlMsgList.end ());
509  }
510  else
511  {
512  NS_LOG_LOGIC (this << " not in sync with this signal (cellId=" << dlCtrlRxParams->cellId << ", m_cellId=" << m_cellId << ")");
513  }
514  break;
515  }
516 
517  case IDLE:
518  {
519  // the behavior is similar when we're IDLE or RX because we can receive more signals
520  // simultaneously (e.g., at the eNB).
521 
522  // To check if we're synchronized to this signal, we check for the CellId
523  if (dlCtrlRxParams->cellId == m_cellId)
524  {
525  NS_LOG_LOGIC (this << " synchronized with this signal (cellId=" << m_cellId << ")");
526 
527  // first transmission, i.e., we're IDLE and we start RX
528  NS_ASSERT (m_rxControlMessageList.empty ());
529  m_firstRxStart = Simulator::Now ();
530  m_firstRxDuration = dlCtrlRxParams->duration;
531  NS_LOG_LOGIC (this << " scheduling EndRx with delay " << dlCtrlRxParams->duration);
532 
533  // store the DCIs
534  m_rxControlMessageList = dlCtrlRxParams->ctrlMsgList;
535  m_endRxDlCtrlEvent = Simulator::Schedule (dlCtrlRxParams->duration, &MmWaveSpectrumPhy::EndRxCtrl, this);
536  ChangeState (RX_CTRL);
537  }
538  else
539  {
540  NS_LOG_LOGIC (this << " not in sync with this signal (cellId=" << dlCtrlRxParams->cellId << ", m_cellId=" << m_cellId << ")");
541  }
542  break;
543  }
544 
545  default:
546  {
547  NS_FATAL_ERROR ("unknown state");
548  break;
549  }
550  }

553 void
554 MmWaveSpectrumPhy::EndRxData ()
555 {
556  NS_ASSERT (m_state = RX_DATA);
557 
558  m_interferenceData->EndRx (); // trigger the SINR computation
559 
560  // compute the average SINR
561  double sinrAvg = Sum (m_sinrPerceived) / (m_sinrPerceived.GetSpectrumModel ()->GetNumBands ());
562  NS_LOG_DEBUG ("m_sinrPerceived=" << m_sinrPerceived << ", sinrAvg=" << sinrAvg << ", SINR dB=" << 10*std::log10(sinrAvg) << ", GetNumBands=" << m_sinrPerceived.GetSpectrumModel ()->GetNumBands ());
563 
564  // check if the transmissions succeeded or failed
565  ExpectedTbMap_t::iterator itTb = m_expectedTbs.begin ();
566  while (itTb != m_expectedTbs.end ())
567  {
568  if ((m_dataErrorModelEnabled) && (m_rxPacketBurstList.size () > 0))
569  {
570  MmWaveHarqProcessInfoList_t harqInfoList;
571  uint8_t rv = 0;
572  if (itTb->second.ndi == 0)
573  {
574  // TB retxed: retrieve HARQ history
575  if (itTb->second.downlink)
576  {
577  harqInfoList = m_harqPhyModule->GetHarqProcessInfoDl (itTb->first, itTb->second.harqProcessId);
578  }
579  else
580  {
581  harqInfoList = m_harqPhyModule->GetHarqProcessInfoUl (itTb->first, itTb->second.harqProcessId);
582  }
583  if (harqInfoList.size () > 0)
584  {
585  rv = harqInfoList.back ().m_rv;
586  }
587  }
588 
589  MmWaveTbStats_t tbStats = MmWaveMiErrorModel::GetTbDecodificationStats (m_sinrPerceived, itTb->second.rbBitmap, itTb->second.size, itTb->second.mcs, harqInfoList);
590  itTb->second.tbler = tbStats.tbler;
591  itTb->second.mi = tbStats.miTotal;
592  itTb->second.corrupt = m_random->GetValue () > tbStats.tbler ? false : true;
593  if (itTb->second.corrupt)
594  {
595  NS_LOG_INFO (this << " RNTI " << itTb->first << " size " << itTb->second.size << " mcs " << (uint32_t)itTb->second.mcs << " bitmap " << itTb->second.rbBitmap.size () << " rv " << rv << " TBLER " << tbStats.tbler << " corrupted " << itTb->second.corrupt);
596  }
597  }
598  itTb++;
599  }
600 
601  // fire the traces and send the ACKs/NACKs
602  std::map <uint16_t, DlHarqInfo> harqDlInfoMap;
603  for (std::list<Ptr<PacketBurst> >::const_iterator i = m_rxPacketBurstList.begin (); i != m_rxPacketBurstList.end (); ++i)
604  {
605  for (std::list<Ptr<Packet> >::const_iterator j = (*i)->Begin (); j != (*i)->End (); ++j)
606  {
607  if ((*j)->GetSize () == 0)
608  {
609  continue;
610  }
611 
612  LteRadioBearerTag bearerTag;
613  if ((*j)->PeekPacketTag (bearerTag) == false)
614  {
615  NS_FATAL_ERROR ("No radio bearer tag found");
616  }
617  uint16_t rnti = bearerTag.GetRnti ();
618  itTb = m_expectedTbs.find (rnti);
619  if (itTb != m_expectedTbs.end ())
620  {
621  if (!itTb->second.corrupt)
622  {
623  m_phyRxDataEndOkCallback (*j);
624  }
625  else
626  {
627  NS_LOG_INFO ("TB failed");
628  }
629 
630  MmWaveMacPduTag pduTag;
631  if ((*j)->PeekPacketTag (pduTag) == false)
632  {
633  NS_FATAL_ERROR ("No radio bearer tag found");
634  }
635 
636  RxPacketTraceParams traceParams;
637  traceParams.m_tbSize = itTb->second.size;
638  traceParams.m_cellId = m_cellId;
639  traceParams.m_frameNum = pduTag.GetSfn ().m_frameNum;
640  traceParams.m_sfNum = pduTag.GetSfn ().m_sfNum;
641  traceParams.m_slotNum = pduTag.GetSfn ().m_slotNum;
642  traceParams.m_rnti = rnti;
643  traceParams.m_mcs = itTb->second.mcs;
644  traceParams.m_rv = itTb->second.rv;
645  traceParams.m_sinr = sinrAvg;
646  traceParams.m_sinrMin = itTb->second.mi; // TODO why this field does not carry sinrMin?
