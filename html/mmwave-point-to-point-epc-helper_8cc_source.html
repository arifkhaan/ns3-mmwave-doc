71 MmWavePointToPointEpcHelper::MmWavePointToPointEpcHelper ()
72  : m_gtpuUdpPort (2152),
73  // fixed by the standard
74  m_s1apUdpPort (36412)
75 {
76  NS_LOG_FUNCTION (this);
77 
78  // since we use point-to-point links for all S1-U and S1-AP links,
79  // we use a /30 subnet which can hold exactly two addresses
80  // (remember that net broadcast and null address are not valid)
81  m_s1uIpv4AddressHelper.SetBase ("10.0.0.0", "255.255.255.252");
82  m_s1apIpv4AddressHelper.SetBase ("11.0.0.0", "255.255.255.252");
83  m_x2Ipv4AddressHelper.SetBase ("12.0.0.0", "255.255.255.252");
84 
85  // we use a /8 net for all UEs
86  m_uePgwAddressHelper.SetBase ("7.0.0.0", "255.0.0.0");
87 
88  // we use a /64 IPv6 net all UEs
89  m_uePgwAddressHelper6.SetBase ("7777:f00d::", Ipv6Prefix (64));
90 
91  // create SgwPgwNode
92  m_sgwPgw = CreateObject<Node> ();
93  InternetStackHelper internet;
94  internet.Install (m_sgwPgw);
95 
96  // create MmeNode
97  m_mmeNode = CreateObject<Node> ();
98  internet.Install (m_mmeNode);
99 
100  // create S1-U socket
101  // The Tun device resides in different 64 bit subnet.
102  // We must create an unique route to tun device for all the packets destined
103  // to all 64 bit IPv6 prefixes of UEs, based by the unique 48 bit network
104  // prefix of this EPC network
105  Ipv6StaticRoutingHelper ipv6RoutingHelper;
106  Ptr<Ipv6StaticRouting> pgwStaticRouting = ipv6RoutingHelper.GetStaticRouting (m_sgwPgw->GetObject<Ipv6> ());
107  pgwStaticRouting->AddNetworkRouteTo ("7777:f00d::", Ipv6Prefix (64), Ipv6Address ("::"), 1, 0);
108 
109  // create S1-U socket for SgwPgwNode
110  Ptr<Socket> sgwPgwS1uSocket = Socket::CreateSocket (m_sgwPgw, TypeId::LookupByName ("ns3::UdpSocketFactory"));
111  int retval = sgwPgwS1uSocket->Bind (InetSocketAddress (Ipv4Address::GetAny (), m_gtpuUdpPort));
112  NS_ASSERT (retval == 0);
113 
114  // create S1-AP socket for MmeNode
115  Ptr<Socket> mmeS1apSocket = Socket::CreateSocket (m_mmeNode, TypeId::LookupByName ("ns3::UdpSocketFactory"));
116  retval = mmeS1apSocket->Bind (InetSocketAddress (Ipv4Address::GetAny (), m_s1apUdpPort)); // it listens on any IP, port m_s1apUdpPort
117  NS_ASSERT (retval == 0);
118 
119  // create TUN device implementing tunneling of user data over GTP-U/UDP/IP
120  m_tunDevice = CreateObject<VirtualNetDevice> ();
121  // allow jumbo packets
122  m_tunDevice->SetAttribute ("Mtu", UintegerValue (30000));
123 
124  // yes we need this
125  m_tunDevice->SetAddress (Mac48Address::Allocate ());
126 
127  m_sgwPgw->AddDevice (m_tunDevice);
128  NetDeviceContainer tunDeviceContainer;
129  tunDeviceContainer.Add (m_tunDevice);
130 
131  // the TUN device is on the same subnet as the UEs, so when a packet
132  // addressed to an UE arrives at the intenet to the WAN interface of
133  // the PGW it will be forwarded to the TUN device.
134  Ipv4InterfaceContainer tunDeviceIpv4IfContainer = AssignUeIpv4Address (tunDeviceContainer);
135 
136  // the TUN device for IPv6 address is on the different subnet as the
137  // UEs, it will forward the UE packets as we have inserted the route
138  // for all UEs at the time of assigning UE addresses
139  Ipv6InterfaceContainer tunDeviceIpv6IfContainer = AssignUeIpv6Address (tunDeviceContainer);
140 
141  //Set Forwarding of the IPv6 interface
142  tunDeviceIpv6IfContainer.SetForwarding (0,true);
143  tunDeviceIpv6IfContainer.SetDefaultRouteInAllNodes (0);
144 
145  // create EpcSgwPgwApplication
146  m_sgwPgwApp = CreateObject<EpcSgwPgwApplication> (m_tunDevice, sgwPgwS1uSocket);
147  m_sgwPgw->AddApplication (m_sgwPgwApp);
148 
149  // connect SgwPgwApplication and virtual net device for tunneling
150  m_tunDevice->SetSendCallback (MakeCallback (&EpcSgwPgwApplication::RecvFromTunDevice, m_sgwPgwApp));
151 
152  // create S1apMme object and aggregate it with the m_mmeNode
153  Ptr<EpcS1apMme> s1apMme = CreateObject<EpcS1apMme> (mmeS1apSocket, 1); // for now, only one mme!
154  m_mmeNode->AggregateObject (s1apMme);
155 
156  // create EpcMmeApplication and connect with SGW via S11 interface
157  m_mmeApp = CreateObject<EpcMmeApplication> ();
158  m_mmeNode->AddApplication (m_mmeApp);
159  m_mmeApp->SetS11SapSgw (m_sgwPgwApp->GetS11SapSgw ());
160  m_sgwPgwApp->SetS11SapMme (m_mmeApp->GetS11SapMme ());
161  // connect m_mmeApp to the s1apMme
162  m_mmeApp->SetS1apSapMmeProvider (s1apMme->GetEpcS1apSapMmeProvider ());
163  s1apMme->SetEpcS1apSapMmeUser (m_mmeApp->GetS1apSapMme ());
164 }

238 void
239 MmWavePointToPointEpcHelper::AddEnb (Ptr<Node> enb, Ptr<NetDevice> lteEnbNetDevice, uint16_t cellId)
240 {
241  NS_LOG_FUNCTION (this << enb << lteEnbNetDevice << cellId);
242 
243  NS_ASSERT (enb == lteEnbNetDevice->GetNode ());
244 
245  // add an IPv4 stack to the previously created eNB
246  InternetStackHelper internet;
247  internet.Install (enb);
248  NS_LOG_LOGIC ("number of Ipv4 ifaces of the eNB after node creation: " << enb->GetObject<Ipv4> ()->GetNInterfaces ());
249 
250  // create a point to point link between the new eNB and the SGW with
251  // the corresponding new NetDevices on each side
252  NodeContainer enbSgwNodes;
253  enbSgwNodes.Add (m_sgwPgw);
254  enbSgwNodes.Add (enb);
255  PointToPointHelper p2ph;
256  p2ph.SetDeviceAttribute ("DataRate", DataRateValue (m_s1uLinkDataRate));
257  p2ph.SetDeviceAttribute ("Mtu", UintegerValue (m_s1uLinkMtu));
258  p2ph.SetChannelAttribute ("Delay", TimeValue (m_s1uLinkDelay));
259  NetDeviceContainer enbSgwDevices = p2ph.Install (enb, m_sgwPgw);
260  NS_LOG_LOGIC ("number of Ipv4 ifaces of the eNB after installing p2p dev: " << enb->GetObject<Ipv4> ()->GetNInterfaces ());
261  Ptr<NetDevice> enbDev = enbSgwDevices.Get (0);
262  Ptr<NetDevice> sgwDev = enbSgwDevices.Get (1);
263  m_s1uIpv4AddressHelper.NewNetwork ();
264  Ipv4InterfaceContainer enbSgwIpIfaces = m_s1uIpv4AddressHelper.Assign (enbSgwDevices);
265  NS_LOG_LOGIC ("number of Ipv4 ifaces of the eNB after assigning Ipv4 addr to S1 dev: " << enb->GetObject<Ipv4> ()->GetNInterfaces ());
266 
267  Ipv4Address enbAddress = enbSgwIpIfaces.GetAddress (0);
268  Ipv4Address sgwAddress = enbSgwIpIfaces.GetAddress (1);
269 
270  // create S1-U socket for the ENB
271  Ptr<Socket> enbS1uSocket = Socket::CreateSocket (enb, TypeId::LookupByName ("ns3::UdpSocketFactory"));
272  int retval = enbS1uSocket->Bind (InetSocketAddress (enbAddress, m_gtpuUdpPort));
273  NS_ASSERT (retval == 0);
274 
275  // create a point to point link between the new eNB and the MME with
276  // the corresponding new NetDevices on each side
277  NodeContainer enbMmeNodes;
278  enbMmeNodes.Add (m_mmeNode);
279  enbMmeNodes.Add (enb);
280  PointToPointHelper p2ph_mme;
281  p2ph_mme.SetDeviceAttribute ("DataRate", DataRateValue (m_s1apLinkDataRate));
282  p2ph_mme.SetDeviceAttribute ("Mtu", UintegerValue (m_s1apLinkMtu));
283  p2ph_mme.SetChannelAttribute ("Delay", TimeValue (m_s1apLinkDelay));
284  NetDeviceContainer enbMmeDevices = p2ph_mme.Install (enb, m_mmeNode);
285  NS_LOG_LOGIC ("number of Ipv4 ifaces of the eNB after installing p2p dev: " << enb->GetObject<Ipv4> ()->GetNInterfaces ());
286 
287  m_s1apIpv4AddressHelper.NewNetwork ();
288  Ipv4InterfaceContainer enbMmeIpIfaces = m_s1apIpv4AddressHelper.Assign (enbMmeDevices);
289  NS_LOG_LOGIC ("number of Ipv4 ifaces of the eNB after assigning Ipv4 addr to S1 dev: " << enb->GetObject<Ipv4> ()->GetNInterfaces ());
290 
291  Ipv4Address mme_enbAddress = enbMmeIpIfaces.GetAddress (0);
292  Ipv4Address mmeAddress = enbMmeIpIfaces.GetAddress (1);
293 
294  // create S1-AP socket for the ENB
295  Ptr<Socket> enbS1apSocket = Socket::CreateSocket (enb, TypeId::LookupByName ("ns3::UdpSocketFactory"));
296  retval = enbS1apSocket->Bind (InetSocketAddress (mme_enbAddress, m_s1apUdpPort));
297  NS_ASSERT (retval == 0);
298 
299  // give PacketSocket powers to the eNB
300  //PacketSocketHelper packetSocket;
301  //packetSocket.Install (enb);
302 
303  // create LTE socket for the ENB
304  Ptr<Socket> enbLteSocket = Socket::CreateSocket (enb, TypeId::LookupByName ("ns3::PacketSocketFactory"));
305  PacketSocketAddress enbLteSocketBindAddress;
306  enbLteSocketBindAddress.SetSingleDevice (lteEnbNetDevice->GetIfIndex ());
307  enbLteSocketBindAddress.SetProtocol (Ipv4L3Protocol::PROT_NUMBER);
308  retval = enbLteSocket->Bind (enbLteSocketBindAddress);
309  NS_ASSERT (retval == 0);
310  PacketSocketAddress enbLteSocketConnectAddress;
311  enbLteSocketConnectAddress.SetPhysicalAddress (Mac48Address::GetBroadcast ());
312  enbLteSocketConnectAddress.SetSingleDevice (lteEnbNetDevice->GetIfIndex ());
313  enbLteSocketConnectAddress.SetProtocol (Ipv4L3Protocol::PROT_NUMBER);
314  retval = enbLteSocket->Connect (enbLteSocketConnectAddress);
315  NS_ASSERT (retval == 0);
316 
317  // create LTE socket for the ENB
318  Ptr<Socket> enbLteSocket6 = Socket::CreateSocket (enb, TypeId::LookupByName ("ns3::PacketSocketFactory"));
319  PacketSocketAddress enbLteSocketBindAddress6;
320  enbLteSocketBindAddress6.SetSingleDevice (lteEnbNetDevice->GetIfIndex ());
321  enbLteSocketBindAddress6.SetProtocol (Ipv6L3Protocol::PROT_NUMBER);
322  retval = enbLteSocket6->Bind (enbLteSocketBindAddress6);
323  NS_ASSERT (retval == 0);
324  PacketSocketAddress enbLteSocketConnectAddress6;
325  enbLteSocketConnectAddress6.SetPhysicalAddress (Mac48Address::GetBroadcast ());
326  enbLteSocketConnectAddress6.SetSingleDevice (lteEnbNetDevice->GetIfIndex ());
327  enbLteSocketConnectAddress6.SetProtocol (Ipv6L3Protocol::PROT_NUMBER);
328  retval = enbLteSocket6->Connect (enbLteSocketConnectAddress6);
329  NS_ASSERT (retval == 0);
330 
331  NS_LOG_INFO ("create EpcEnbApplication");
332  Ptr<EpcEnbApplication> enbApp = CreateObject<EpcEnbApplication> (enbLteSocket, enbLteSocket6, enbS1uSocket, enbAddress, sgwAddress, cellId);
333  enb->AddApplication (enbApp);
334  NS_ASSERT (enb->GetNApplications () == 1);
335  NS_ASSERT_MSG (enb->GetApplication (0)->GetObject<EpcEnbApplication> () != 0, "cannot retrieve EpcEnbApplication");
336  NS_LOG_LOGIC ("enb: " << enb << ", enb->GetApplication (0): " << enb->GetApplication (0));
337 
338 
339  NS_LOG_INFO ("Create EpcX2 entity");
340  Ptr<EpcX2> x2 = CreateObject<EpcX2> ();
341  enb->AggregateObject (x2);
342 
343  NS_LOG_INFO ("connect S1-AP interface");
344 
345  uint16_t mmeId = 1;
346  Ptr<EpcS1apEnb> s1apEnb = CreateObject<EpcS1apEnb> (enbS1apSocket, mme_enbAddress, mmeAddress, cellId, mmeId); // only one mme!
347  enb->AggregateObject (s1apEnb);
348  enbApp->SetS1apSapMme (s1apEnb->GetEpcS1apSapEnbProvider ());
349  s1apEnb->SetEpcS1apSapEnbUser (enbApp->GetS1apSapEnb ());
350  m_mmeApp->AddEnb (cellId, mme_enbAddress); // TODO consider if this can be removed
351  // add the interface to the S1AP endpoint on the MME
352  Ptr<EpcS1apMme> s1apMme = m_mmeNode->GetObject<EpcS1apMme> ();
353  s1apMme->AddS1apInterface (cellId, mme_enbAddress);
354 
355  m_sgwPgwApp->AddEnb (cellId, enbAddress, sgwAddress);
356 }

359 void
360 MmWavePointToPointEpcHelper::AddX2Interface (Ptr<Node> enb1, Ptr<Node> enb2)
361 {
362  NS_LOG_FUNCTION (this << enb1 << enb2);
363 
364  // Create a point to point link between the two eNBs with
365  // the corresponding new NetDevices on each side
366  NodeContainer enbNodes;
367  enbNodes.Add (enb1);
368  enbNodes.Add (enb2);
369  PointToPointHelper p2ph;
370  p2ph.SetDeviceAttribute ("DataRate", DataRateValue (m_x2LinkDataRate));
371  p2ph.SetDeviceAttribute ("Mtu", UintegerValue (m_x2LinkMtu));
372  p2ph.SetChannelAttribute ("Delay", TimeValue (m_x2LinkDelay));
373  // in case the RLC buffer is full, the forwarding of packets from the source eNB to the target eNB during a handover
374  // or a switch may overflow the transmission queue of point to point devices. Therefore the following line increases
375  // the size of tx queue in p2p devices. The parameter should be related to the maximum size of the RLC buffer.
376  p2ph.SetQueue ("ns3::DropTailQueue", "MaxSize", QueueSizeValue (QueueSize ("4294967295p")));
377  NetDeviceContainer enbDevices = p2ph.Install (enb1, enb2);
378  NS_LOG_LOGIC ("number of Ipv4 ifaces of the eNB #1 after installing p2p dev: " << enb1->GetObject<Ipv4> ()->GetNInterfaces ());
379  NS_LOG_LOGIC ("number of Ipv4 ifaces of the eNB #2 after installing p2p dev: " << enb2->GetObject<Ipv4> ()->GetNInterfaces ());
380  Ptr<NetDevice> enb1Dev = enbDevices.Get (0);
381  Ptr<NetDevice> enb2Dev = enbDevices.Get (1);
382 
383  m_x2Ipv4AddressHelper.NewNetwork ();
384  Ipv4InterfaceContainer enbIpIfaces = m_x2Ipv4AddressHelper.Assign (enbDevices);
385  NS_LOG_LOGIC ("number of Ipv4 ifaces of the eNB #1 after assigning Ipv4 addr to X2 dev: " << enb1->GetObject<Ipv4> ()->GetNInterfaces ());
386  NS_LOG_LOGIC ("number of Ipv4 ifaces of the eNB #2 after assigning Ipv4 addr to X2 dev: " << enb2->GetObject<Ipv4> ()->GetNInterfaces ());
387 
388  Ipv4Address enb1X2Address = enbIpIfaces.GetAddress (0);
389  Ipv4Address enb2X2Address = enbIpIfaces.GetAddress (1);
390 
391  // Add X2 interface to both eNBs' X2 entities
392  Ptr<EpcX2> enb1X2 = enb1->GetObject<EpcX2> ();
393  Ptr<MmWaveEnbNetDevice> enb1MmWaveDev = enb1->GetDevice (0)->GetObject<MmWaveEnbNetDevice> ();
394  Ptr<LteEnbNetDevice> enb1LteDev = enb1->GetDevice (0)->GetObject<LteEnbNetDevice> ();
395  // we may have a LTE or a MmWave eNB
396  uint16_t enb1CellId = 0;
397  if (enb1MmWaveDev != 0)
398  {
399  enb1CellId = enb1MmWaveDev->GetCellId ();
400  NS_LOG_INFO ("MmWaveEnbNetDevice #1 = " << enb1MmWaveDev << " - CellId = " << enb1CellId);
401  }
402  else if (enb1LteDev != 0)
403  {
404  enb1CellId = enb1LteDev->GetCellId ();
405  NS_LOG_INFO ("LteEnbNetDevice #1 = " << enb1LteDev << " - CellId = " << enb1CellId);
406  }
407 
408  Ptr<EpcX2> enb2X2 = enb2->GetObject<EpcX2> ();
409  Ptr<MmWaveEnbNetDevice> enb2MmWaveDev = enb2->GetDevice (0)->GetObject<MmWaveEnbNetDevice> ();
410  Ptr<LteEnbNetDevice> enb2LteDev = enb2->GetDevice (0)->GetObject<LteEnbNetDevice> ();
411  // we may have a LTE or a MmWave eNB
412  uint16_t enb2CellId = 0;
413  if (enb2MmWaveDev != 0)
414  {
415  enb2CellId = enb2MmWaveDev->GetCellId ();
416  NS_LOG_INFO ("MmWaveEnbNetDevice #2 = " << enb2MmWaveDev << " - CellId = " << enb2CellId);
417  enb2MmWaveDev->GetRrc ()->AddX2Neighbour (enb1CellId);
418  }
419  else if (enb2LteDev != 0)
420  {
421  enb2CellId = enb2LteDev->GetCellId ();
422  NS_LOG_INFO ("LteEnbNetDevice #2 = " << enb2LteDev << " - CellId = " << enb2CellId);
423  enb2LteDev->GetRrc ()->AddX2Neighbour (enb1CellId);
424  }
425 
426  enb1X2->AddX2Interface (enb1CellId, enb1X2Address, enb2CellId, enb2X2Address);
427  enb2X2->AddX2Interface (enb2CellId, enb2X2Address, enb1CellId, enb1X2Address);

458 MmWavePointToPointEpcHelper::ActivateEpsBearer (Ptr<NetDevice> ueDevice, uint64_t imsi, Ptr<EpcTft> tft, EpsBearer bearer)
459 {
460  NS_LOG_FUNCTION (this << ueDevice << imsi);
461 
462  // we now retrieve the IPv4/IPv6 address of the UE and notify it to the SGW;
463  // we couldn't do it before since address assignment is triggered by
464  // the user simulation program, rather than done by the EPC
465  Ptr<Node> ueNode = ueDevice->GetNode ();
466  Ptr<Ipv4> ueIpv4 = ueNode->GetObject<Ipv4> ();
467 
468  Ptr<Ipv6> ueIpv6 = ueNode->GetObject<Ipv6> ();
469  NS_ASSERT_MSG (ueIpv4 != 0 || ueIpv6 != 0, "UEs need to have IPv4/IPv6 installed before EPS bearers can be activated");
470  if (ueIpv4)
471  {
472  int32_t interface = ueIpv4->GetInterfaceForDevice (ueDevice);
473  if (interface >= 0 && ueIpv4->GetNAddresses (interface) == 1)
474  {
475  Ipv4Address ueAddr = ueIpv4->GetAddress (interface, 0).GetLocal ();
476  NS_LOG_LOGIC (" UE IP address: " << ueAddr);
477  m_sgwPgwApp->SetUeAddress (imsi, ueAddr);
478  }
479  }
480  if (ueIpv6)
481  {
482  int32_t interface6 = ueIpv6->GetInterfaceForDevice (ueDevice);
483  if (interface6 >= 0 && ueIpv6->GetNAddresses (interface6) == 2)
484  {
485  Ipv6Address ueAddr6 = ueIpv6->GetAddress (interface6, 1).GetAddress ();
486  NS_LOG_LOGIC (" UE IPv6 address: " << ueAddr6);
487  m_sgwPgwApp->SetUeAddress6 (imsi, ueAddr6);
488  }
489  }
490 
491  uint8_t bearerId = m_mmeApp->AddBearer (imsi, tft, bearer);
492  Ptr<mmwave::MmWaveUeNetDevice> ueLteDevice = ueDevice->GetObject<mmwave::MmWaveUeNetDevice> ();
493  if (ueLteDevice)
494  {
495  Simulator::ScheduleNow (&EpcUeNas::ActivateEpsBearer, ueLteDevice->GetNas (), bearer, tft);
496  }
497  NS_LOG_LOGIC ("Bearer Id added in mmeApp " << bearerId);
498  return bearerId;
499 }
