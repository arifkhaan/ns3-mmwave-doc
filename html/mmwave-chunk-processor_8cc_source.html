68 mmWaveChunkProcessor::EvaluateChunk (const SpectrumValue& sinr, Time duration)
69 {
70  NS_LOG_FUNCTION (this << sinr << duration);
71  if (m_sumValues == 0)
72  {
73  m_sumValues = Create<SpectrumValue> (sinr.GetSpectrumModel ());
74  }
75  (*m_sumValues) += sinr * duration.GetSeconds ();
76  m_totDuration += duration;
