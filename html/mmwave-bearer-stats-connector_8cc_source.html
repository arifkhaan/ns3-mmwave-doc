293 {

297  Config::Connect ("/NodeList/*/DeviceList/*/LteEnbRrc/NewUeContext",
298  MakeBoundCallback (&MmWaveBearerStatsConnector::NotifyNewUeContextEnb, this));
299  Config::Connect ("/NodeList/*/DeviceList/*/LteUeRrc/RandomAccessSuccessful",
300  MakeBoundCallback (&MmWaveBearerStatsConnector::NotifyRandomAccessSuccessfulUe, this));
301  Config::Connect ("/NodeList/*/DeviceList/*/MmWaveUeRrc/RandomAccessSuccessful",
302  MakeBoundCallback (&MmWaveBearerStatsConnector::NotifyRandomAccessSuccessfulUe, this));
303  Config::Connect ("/NodeList/*/DeviceList/*/LteEnbRrc/ConnectionReconfiguration",
304  MakeBoundCallback (&MmWaveBearerStatsConnector::NotifyConnectionReconfigurationEnb, this));
305  Config::Connect ("/NodeList/*/DeviceList/*/LteUeRrc/ConnectionReconfiguration",
306  MakeBoundCallback (&MmWaveBearerStatsConnector::NotifyConnectionReconfigurationUe, this));
307  Config::Connect ("/NodeList/*/DeviceList/*/MmWaveUeRrc/ConnectionReconfiguration",
308  MakeBoundCallback (&MmWaveBearerStatsConnector::NotifyConnectionReconfigurationUe, this));
309  Config::Connect ("/NodeList/*/DeviceList/*/LteEnbRrc/HandoverStart",
310  MakeBoundCallback (&MmWaveBearerStatsConnector::NotifyHandoverStartEnb, this));
311  Config::Connect ("/NodeList/*/DeviceList/*/LteUeRrc/HandoverStart",
312  MakeBoundCallback (&MmWaveBearerStatsConnector::NotifyHandoverStartUe, this));
313  Config::Connect ("/NodeList/*/DeviceList/*/MmWaveUeRrc/HandoverStart",
314  MakeBoundCallback (&MmWaveBearerStatsConnector::NotifyHandoverStartUe, this));
315  Config::Connect ("/NodeList/*/DeviceList/*/LteEnbRrc/HandoverEndOk",
316  MakeBoundCallback (&MmWaveBearerStatsConnector::NotifyHandoverEndOkEnb, this));
317  Config::Connect ("/NodeList/*/DeviceList/*/LteUeRrc/HandoverEndOk",
318  MakeBoundCallback (&MmWaveBearerStatsConnector::NotifyHandoverEndOkUe, this));
319  Config::Connect ("/NodeList/*/DeviceList/*/MmWaveUeRrc/HandoverEndOk",
320  MakeBoundCallback (&MmWaveBearerStatsConnector::NotifyHandoverEndOkUe, this));
321  Config::Connect ("/NodeList/*/DeviceList/*/LteUeRrc/SwitchToMmWave",
322  MakeBoundCallback (&MmWaveBearerStatsConnector::NotifySwitchToMmWaveUe, this));
323  // mmWave SINR from RT, LTE SINR from the PHY callbacks

325  MakeBoundCallback (&MmWaveBearerStatsConnector::NotifyMmWaveSinr, this));
326  Config::Connect ("/NodeList/*/DeviceList/*/LteUePhy/ReportCurrentCellRsrpSinr",

329  }
