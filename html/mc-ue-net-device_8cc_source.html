144 void
145 McUeNetDevice::DoInitialize (void)
146 {
147  NS_LOG_FUNCTION (this);
148  m_isConstructed = true;
149  UpdateConfig ();
150 
151  std::map< uint8_t, Ptr<ComponentCarrierUe> >::iterator lteIt;
152  for (lteIt = m_lteCcMap.begin (); lteIt != m_lteCcMap.end (); ++lteIt)
153  {
154  lteIt->second->GetPhy ()->Initialize ();
155  lteIt->second->GetMac ()->Initialize ();
156  }
157 
158  std::map< uint8_t, Ptr<MmWaveComponentCarrierUe> >::iterator mmWaveIt;
159  for (mmWaveIt = m_mmWaveCcMap.begin (); mmWaveIt != m_mmWaveCcMap.end (); ++mmWaveIt)
160  {
161  mmWaveIt->second->GetPhy ()->Initialize ();
162  mmWaveIt->second->GetMac ()->Initialize ();
163  }
164 
165  if (m_mmWaveRrc != 0)
166  {
167  m_mmWaveRrc->Initialize ();
168  }
169 
170  m_lteRrc->Initialize ();
171 }

174 void
175 McUeNetDevice::DoDispose (void)
176 {
177  NS_LOG_FUNCTION (this);
178  m_lteTargetEnb = 0;
179  m_lteRrc->Dispose ();
180  m_lteRrc = 0;
181  m_nas->Dispose ();
182  m_nas = 0;
183 
184  for (uint32_t i = 0; i < m_lteCcMap.size (); i++)
185  {
186  m_lteCcMap.at (i)->Dispose ();
187  }
188  m_lteComponentCarrierManager->Dispose ();
189 
190  m_mmWaveTargetEnb = 0;
191  if (m_mmWaveRrc != 0)
192  {
193  m_mmWaveRrc->Dispose ();
194  }
195  m_mmWaveRrc = 0;
196 
197  for (uint32_t i = 0; i < m_mmWaveCcMap.size (); i++)
198  {
199  m_mmWaveCcMap.at (i)->Dispose ();
200  }
201  m_mmWaveComponentCarrierManager->Dispose ();
202 
203  m_node = 0;
204  NetDevice::DoDispose ();
205 }

398 void
399 McUeNetDevice::Receive (Ptr<Packet> p)
400 {
401  NS_LOG_FUNCTION (this << p);
402  Ipv4Header ipv4Header;
403  Ipv6Header ipv6Header;
404 
405  if (p->PeekHeader (ipv4Header) != 0)
406  {
407  NS_LOG_LOGIC ("IPv4 stack...");
408  m_rxCallback (this, p, Ipv4L3Protocol::PROT_NUMBER, Address ());
409  }
410  else if (p->PeekHeader (ipv6Header) != 0)
411  {
412  NS_LOG_LOGIC ("IPv6 stack...");
413  m_rxCallback (this, p, Ipv6L3Protocol::PROT_NUMBER, Address ());
414  }
415  else
416  {
417  NS_ABORT_MSG ("McUeNetDevice::Receive - Unknown IP type...");
418  }
419 }

442 McUeNetDevice::UpdateConfig (void)
443 {
444  NS_LOG_FUNCTION (this);

447  {
448  NS_LOG_LOGIC (this << " Updating configuration: IMSI " << m_imsi
449  << " CSG ID " << m_csgId);

451

454  {

459  }
460  else
461  {
462  /*
463  * NAS and RRC instances are not be ready yet, so do nothing now and
464  * expect ``DoInitialize`` to re-invoke this function.
465  */
466  }
467 }
