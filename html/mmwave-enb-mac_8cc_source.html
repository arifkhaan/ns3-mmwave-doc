461 void
462 MmWaveEnbMac::DoSlotIndication (SfnSf sfnSf)
463 {
464  NS_LOG_FUNCTION (this);
465  m_frameNum = sfnSf.m_frameNum;
466  m_sfNum = sfnSf.m_sfNum;
467  m_slotNum = sfnSf.m_slotNum;
468  bool slotStart = (sfnSf.m_symStart == 0);
469 
470  // --- DOWNLINK ---
471  // Send Dl-CQI info to the scheduler if(m_dlCqiReceived.size () > 0)
472  {
473  MmWaveMacSchedSapProvider::SchedDlCqiInfoReqParameters dlCqiInfoReq;
474  dlCqiInfoReq.m_sfnsf = sfnSf;
475 
476  dlCqiInfoReq.m_cqiList.insert (dlCqiInfoReq.m_cqiList.begin (), m_dlCqiReceived.begin (), m_dlCqiReceived.end ());
477  m_dlCqiReceived.erase (m_dlCqiReceived.begin (), m_dlCqiReceived.end ());
478 
479  m_macSchedSapProvider->SchedDlCqiInfoReq (dlCqiInfoReq);
480  }
481 
482  if (!m_receivedRachPreambleCount.empty ())
483  {
484  // process received RACH preambles and notify the scheduler
485  Ptr<MmWaveRarMessage> rarMsg = Create<MmWaveRarMessage> ();
486 
487  for (std::map<uint8_t, uint32_t>::const_iterator it = m_receivedRachPreambleCount.begin ();
488  it != m_receivedRachPreambleCount.end ();
489  ++it)
490  {
491  uint16_t rnti;
492  std::map<uint8_t, NcRaPreambleInfo>::iterator jt = m_allocatedNcRaPreambleMap.find (it->first);
493  NS_LOG_LOGIC ("received RapId " << (uint16_t)it->first);
494  if (jt != m_allocatedNcRaPreambleMap.end ())
495  {
496  rnti = jt->second.rnti;
497  NS_LOG_INFO ("preambleId previously allocated for NC based RA, RNTI =" << (uint32_t) rnti << ", sending RAR");
498  m_allocatedNcRaPreambleMap.erase (jt);
499  }
500  else
501  {
502  rnti = m_cmacSapUser->AllocateTemporaryCellRnti ();
503  NS_LOG_INFO ("preambleId " << (uint32_t) it->first << ": allocated T-C-RNTI " << (uint32_t) rnti << ", sending RAR");
504  }
505  MmWaveRarMessage::Rar rar;
506  rar.rapId = (*it).first;
507  rar.rarPayload.m_rnti = rnti;
508  rarMsg->AddRar (rar);
509  //NS_ASSERT_MSG((*it).second ==1, "Two user send the same Rach ID, collision detected");
510  }
511  m_phySapProvider->SendControlMessage (rarMsg);
512  m_receivedRachPreambleCount.clear ();
513  }
514 
515  // --- UPLINK ---
516  // Send UL-CQI info to the scheduler
517  std::vector <MmWaveMacSchedSapProvider::SchedUlCqiInfoReqParameters>::iterator itCqi;
518  for (uint16_t i = 0; i < m_ulCqiReceived.size (); i++)
519  {
520  //m_ulCqiReceived.at (i).m_sfnSf = ((0x3FF & frameNum) << 16) | ((0xFF & subframeNum) << 8) | (0xFF & slotNum);
521  m_macSchedSapProvider->SchedUlCqiInfoReq (m_ulCqiReceived.at (i));
522  }
523  m_ulCqiReceived.clear ();
524 
525  // Send UL BSR reports to the scheduler
526  if (m_ulCeReceived.size () > 0)
527  {
528  MmWaveMacSchedSapProvider::SchedUlMacCtrlInfoReqParameters ulMacReq;
529  ulMacReq.m_sfnSf = sfnSf;
530  NS_LOG_DEBUG ("ulMacReq.m_macCeList size " << m_ulCeReceived.size ());
531  ulMacReq.m_macCeList.insert (ulMacReq.m_macCeList.begin (), m_ulCeReceived.begin (), m_ulCeReceived.end ());
532  m_ulCeReceived.erase (m_ulCeReceived.begin (), m_ulCeReceived.end ());
533  m_macSchedSapProvider->SchedUlMacCtrlInfoReq (ulMacReq);
534  }
535 
536  if (slotStart)
537  {
538  NS_LOG_LOGIC ("Starting a new NR slot - DoSlotIndication");
539  NS_LOG_DEBUG ("Current frame " << sfnSf.m_frameNum << " subframe " << (unsigned)sfnSf.m_sfNum << " slot " << (unsigned)sfnSf.m_slotNum);
540  // Trigger scheduler, taking into consideration the L1L2 delay
541 
542  uint8_t delayedSlotNum = (m_slotNum + m_phyMacConfig->GetL1L2Latency ()) % m_phyMacConfig->GetSlotsPerSubframe ();
543  uint8_t deltaSubframe = (m_slotNum + m_phyMacConfig->GetL1L2Latency ()) / m_phyMacConfig->GetSlotsPerSubframe ();
544  uint8_t delayedSchedSfNum = (m_sfNum + deltaSubframe) % m_phyMacConfig->GetSubframesPerFrame ();
545  uint16_t delayedSchedFrameNum = m_frameNum + ((m_sfNum + deltaSubframe) / m_phyMacConfig->GetSubframesPerFrame ());
546 
547  NS_ASSERT ((delayedSlotNum < m_phyMacConfig->GetSlotsPerSubframe ()) && (delayedSchedSfNum < m_phyMacConfig->GetSubframesPerFrame ())
548  && (deltaSubframe >= 0) && (delayedSlotNum >= 0) && (delayedSchedSfNum >= 0) && (delayedSchedFrameNum >= m_frameNum));
549 
550  NS_LOG_DEBUG ("Requesting scheduler allocation for frame " << delayedSchedFrameNum << " subframe " << (unsigned)delayedSchedSfNum
551  << " slot " << (unsigned)delayedSlotNum);
552 
553  MmWaveMacSchedSapProvider::SchedTriggerReqParameters params;
554  SfnSf schedSfn (delayedSchedFrameNum, delayedSchedSfNum, delayedSlotNum);
555  params.m_snfSf = schedSfn;
556 
557  // Forward DL HARQ feebacks collected during last subframe TTI
558  if (m_dlHarqInfoReceived.size () > 0)
559  {
560  params.m_dlHarqInfoList = m_dlHarqInfoReceived;
561  // empty local buffer
562  m_dlHarqInfoReceived.clear ();
563  }
564 
565  // Forward UL HARQ feebacks collected during last TTI
566  if (m_ulHarqInfoReceived.size () > 0)
567  {
568  params.m_ulHarqInfoList = m_ulHarqInfoReceived;
569  // empty local buffer
570  m_ulHarqInfoReceived.clear ();
571  }
572 
573  params.m_ueList = m_associatedUe;
574  m_macSchedSapProvider->SchedTriggerReq (params);
575  }

605 void
606 MmWaveEnbMac::DoReceivePhyPdu (Ptr<Packet> p)
607 {
608  NS_LOG_FUNCTION (this);
609  LteRadioBearerTag tag;
610  p->RemovePacketTag (tag);
611  uint16_t rnti = tag.GetRnti ();
612  MmWaveMacPduHeader macHeader;
613  p->RemoveHeader (macHeader);

617  uint32_t currPos = 0;

621  {

933  m_macPduMap.insert (std::pair<uint32_t, struct MacPduInfo> (tbMapKey, macPduInfo));
