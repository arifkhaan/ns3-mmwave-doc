245 MmWaveUePhy::UpdateSinrEstimate (uint16_t cellId, double sinr)
246 {
247  NS_LOG_FUNCTION (this);
248  if (m_cellSinrMap.find (cellId) != m_cellSinrMap.end ())
249  {
250  m_cellSinrMap.find (cellId)->second = sinr;
251  }
252  else
253  {
254  m_cellSinrMap.insert (std::pair<uint16_t, double> (cellId, sinr));
255  }
256 
257  if (cellId == m_cellId) // update for SNR of the current cell
258  {
259  long double currentCellSinr = 10 * std::log10 (m_cellSinrMap.find (m_cellId)->second);
260  if (currentCellSinr < m_outageThreshold)
261  {
262  m_consecutiveSinrBelowThreshold++;
263  if (m_consecutiveSinrBelowThreshold > m_n310)
264  {
265  // TODO raise a call to upper layers
266  NS_LOG_DEBUG ("Phy layer detects SNR below threshold for " << m_n310 << " times");
267  }
268  }
269  else
270  {
271  m_consecutiveSinrBelowThreshold = 0;
272  }
273  NS_LOG_DEBUG ("Phy layers: update sinr value for cell " << m_cellId << " to " << currentCellSinr << " m_consecutiveSinrBelowThreshold " << (uint16_t)m_consecutiveSinrBelowThreshold << " at time " << Simulator::Now ());
274  }

370 void
371 MmWaveUePhy::ReceiveControlMessageList (std::list<Ptr<MmWaveControlMessage> > msgList)
372 {
373  NS_LOG_FUNCTION (this);
374 
375  std::list<Ptr<MmWaveControlMessage> >::iterator it;
376  for (it = msgList.begin (); it != msgList.end (); it++)
377  {
378  Ptr<MmWaveControlMessage> msg = (*it);
379 
380  if (msg->GetMessageType () == MmWaveControlMessage::DCI_TDMA)
381  {
382  NS_ASSERT_MSG (m_ttiIndex == 0, "UEs" << m_rnti << " should receive DCIs only at the beginning of new slots");
383  Ptr<MmWaveTdmaDciMessage> dciMsg = DynamicCast<MmWaveTdmaDciMessage> (msg);
384  DciInfoElementTdma dciInfoElem = dciMsg->GetDciInfoElement ();
385  SfnSf dciSfn = dciMsg->GetSfnSf ();
386 
387  if (dciSfn.m_frameNum != m_frameNum || dciSfn.m_sfNum != m_sfNum)
388  {
389  NS_FATAL_ERROR ("DCI intended for different subframe (dci= "
390  << dciSfn.m_frameNum << " " << dciSfn.m_sfNum << ", actual= " << m_frameNum << " " << m_sfNum);
391  }
392 
393  NS_LOG_DEBUG ("UE" << m_rnti << " DCI received for RNTI " << dciInfoElem.m_rnti << " in frame " << m_frameNum << " subframe " << (unsigned)m_sfNum << " slot " <<
394  (unsigned)m_slotNum << " format " << (unsigned)dciInfoElem.m_format << " symStart " << (unsigned)dciInfoElem.m_symStart << " numSym " << (unsigned)dciInfoElem.m_numSym);
395 
396  if (dciInfoElem.m_rnti != m_rnti)
397  {
398  continue; // DCI not for me
399  }
400 
401  if (dciInfoElem.m_format == DciInfoElementTdma::DL_dci) // set downlink slot schedule for current slot
402  {
403  NS_LOG_DEBUG ("UE" << m_rnti << " DL-DCI received for frame " << m_frameNum << " subframe " << (unsigned)m_sfNum
404  << " symStart " << (unsigned)dciInfoElem.m_symStart << " numSym " << (unsigned)dciInfoElem.m_numSym << " tbs " << dciInfoElem.m_tbSize
405  << " harqId " << (unsigned)dciInfoElem.m_harqProcess);
406 
407  TtiAllocInfo ttiInfo;
408  ttiInfo.m_tddMode = TtiAllocInfo::DL_slotAllocInfo;
409  ttiInfo.m_dci = dciInfoElem;
410  ttiInfo.m_ttiIdx = 0;
411  std::deque <TtiAllocInfo>::iterator itTti;
412  for (itTti = m_currSlotAllocInfo.m_ttiAllocInfo.begin ();
413  itTti != m_currSlotAllocInfo.m_ttiAllocInfo.end (); itTti++)
414  {
415  if (itTti->m_tddMode == TtiAllocInfo::UL_slotAllocInfo)
416  {
417  break;
418  }
419  ttiInfo.m_ttiIdx++;
420  }
421  //m_currSfAllocInfo.m_slotAllocInfo.push_back (slotInfo); // add SlotAllocInfo to current SfAllocInfo
422  m_currSlotAllocInfo.m_ttiAllocInfo.insert (itTti, ttiInfo);
423  }
424  else if (dciInfoElem.m_format == DciInfoElementTdma::UL_dci) // set UL slot schedule for t+ulSchedDelay slot
425  {
426  uint8_t ulSlotIdx = (m_slotNum + m_phyMacConfig->GetUlSchedDelay ()) % m_phyMacConfig->GetSlotsPerSubframe ();
427  uint8_t dciSubframe = m_sfNum + (((m_slotNum + m_phyMacConfig->GetUlSchedDelay ()) / m_phyMacConfig->GetSlotsPerSubframe ())
428  % m_phyMacConfig->GetSubframesPerFrame ());
429  uint16_t dciFrame = m_frameNum + (((m_slotNum + m_phyMacConfig->GetUlSchedDelay ()) / m_phyMacConfig->GetSlotsPerSubframe ())
430  / m_phyMacConfig->GetSubframesPerFrame ());
431 
432  NS_LOG_DEBUG ("UE" << m_rnti << " UL-DCI received for frame " << dciFrame << " subframe " << (unsigned)dciSubframe
433  << " slot " << (unsigned)ulSlotIdx << " symStart " << (unsigned)dciInfoElem.m_symStart << " numSym "
434  << (unsigned)dciInfoElem.m_numSym << " tbs " << dciInfoElem.m_tbSize
435  << " harqId " << (unsigned)dciInfoElem.m_harqProcess);
436 
437  TtiAllocInfo ttiInfo;
438  ttiInfo.m_tddMode = TtiAllocInfo::UL_slotAllocInfo;
439  ttiInfo.m_dci = dciInfoElem;
440  TtiAllocInfo ulCtrlTti = m_slotAllocInfo[ulSlotIdx].m_ttiAllocInfo.back ();
441  m_slotAllocInfo[ulSlotIdx].m_ttiAllocInfo.pop_back ();
442  //ulCtrlSlot.m_slotIdx++;
443  ttiInfo.m_ttiIdx = m_slotAllocInfo[ulSlotIdx].m_ttiAllocInfo.size ();
444  m_slotAllocInfo[ulSlotIdx].m_ttiAllocInfo.push_back (ttiInfo);
445  m_slotAllocInfo[ulSlotIdx].m_ttiAllocInfo.push_back (ulCtrlTti);
446  }
447 
448  m_phySapUser->ReceiveControlMessage (msg);
449  }
450  else if (msg->GetMessageType () == MmWaveControlMessage::MIB)
451  {
452  NS_LOG_INFO ("received MIB");
453  NS_ASSERT (m_cellId > 0);
454  Ptr<MmWaveMibMessage> msg2 = DynamicCast<MmWaveMibMessage> (msg);
455  m_ueCphySapUser->RecvMasterInformationBlock (m_cellId, msg2->GetMib ());
456  }
457  else if (msg->GetMessageType () == MmWaveControlMessage::SIB1)
458  {
459  NS_ASSERT (m_cellId > 0);
460  Ptr<MmWaveSib1Message> msg2 = DynamicCast<MmWaveSib1Message> (msg);
461  m_ueCphySapUser->RecvSystemInformationBlockType1 (m_cellId, msg2->GetSib1 ());
462  }
463  else if (msg->GetMessageType () == MmWaveControlMessage::RAR)
464  {
465  NS_LOG_INFO ("received RAR");
466  NS_ASSERT (m_cellId > 0);
467 
468  Ptr<MmWaveRarMessage> rarMsg = DynamicCast<MmWaveRarMessage> (msg);
469 
470  for (std::list<MmWaveRarMessage::Rar>::const_iterator it = rarMsg->RarListBegin ();
471  it != rarMsg->RarListEnd ();
472  ++it)
473  {
474  if (it->rapId == m_raPreambleId)
475  {
476  m_phySapUser->ReceiveControlMessage (rarMsg);
477  }
478  }
479  }
480  else
481  {
482  NS_LOG_DEBUG ("Control message not handled. Type: " << msg->GetMessageType ());
483  }
484  }

522 void
523 MmWaveUePhy::StartTti ()
524 {
525  NS_LOG_FUNCTION (this);
526  NS_LOG_DEBUG ("Current time: " << Simulator::Now ().GetNanoSeconds ());
527 
528  if (m_cellId > 0)
529  {
530  // point the beam towards the serving BS
531  NS_LOG_LOGIC("UE " << m_rnti << " cid " << m_cellId << " BF");
532  m_downlinkSpectrumPhy->ConfigureBeamforming (m_registeredEnb.find (m_cellId)->second.second);
533  }
534 
535  TtiAllocInfo currTti = m_currSlotAllocInfo.m_ttiAllocInfo[m_ttiIndex];
536  Time currTtiDuration; // Duration of the current TTI
537 
538  m_currTti = currTti;
539 
540  NS_LOG_INFO ("MmWave UE " << m_rnti << " frame " << m_frameNum << " subframe " << (uint16_t) m_sfNum << " slot "
541  << (uint16_t) m_slotNum << " TTI index " << (uint16_t) m_ttiIndex );
542 
543  m_receptionEnabled = false;
544 
545  if (m_ttiIndex == 0) // First TTI: reserved DL control
546  {
547  currTtiDuration = m_phyMacConfig->GetDlCtrlSymbols () * m_phyMacConfig->GetSymbolPeriod ();
548  NS_LOG_DEBUG ("UE" << m_rnti << " imsi" << m_imsi << " RXing DL CTRL frame " << m_frameNum << " subframe " << (unsigned)m_sfNum << " symbols "
549  << (unsigned)currTti.m_dci.m_symStart << "-" << (unsigned)(currTti.m_dci.m_symStart + currTti.m_dci.m_numSym - 1) <<
550  "\t start " << Simulator::Now () << " end " << (Simulator::Now () + currTtiDuration ));
551 
552  }
553  else if (m_ttiIndex == m_currSlotAllocInfo.m_ttiAllocInfo.size () - 1) // Last TTI of this slot: reserved UL control
554  {
555  SetSubChannelsForTransmission (m_channelChunks);
556  currTtiDuration = m_phyMacConfig->GetUlCtrlSymbols () * m_phyMacConfig->GetSymbolPeriod ();
