198 void
199 MmWaveUeRrcProtocolIdeal::SetEnbRrcSapProvider ()
200 {
201  uint16_t cellId = m_rrc->GetCellId ();
202 
203  // walk list of all nodes to get the peer eNB
204  Ptr<MmWaveEnbNetDevice> enbDev;
205  NodeList::Iterator listEnd = NodeList::End ();
206  bool found = false;
207  for (NodeList::Iterator i = NodeList::Begin ();
208  (i != listEnd) && (!found);
209  ++i)
210  {
211  Ptr<Node> node = *i;
212  int nDevs = node->GetNDevices ();
213  for (int j = 0;
214  (j < nDevs) && (!found);
215  j++)
216  {
217  enbDev = node->GetDevice (j)->GetObject <MmWaveEnbNetDevice> ();
218  if (enbDev == 0)
219  {
220  continue;
221  }
222  else
223  {
224  if (enbDev->HasCellId (cellId))
225  {
226  found = true;
227  break;
228  }
229  }
230  }
231  }
232  NS_ASSERT_MSG (found, " Unable to find eNB with CellId =" << cellId);
233  m_enbRrcSapProvider = enbDev->GetRrc ()->GetLteEnbRrcSapProvider ();
234  Ptr<MmWaveEnbRrcProtocolIdeal> enbRrcProtocolIdeal = enbDev->GetRrc ()->GetObject<MmWaveEnbRrcProtocolIdeal> ();
235  enbRrcProtocolIdeal->SetUeRrcSapProvider (m_rnti, m_ueRrcSapProvider);
236 }

321 void
322 MmWaveEnbRrcProtocolIdeal::DoSendSystemInformation (uint16_t cellId, LteRrcSap::SystemInformation msg)
323 {
324  NS_LOG_FUNCTION (this << cellId);
325  // walk list of all nodes to get UEs with this cellId
326  Ptr<LteUeRrc> ueRrc;
327  for (NodeList::Iterator i = NodeList::Begin (); i != NodeList::End (); ++i)
328  {
329  Ptr<Node> node = *i;
330  int nDevs = node->GetNDevices ();
331  for (int j = 0; j < nDevs; ++j)
332  {
333  Ptr<mmwave::MmWaveUeNetDevice> mmWaveUeDev = node->GetDevice (j)->GetObject <mmwave::MmWaveUeNetDevice> ();
334  if (mmWaveUeDev != 0)
335  {
336  ueRrc = mmWaveUeDev->GetRrc ();
337  NS_LOG_LOGIC ("considering UE IMSI " << mmWaveUeDev->GetImsi () << " that has cellId " << ueRrc->GetCellId ());
338  if (ueRrc->GetCellId () == cellId)
339  {
340  NS_LOG_LOGIC ("sending SI to IMSI " << mmWaveUeDev->GetImsi ());
341  ueRrc->GetLteUeRrcSapProvider ()->RecvSystemInformation (msg);
342  Simulator::Schedule (RRC_IDEAL_MSG_DELAY,
343  &LteUeRrcSapProvider::RecvSystemInformation,
344  ueRrc->GetLteUeRrcSapProvider (),
345  msg);
346  }
347  }
348  else
349  {
350  // it may be a McUeNetDevice
351  Ptr<McUeNetDevice> mcUeDev = node->GetDevice (j)->GetObject <McUeNetDevice> ();
352  if (mcUeDev != 0)
353  {
354  NS_FATAL_ERROR ("-----------This implementation does not support McUeNetDevice. Use MmWaveRrcProtocolReal.");
355  ueRrc = mcUeDev->GetMmWaveRrc ();
356  if (ueRrc != 0) // actually is using 2 connections
357  {
358  NS_LOG_LOGIC ("considering UE IMSI " << mcUeDev->GetImsi () << " that has cellId " << ueRrc->GetCellId ());
359  NS_LOG_LOGIC ("UE cellId " << (uint32_t)ueRrc->GetCellId () << " ENB cellId " << (uint32_t)cellId);
360  if (ueRrc->GetCellId () == m_cellId)
361  {
362  NS_LOG_LOGIC ("sending SI to IMSI " << mcUeDev->GetImsi ());
363  ueRrc->GetLteUeRrcSapProvider ()->RecvSystemInformation (msg);
364  Simulator::Schedule (RRC_IDEAL_MSG_DELAY,
365  &LteUeRrcSapProvider::RecvSystemInformation,
366  ueRrc->GetLteUeRrcSapProvider (),
367  msg);
368  }
369  }
370  else // it may have just a double stack up to MAC layer
371  {
372  ueRrc = mcUeDev->GetLteRrc ();
373  if (ueRrc != 0)
374  {
375  NS_LOG_LOGIC ("considering UE IMSI " << mcUeDev->GetImsi () << " that has cellId " << ueRrc->GetCellId ());
376  NS_LOG_LOGIC ("UE cellId " << (uint32_t)ueRrc->GetCellId () << " ENB cellId " << (uint32_t)cellId);
377  if (ueRrc->GetCellId () == m_cellId)
378  {
379  NS_LOG_LOGIC ("sending SI to IMSI " << mcUeDev->GetImsi ());
380  ueRrc->GetLteUeRrcSapProvider ()->RecvSystemInformation (msg);
381  Simulator::Schedule (RRC_IDEAL_MSG_DELAY,
382  &LteUeRrcSapProvider::RecvSystemInformation,
383  ueRrc->GetLteUeRrcSapProvider (),
384  msg);
385  }
386  }
387  }
388  }
389  }
390  }
391  }

464 /*
465  * The purpose of MmWaveEnbRrcProtocolIdeal is to avoid encoding
466  * messages. In order to do so, we need to have some form of encoding for
467  * inter-node RRC messages like HandoverPreparationInfo and HandoverCommand. Doing so
468  * directly is not practical (these messages includes a lot of
469  * information elements, so encoding all of them would defeat the
470  * purpose of MmWaveEnbRrcProtocolIdeal. The workaround is to store the
471  * actual message in a global map, so that then we can just encode the
472  * key in a header and send that between eNBs over X2.
473  *
474  */
475 
476 static std::map<uint32_t, LteRrcSap::HandoverPreparationInfo> g_handoverPreparationInfoMsgMap;
477 static uint32_t g_handoverPreparationInfoMsgIdCounter = 0;

488  void SetMsgId (uint32_t id);

493  virtual void Serialize (Buffer::Iterator start) const;

496 private:

552 Ptr<Packet>
553 MmWaveEnbRrcProtocolIdeal::DoEncodeHandoverPreparationInformation (LteRrcSap::HandoverPreparationInfo msg)
554 {
555  uint32_t msgId = ++g_handoverPreparationInfoMsgIdCounter;
556  NS_ASSERT_MSG (g_handoverPreparationInfoMsgMap.find (msgId) == g_handoverPreparationInfoMsgMap.end (), "msgId " << msgId << " already in use");
557  NS_LOG_INFO (" encoding msgId = " << msgId);
558  g_handoverPreparationInfoMsgMap.insert (std::pair<uint32_t, LteRrcSap::HandoverPreparationInfo> (msgId, msg));
559  MmWaveIdealHandoverPreparationInfoHeader h;
560  h.SetMsgId (msgId);
561  Ptr<Packet> p = Create<Packet> ();
562  p->AddHeader (h);
563  return p;
564 }
565 
566 LteRrcSap::HandoverPreparationInfo
567 MmWaveEnbRrcProtocolIdeal::DoDecodeHandoverPreparationInformation (Ptr<Packet> p)
568 {
569  MmWaveIdealHandoverPreparationInfoHeader h;
570  p->RemoveHeader (h);
571  uint32_t msgId = h.GetMsgId ();
572  NS_LOG_INFO (" decoding msgId = " << msgId);
573  std::map<uint32_t, LteRrcSap::HandoverPreparationInfo>::iterator it = g_handoverPreparationInfoMsgMap.find (msgId);
574  NS_ASSERT_MSG (it != g_handoverPreparationInfoMsgMap.end (), "msgId " << msgId << " not found");
575  LteRrcSap::HandoverPreparationInfo msg = it->second;
576  g_handoverPreparationInfoMsgMap.erase (it);
577  return msg;
578 }
