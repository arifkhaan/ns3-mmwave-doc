43 struct SfnSf

45 
54  SfnSf (uint16_t frameNum = 0, uint8_t sfNum = 0, uint8_t slotNum = 0, uint8_t symStart = 0)
55  : m_frameNum (frameNum), m_sfNum (sfNum), m_slotNum (slotNum), m_symStart (symStart)
56  {
57  // Non-negative
58  NS_ASSERT ((m_frameNum >= 0) && (m_sfNum >= 0) && (m_slotNum >= 0) && (m_symStart >= 0));
59  }
60 
64  uint64_t
65  Encode () const
66  {
67  return ((uint64_t (m_frameNum) << 24 ) | ((m_sfNum & 0xFF) << 16) | ((m_slotNum & 0xFF) << 8) | (m_symStart & 0xFF));
68  }
69 
73  void
74  Decode (uint64_t sfn)
75  {
76  m_frameNum = sfn >> 24;
77  m_sfNum = ((sfn & 0xFF0000) >> 16);
78  m_slotNum = ((sfn & 0xFF00) >> 8);
79  m_symStart = sfn & 0xFF;
80 
81  }
82 
83  uint16_t m_frameNum;
84  uint8_t m_sfNum;
85  uint8_t m_slotNum;
86  uint8_t m_symStart;
87 };
