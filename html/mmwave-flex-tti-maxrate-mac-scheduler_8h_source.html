208  static bool CompareUeWeightsMaxRate (UeSchedInfo* lue, UeSchedInfo* rue)
209  {
210  double lPfMetric = std::max (lue->m_currTputDl,lue->m_currTputUl);
211  double rPfMetric = std::max (rue->m_currTputDl,rue->m_currTputUl);
212  return (lPfMetric > rPfMetric);
213  }

219  BsrId2BufferSize (uint8_t val)
220  {
221  NS_ABORT_MSG_UNLESS (val < 64, "val = " << val << " is out of range");
222  return BufferSizeLevelBsrTable[val];

224 
225  uint8_t
226  BufferSize2BsrId (uint32_t val)
227  {
228  int index = 0;
229  if (BufferSizeLevelBsrTable[63] < val)
230  {
231  index = 63;
232  }
233  else
234  {
235  while (BufferSizeLevelBsrTable[index] < val)
236  {
237  NS_ASSERT (index < 64);
238  index++;
239  }
240  }
241 
242  return (index);
243  }
