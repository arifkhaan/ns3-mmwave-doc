113 void
114 MmWaveUeNetDevice::DoInitialize (void)
115 {
116  NS_LOG_FUNCTION (this);
117  m_isConstructed = true;
118  UpdateConfig ();
119 
120  for (auto it = m_ccMap.begin (); it != m_ccMap.end (); ++it)
121  {
122  Ptr<MmWaveComponentCarrierUe> ccUe = DynamicCast<MmWaveComponentCarrierUe> (it->second);
123  ccUe->GetPhy ()->Initialize ();
124  ccUe->GetMac ()->Initialize ();
125  }
126  m_rrc->Initialize ();
127 
128 }

130 MmWaveUeNetDevice::DoDispose ()
131 {
132  NS_LOG_FUNCTION (this);

134 
135  m_rrc->Dispose ();
136  m_rrc = 0;

142  {
143  m_ccMap.at (i)->Dispose ();
144  }
145  m_componentCarrierManager->Dispose ();
146 }

163 void
164 MmWaveUeNetDevice::UpdateConfig (void)
165 {
166  NS_LOG_FUNCTION (this);
167 
168  if (m_isConstructed)
169  {
170  NS_LOG_LOGIC (this << " Updating configuration: IMSI " << m_imsi
171  << " CSG ID " << m_csgId);
172  m_nas->SetImsi (m_imsi);
173  m_rrc->SetImsi (m_imsi);
174  m_nas->SetCsgId (m_csgId); // this also handles propagation to RRC
175  }
176  else
177  {
178  /*
179  * NAS and RRC instances are not be ready yet, so do nothing now and
180  * expect ``DoInitialize`` to re-invoke this function.
181  */
182  }
183 }
