212 void
213 MmWavePhy::SetMacPdu (Ptr<Packet> p)
214 {
215  MmWaveMacPduTag tag;
216  if (p->PeekPacketTag (tag))
217  {
218  NS_ASSERT ((tag.GetSfn ().m_sfNum >= 0) && (tag.GetSfn ().m_sfNum < m_phyMacConfig->GetSubframesPerFrame ()));
219  std::map<uint64_t, Ptr<PacketBurst> >::iterator it = m_packetBurstMap.find (tag.GetSfn ().Encode ());
220  if (it == m_packetBurstMap.end ())
221  {
222  it = m_packetBurstMap.insert (std::pair<uint64_t, Ptr<PacketBurst> > (tag.GetSfn ().Encode (), CreateObject<PacketBurst> ())).first;
223  }
224  else
225  {
226  NS_FATAL_ERROR ("Packet burst map entry already exists");
227  }
228  it->second->AddPacket (p);
229  }
230  else
231  {
232  NS_FATAL_ERROR ("No MAC packet PDU header available");
233  }
234 }

236 Ptr<PacketBurst>
237 MmWavePhy::GetPacketBurst (SfnSf sfn)
238 {
239  Ptr<PacketBurst> pburst;
240  std::map<uint64_t, Ptr<PacketBurst> >::iterator it = m_packetBurstMap.find (sfn.Encode ());
241  if (it == m_packetBurstMap.end ())
242  {
243  NS_LOG_ERROR ("GetPacketBurst(): Packet burst not found for frame " << (unsigned)sfn.m_frameNum << " subframe "
244  << (unsigned)sfn.m_sfNum << " slot " << (unsigned)sfn.m_slotNum << "sym start " << (unsigned)sfn.m_symStart);
245  return pburst;
246  }
247  else
248  {
249  pburst = it->second;
250  m_packetBurstMap.erase (it);
251  }
252  return pburst;
253 }

255 void
256 MmWavePhy::SetControlMessage (Ptr<MmWaveControlMessage> m)
257 {
258  if (m_controlMessageQueue.empty ())
259  {
260  std::list<Ptr<MmWaveControlMessage> > l;
261  l.push_back (m);
262  m_controlMessageQueue.push_back (l);
263  }
264  else
265  {
266  m_controlMessageQueue.at (m_controlMessageQueue.size () - 1).push_back (m);
267  }
268 }
269 
270 std::list<Ptr<MmWaveControlMessage> >
271 MmWavePhy::GetControlMessages (void)
272 {
273  NS_LOG_FUNCTION (this);
274  if (m_controlMessageQueue.empty ())
275  {
276  std::list<Ptr<MmWaveControlMessage> > emptylist;
277  return (emptylist);
278  }
279 
280  if (m_controlMessageQueue.at (0).size () > 0)
281  {
282  std::list<Ptr<MmWaveControlMessage> > ret = m_controlMessageQueue.front ();
283  m_controlMessageQueue.erase (m_controlMessageQueue.begin ());
284  std::list<Ptr<MmWaveControlMessage> > newlist;
285  m_controlMessageQueue.push_back (newlist);
286  return (ret);
287  }
288  else
289  {
290  m_controlMessageQueue.erase (m_controlMessageQueue.begin ());
291  std::list<Ptr<MmWaveControlMessage> > newlist;
292  m_controlMessageQueue.push_back (newlist);
293  std::list<Ptr<MmWaveControlMessage> > emptylist;
294  return (emptylist);
295  }

298 void
299 MmWavePhy::SetSlotCtrlStructure (uint8_t slotIndex)
300 {
301  // Currently hardcoded: first OFDM symbol = DL control, last OFDM symbol = UL control
302  TtiAllocInfo dlCtrlTti;
303  dlCtrlTti.m_ttiType = TtiAllocInfo::CTRL;
304  dlCtrlTti.m_numCtrlSym = 1;
305  dlCtrlTti.m_tddMode = TtiAllocInfo::DL_slotAllocInfo;
306  dlCtrlTti.m_dci.m_numSym = 1;
307  dlCtrlTti.m_dci.m_symStart = 0;
308  TtiAllocInfo ulCtrlTti;
309  ulCtrlTti.m_ttiType = TtiAllocInfo::CTRL;
310  ulCtrlTti.m_numCtrlSym = 1;
311  ulCtrlTti.m_tddMode = TtiAllocInfo::UL_slotAllocInfo;
312  ulCtrlTti.m_ttiIdx = 0xFF;
313  ulCtrlTti.m_dci.m_numSym = 1;
314  ulCtrlTti.m_dci.m_symStart = m_phyMacConfig->GetSymbPerSlot () - 1;
315  m_slotAllocInfo[slotIndex].m_ttiAllocInfo.push_front (dlCtrlTti);
316  m_slotAllocInfo[slotIndex].m_ttiAllocInfo.push_back (ulCtrlTti);
317 }
