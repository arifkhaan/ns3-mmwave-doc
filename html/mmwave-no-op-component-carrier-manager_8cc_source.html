146 void
147 MmWaveNoOpComponentCarrierManager::DoAddUe (uint16_t rnti, uint8_t state)
148 {
149  NS_LOG_FUNCTION (this << rnti << (uint16_t) state);
150  std::map<uint16_t, uint8_t>::iterator stateIt;
151  std::map<uint16_t, uint8_t>::iterator eccIt; // m_enabledComponentCarrier iterator
152  stateIt = m_ueState.find (rnti);
153  if (stateIt == m_ueState.end ())
154  {
155  // NS_ASSERT_MSG ((stateIt == m_ueState.end () && state == 3), " ERROR: Ue was not indexed and current state is CONNECTED_NORMALLY" << (uint16_t) state);
156  NS_LOG_DEBUG (this << " UE " << rnti << " was not found, now it is added in the map");
157  m_ueState.insert (std::pair<uint16_t, uint8_t> (rnti, state));
158  eccIt = m_enabledComponentCarrier.find (rnti);
159  //if ((state == 7 || state == 0) && eccIt == m_enabledComponentCarrier.end ())
160  if (eccIt == m_enabledComponentCarrier.end ())
161  {
162  // the Primary carrier (PC) is enabled by default
163  // on the PC the SRB0 and SRB1 are enabled when the Ue is connected
164  // these are hard-coded and the configuration not pass through the
165  // Component Carrier Manager which is responsible of configure
166  // only DataRadioBearer on the different Component Carrier
167  m_enabledComponentCarrier.insert (std::pair<uint16_t, uint8_t> (rnti, 1));
168  }
169  else
170  {
171  NS_FATAL_ERROR (this << " Ue " << rnti << " had Component Carrier enabled before join the network" << (uint16_t) state);
172  }
173  // preparing the rnti,lcid,LteMacSapUser map
174  std::map<uint8_t, LteMacSapUser*> empty;
175  std::pair <std::map <uint16_t, std::map<uint8_t, LteMacSapUser*> >::iterator, bool>
176  ret = m_ueAttached.insert (std::pair <uint16_t, std::map<uint8_t, LteMacSapUser*> >
177  (rnti, empty));
178  NS_LOG_DEBUG (this << "AddUe: UE Pointer LteMacSapUser Map " << rnti << " added " << (uint16_t) ret.second);
179  NS_ASSERT_MSG (ret.second, "element already present, RNTI already existed");
180 
181  //add new rnti in the map
182  std::map<uint8_t, LteEnbCmacSapProvider::LcInfo> emptyA;
183  std::pair <std::map <uint16_t, std::map<uint8_t, LteEnbCmacSapProvider::LcInfo> >::iterator, bool>
184  retA = m_rlcLcInstantiated.insert (std::pair <uint16_t, std::map<uint8_t, LteEnbCmacSapProvider::LcInfo> >
185  (rnti, emptyA));
186  NS_ASSERT_MSG (retA.second, "element already present, RNTI already existed");
187  NS_LOG_DEBUG (this << "AddUe: UE " << rnti << " added " << (uint16_t) retA.second);
188 
189  }
190  else
191  {
192  NS_LOG_DEBUG (this << " UE " << rnti << "found, updating the state from " << (uint16_t) stateIt->second << " to " << (uint16_t) state);
193  stateIt->second = state;
194  }

314 MmWaveNoOpComponentCarrierManager::DoReleaseDataRadioBearer (uint16_t rnti, uint8_t lcid)

323  lcsIt = m_rlcLcInstantiated.find (rnti);

326  NS_LOG_DEBUG (this << " remove lcid " << (uint16_t) lcid << " for rnti " << rnti);
327  lcIt = lcsIt->second.find (lcid);
328  NS_ASSERT_MSG (lcIt != lcsIt->second.end (), " Logical Channel not found");
329  std::vector<uint8_t> res;
330  for (uint16_t i = 0; i < eccIt->second; i++)
331  {
332  res.insert (res.end (), i);
333  }
334  //Find user based on rnti and then erase lcid stored against the same
335  std::map <uint16_t, std::map<uint8_t, LteMacSapUser*> >::iterator rntiIt = m_ueAttached.find (rnti);
336  rntiIt->second.erase (lcid);
337  std::map <uint16_t, std::map<uint8_t, LteEnbCmacSapProvider::LcInfo> >::iterator rlcInstancesIt = m_rlcLcInstantiated.find (rnti);
338  std::map<uint8_t, LteEnbCmacSapProvider::LcInfo>::iterator rclLcIt;

340  NS_ASSERT_MSG (lcIt != lcsIt->second.end (), " Erasing: Logical Channel not found");
341  lcsIt->second.erase (lcid);
342  return res;

376 MmWaveNoOpComponentCarrierManager::DoUlReceiveMacCe (MacCeListElement_s bsr, uint8_t componentCarrierId)
377 {
378  NS_LOG_FUNCTION (this);
379  NS_ASSERT_MSG (bsr.m_macCeType == MacCeListElement_s::BSR, "Received a Control Message not allowed " << bsr.m_macCeType);
380  if ( bsr.m_macCeType == MacCeListElement_s::BSR)
381  {
382  MacCeListElement_s newBsr;
383  newBsr.m_rnti = bsr.m_rnti;
384  newBsr.m_macCeType = bsr.m_macCeType;
385  newBsr.m_macCeValue.m_phr = bsr.m_macCeValue.m_phr;
386  newBsr.m_macCeValue.m_crnti = bsr.m_macCeValue.m_crnti;
387  newBsr.m_macCeValue.m_bufferStatus.resize (4);
388  for (uint16_t i = 0; i < 4; i++)
389  {
390  uint8_t bsrId = bsr.m_macCeValue.m_bufferStatus.at (i);
391  uint32_t buffer = BufferSizeLevelBsr::BsrId2BufferSize (bsrId);
392  // here the buffer should be divide among the different sap
393  // since the buffer status report are compressed information
394  // it is needed to use BsrId2BufferSize to uncompress
395  // after the split over all component carriers is is needed to
396  // compress again the information to fit MacCeListEkement_s structure
397  // verify how many Component Carrier are enabled per UE
398  // in this simple code the BufferStatus will be notify only
399  // to the primary carrier component
400  newBsr.m_macCeValue.m_bufferStatus.at (i) = BufferSizeLevelBsr::BufferSize2BsrId (buffer);
401  }
402  auto sapIt = m_ccmMacSapProviderMap.find (componentCarrierId);
403  if (sapIt == m_ccmMacSapProviderMap.end ())
404  {
405  NS_FATAL_ERROR ("Sap not found in the CcmMacSapProviderMap");
406  }
407  else
408  {
409  // in the current implementation bsr in uplink is forwarded only to the primary carrier.
410  // above code demonstrates how to resize buffer status if more carriers are being used in future
411  sapIt->second->ReportMacCeToScheduler (newBsr);
412  }
413  }
414  else
415  {
416  NS_FATAL_ERROR ("Expected BSR type of message.");
417  }
418 }

448 void
449 MmWaveRrComponentCarrierManager::DoReportBufferStatus (LteMacSapProvider::ReportBufferStatusParameters params)
450 {
451  NS_LOG_FUNCTION (this);
452 
453  NS_ASSERT_MSG ( m_enabledComponentCarrier.find (params.rnti) != m_enabledComponentCarrier.end (), " UE with provided RNTI not found. RNTI:" << params.rnti);
454 
455  uint32_t numberOfCarriersForUe = m_enabledComponentCarrier.find (params.rnti)->second;
456  if (params.lcid == 0 || params.lcid == 1 || numberOfCarriersForUe == 1)
457  {
458  NS_LOG_INFO ("Buffer status forwarded to the primary carrier.");
459  auto ueManager = m_ccmRrcSapUser->GetUeManager (params.rnti);
460  m_macSapProvidersMap.at (ueManager->GetComponentCarrierId ())->ReportBufferStatus (params);
461  }
462  else
463  {
464  params.retxQueueSize /= numberOfCarriersForUe;
465  params.txQueueSize /= numberOfCarriersForUe;
466  for ( uint16_t i = 0; i < numberOfCarriersForUe; i++)
467  {
468  NS_ASSERT_MSG (m_macSapProvidersMap.find (i) != m_macSapProvidersMap.end (), "Mac sap provider does not exist.");
469  if (i == 0)

486 MmWaveRrComponentCarrierManager::DoUlReceiveMacCe (MacCeListElement_s bsr, uint8_t componentCarrierId)
487 {
488  NS_LOG_FUNCTION (this);
489  NS_ASSERT_MSG (componentCarrierId == 0, "Received BSR from a ComponentCarrier not allowed, ComponentCarrierId = " << componentCarrierId);
490  NS_ASSERT_MSG (bsr.m_macCeType == MacCeListElement_s::BSR, "Received a Control Message not allowed " << bsr.m_macCeType);
491 
492  // split traffic in uplink equally among carriers
493  uint32_t numberOfCarriersForUe = m_enabledComponentCarrier.find (bsr.m_rnti)->second;
494 
495  if ( bsr.m_macCeType == MacCeListElement_s::BSR)
496  {
497  MacCeListElement_s newBsr;
498  newBsr.m_rnti = bsr.m_rnti;
499  // mac control element type, values can be BSR, PHR, CRNTI
500  newBsr.m_macCeType = bsr.m_macCeType;
501  // the power headroom, 64 means no valid phr is available
502  newBsr.m_macCeValue.m_phr = bsr.m_macCeValue.m_phr;
503  // indicates that the CRNTI MAC CE was received. The value is not used.
504  newBsr.m_macCeValue.m_crnti = bsr.m_macCeValue.m_crnti;
505  // and value 64 means that the buffer status should not be updated
506  newBsr.m_macCeValue.m_bufferStatus.resize (4);
507  // always all 4 LCGs are present see 6.1.3.1 of 3GPP TS 36.321.
508  for (uint16_t i = 0; i < 4; i++)
509  {
510  uint8_t bsrStatusId = bsr.m_macCeValue.m_bufferStatus.at (i);
511  uint32_t bufferSize = BufferSizeLevelBsr::BsrId2BufferSize (bsrStatusId);
512  // here the buffer should be divide among the different sap
513  // since the buffer status report are compressed information
514  // it is needed to use BsrId2BufferSize to uncompress
515  // after the split over all component carriers is is needed to
516  // compress again the information to fit MacCeListElement_s structure
517  // verify how many Component Carrier are enabled per UE
518  newBsr.m_macCeValue.m_bufferStatus.at (i) = BufferSizeLevelBsr::BufferSize2BsrId (bufferSize / numberOfCarriersForUe);
519  }
520  // notify MAC of each component carrier that is enabled for this UE
521  for ( uint16_t i = 0; i < numberOfCarriersForUe; i++)
522  {
523  NS_ASSERT_MSG (m_ccmMacSapProviderMap.find (i) != m_ccmMacSapProviderMap.end (), "Mac sap provider does not exist.");
524  m_ccmMacSapProviderMap.find (i)->second->ReportMacCeToScheduler (newBsr);
525  }
526  }
527  else
528  {
529  auto ueManager = m_ccmRrcSapUser->GetUeManager (bsr.m_rnti);
530  m_ccmMacSapProviderMap.at (ueManager->GetComponentCarrierId ())->ReportMacCeToScheduler (bsr);
531  }
