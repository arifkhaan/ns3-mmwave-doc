51 void updateSnr (double snrInit, Ptr<MmWaveEnbNetDevice> enbDev, Ptr<SpectrumModel> model, Ptr<MmWaveAmc> amc)
52 {
53  Ptr<MmWavePhyMacCommon> config = enbDev->GetPhy ()->GetConfigurationParameters ();
54 
55  SpectrumValue specVals (model);
56  Values::iterator specIt = specVals.ValuesBegin ();
57  for (unsigned ichunk = 0; ichunk < model->GetNumBands (); ichunk++)
58  {
59  NS_ASSERT (specIt != specVals.ValuesEnd ());
60  *specIt = std::pow (10, snrInit / 10); //sinrLin;
61  specIt++;
62  }
63 
64  int mcsTmp = 28;
65  int mcs = 0;
66  int mcsOrig = 0;
67  while (true)
68  {
69  int tbs = amc->GetTbSizeFromMcsSymbols (mcsTmp, 22) / 8;
70  amc->CreateCqiFeedbackWbTdma (specVals, 22, tbs, mcs);
71  if (mcs == mcsTmp || mcs == mcsOrig)
72  {
73  break;
74  }
75  mcsOrig = mcsTmp;
76  mcsTmp = mcs;
77  }
78  enbDev->GetMac ()->SetMcs (mcs);
79  std::cout << "************* SINR changing to " << snrInit << " (MCS = " << mcs << " ) *************" << std::endl;
80 
81  Simulator::Schedule (MicroSeconds (3 * 100), &MmWaveSpectrumPhy::UpdateSinrPerceived,
82  enbDev->GetPhy ()->GetDlSpectrumPhy (), specVals);
83 
84  if (snrInit > snrMaxDb)
85  {
86  return;
87  }
88 
89  Simulator::Schedule (MilliSeconds (snrUpdateIntervalMs), &updateSnr, snrInit + 0.1, enbDev, model, amc);
90 }
