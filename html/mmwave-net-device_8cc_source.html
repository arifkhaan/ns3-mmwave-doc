229 void
230 MmWaveNetDevice::Receive (Ptr<Packet> p)
231 {
232  NS_LOG_FUNCTION (this << p);
233  Ipv4Header ipv4Header;
234  Ipv6Header ipv6Header;
235 
236  if (p->PeekHeader (ipv4Header) != 0)
237  {
238  NS_LOG_LOGIC ("IPv4 stack...");
239  m_rxCallback (this, p, Ipv4L3Protocol::PROT_NUMBER, Address ());
240  }
241  else if (p->PeekHeader (ipv6Header) != 0)
242  {
243  NS_LOG_LOGIC ("IPv6 stack...");
244  m_rxCallback (this, p, Ipv6L3Protocol::PROT_NUMBER, Address ());
245  }
246  else
247  {
248  NS_ABORT_MSG ("MmWaveNetDevice::Receive - Unknown IP type...");
249  }
250 }
