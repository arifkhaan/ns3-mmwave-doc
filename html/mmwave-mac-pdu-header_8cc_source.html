64 void
65 MmWaveMacPduHeader::Serialize (Buffer::Iterator i) const
66 {
67  // RDF TODO: implement BSR MAC control elements
68 
69  // this part builds the MAC sub-header format from 36.321 sec 6.1.2
70  for (unsigned ipdu = 0; ipdu < m_subheaderList.size (); ipdu++)
71  {
72  // first octet R/R/E/LCID (R=reserved bit, E=extension bit)
73  uint8_t octet1 = m_subheaderList[ipdu].m_lcid & 0x1F;
74  if (ipdu < (m_subheaderList.size () - 1)) // not the last subheader
75  {
76  octet1 |= (1 << 5);
77  }
78  i.WriteU8 (octet1);
79  // second octet F/Length (F=1 if length > 127 bits)
80  uint8_t octet2 = (m_subheaderList[ipdu].m_size & 0x7F);
81  // third octet = upper 8 bits of length if length > 127
82  uint8_t octet3 = 0;
83  if (m_subheaderList[ipdu].m_size > 0x7F)
84  {
85  octet2 |= (1 << 7);
86  octet3 = ((m_subheaderList[ipdu].m_size >> 7) & 0x7F);
87  i.WriteU8 (octet2);
88  if (m_subheaderList[ipdu].m_size > 0x3FFF)
89  {
90  octet3 |= (1 << 7); // write extension flag 2
91  uint8_t octet4 = ((m_subheaderList[ipdu].m_size >> 14) & 0xFF);
92  i.WriteU8 (octet3);
93  i.WriteU8 (octet4);
94  }
95  else
96  {
97  i.WriteU8 (octet3);
98  }
99  }
100  else
101  {
102  i.WriteU8 (octet2);
103  }
104  }
105 }

107 uint32_t
108 MmWaveMacPduHeader::Deserialize (Buffer::Iterator i)
109 {
110  // decode sub-headers and create RLC info elements
111  m_headerSize = 0;
112  bool done = false;
113  while (!done)
114  {
115  uint8_t octet1 = (uint8_t)i.ReadU8 ();
116  uint8_t lcid = (octet1 & 0x1F);
117  if (!(octet1 >> 5)) // extension bit not set
118  {
119  done = true;
120  }
121  uint8_t octet2 = (uint8_t)i.ReadU8 ();
122  bool flag = octet2 >> 7;
123  uint32_t size = (octet2 & 0x7F);
124  if (flag) // size > 127
125  {
126  uint8_t octet3 = (uint8_t)i.ReadU8 ();
127  size |= (octet3 & 0x7F) << 7;
128  bool flag2 = (octet3 >> 7);
129  if (flag2)
130  {
131  uint8_t octet4 = (uint8_t)i.ReadU8 ();
132  size |= (octet4 << 14);
133  m_headerSize += 4;
134  }
135  else
136  {
137  m_headerSize += 3;
138  }
139  }
140  else
141  {
142  m_headerSize += 2;
143  }
144  m_subheaderList.push_back (MacSubheader (lcid, size));
145  }
146 
147  return m_headerSize;
148 }
