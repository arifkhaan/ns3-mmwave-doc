215  double Rcode = McsEcrTable[mcs];
216  double Qm = ModulationSchemeForMcs[mcs];
217  uint16_t cbSize = 6144; //max size of a code-block (including m_crcLen)
218  if (tbSize > cbSize)

228 std::vector<int>
229 MmWaveAmc::CreateCqiFeedbacks (const SpectrumValue& sinr, uint8_t rbgSize)
230 {
231  NS_LOG_FUNCTION (this);
232 
233  std::vector<int> cqi;
234  Values::const_iterator it;
235  if (m_amcModel == PiroEW2010)
236  {
237  //use PiroEW2010 model
238  for (it = sinr.ConstValuesBegin (); it != sinr.ConstValuesEnd (); it++)
239  {
240  double sinr_ = (*it);
241  if (sinr_ == 0.0)
242  {
243  cqi.push_back (-1); // SINR == 0 (linear units) means no signal in this RB
244  }
245  else
246  {
247  /*
248  * Compute the spectral efficiency from the SINR
249  * SINR
250  * spectralEfficiency = log2 (1 + -------------------- )
251  * -ln(5*BER)/1.5
252  * NB: SINR must be expressed in linear units
253  */
254 
255  double s = log2 ( 1 + ( sinr_ / ( (-std::log (5.0 * m_ber )) / 1.5) ));
256 
257  int cqi_ = GetCqiFromSpectralEfficiency (s);
258 
259  NS_LOG_LOGIC (" PRB =" << cqi.size ()
260  << ", sinr = " << sinr_
261  << " (=" << 10 * std::log10 (sinr_) << " dB)"
262  << ", spectral efficiency =" << s
263  << ", CQI = " << cqi_ << ", BER = " << m_ber);
264 
265  cqi.push_back (cqi_);
266  }
267  }
268  }
269  else if (m_amcModel == MiErrorModel)
270  {
271  std::vector <int> rbgMap;
272  int rbId = 0;
273  for (it = sinr.ConstValuesBegin (); it != sinr.ConstValuesEnd (); it++)
274  {
275  rbgMap.push_back (rbId++);
276  if ((rbId % rbgSize == 0)||((it + 1) == sinr.ConstValuesEnd ()))
277  {
278  uint8_t mcs = 0;
279  MmWaveTbStats_t tbStats;
280  while (mcs <= 28)
281  {
282  MmWaveHarqProcessInfoList_t harqInfoList;
283  tbStats = MmWaveMiErrorModel::GetTbDecodificationStats (sinr, rbgMap, GetTbSizeFromMcs (mcs, rbgSize / 18) / 8, mcs, harqInfoList);
284  if (tbStats.tbler > 0.1)
285  {
286  break;
287  }
288  mcs++;
289 
290  }
291  if (mcs > 0)
292  {
293  mcs--;
294  }
295  NS_LOG_DEBUG (this << "\t RBG " << rbId << " MCS " << (uint16_t)mcs << " TBLER " << tbStats.tbler);
296  int rbgCqi = 0;
297  if ((tbStats.tbler > 0.1)&&(mcs == 0))
298  {
299  rbgCqi = 0;
300  }
301  else if (mcs == 28)
302  {
303  rbgCqi = 15; // all MCSs can guarantee the 10 % of BER
304  }
305  else
306  {
307  double s = SpectralEfficiencyForMcs[mcs];
308  rbgCqi = 0;
309  while ((rbgCqi < 15) && (SpectralEfficiencyForCqi[rbgCqi + 1] < s))
310  {
311  ++rbgCqi;
312  }
313  }
314  NS_LOG_DEBUG (this << "\t MCS " << (uint16_t)mcs << "-> CQI " << rbgCqi);
315  // fill the cqi vector (per RB basis)
316  for (uint8_t j = 0; j < rbgSize; j++)
317  {
318  cqi.push_back (rbgCqi);
319  }
320  rbgMap.clear ();
321  }
322 
323  }
324  }
325  return cqi;
326 }
