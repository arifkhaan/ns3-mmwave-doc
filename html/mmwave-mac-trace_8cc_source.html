62 void
63 MmWaveMacTrace::ReportEnbSchedulingInfo (Ptr<MmWaveMacTrace> enbStats, MmWaveEnbMac::MmWaveSchedTraceInfo schedParams)
64 {
65  // Open the output file if it is not open yet
66  if (!m_schedAllocTraceFile.is_open ())
67  {
68  m_schedAllocTraceFile.open (m_schedAllocTraceFilename.c_str ());
69  if (!m_schedAllocTraceFile.is_open ())
70  {
71  NS_FATAL_ERROR ("Could not open tracefile");
72  }
73  m_schedAllocTraceFile << "frame\tsubF\tslot\trnti\tfirstSym\tnumSym\ttype\ttddMode\tretxNum\tccId" << std::endl;
74  }
75 
76 
77  SlotAllocInfo allocInfo = schedParams.m_indParam.m_slotAllocInfo;
78  SfnSf dlSfn = schedParams.m_indParam.m_sfnSf; // Holds the intended slot, subframe and frame info
79 
80  for (auto iTti : allocInfo.m_ttiAllocInfo)
81  {
82  // Trace the incoming alloc info
83  m_schedAllocTraceFile << +dlSfn.m_frameNum << "\t" << +dlSfn.m_sfNum << "\t"
84  << +dlSfn.m_slotNum << "\t" << +iTti.m_dci.m_rnti << "\t"
85  << +iTti.m_dci.m_symStart << "\t" << +iTti.m_dci.m_numSym << "\t"
86  << iTti.m_ttiType << "\t" << iTti.m_tddMode << "\t"
87  << +iTti.m_dci.m_rv << "\t" << +schedParams.m_ccId << std::endl;
88  }
89 }
