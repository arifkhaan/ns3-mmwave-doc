304 MmWaveFlexTtiMacScheduler::DoSchedDlRlcBufferReq (const struct MmWaveMacSchedSapProvider::SchedDlRlcBufferReqParameters& params)
305 {
306  NS_LOG_FUNCTION (this << params.m_rnti << (uint32_t) params.m_logicalChannelIdentity);
307  // API generated by RLC for updating RLC parameters on a LC (tx and retx queues)
308  std::list<MmWaveMacSchedSapProvider::SchedDlRlcBufferReqParameters>::iterator it = m_rlcBufferReq.begin ();
309  bool newLc = true;
310  while (it != m_rlcBufferReq.end ())
311  {
312  // remove old entries of this UE-LC
313  if (((*it).m_rnti == params.m_rnti)&&((*it).m_logicalChannelIdentity == params.m_logicalChannelIdentity))
314  {
315  it = m_rlcBufferReq.erase (it);
316  newLc = false;
317  }
318  else
319  {
320  ++it;
321  }
322  }
323  // add the new parameters
324  m_rlcBufferReq.insert (it, params);
325  NS_LOG_INFO ("BSR for RNTI " << params.m_rnti << " LC " << (uint16_t)params.m_logicalChannelIdentity << " RLC tx size " << params.m_rlcTransmissionQueueSize << " RLC retx size " << params.m_rlcRetransmissionQueueSize << " RLC stat size " << params.m_rlcStatusPduSize);
326  // initialize statistics of the flow in case of new flows
327  if (newLc == true)
328  {
329  m_wbCqiRxed.insert ( std::pair<uint16_t, uint8_t > (params.m_rnti, 1)); // only codeword 0 at this stage (SISO)
330  // initialized to 1 (i.e., the lowest value for transmitting a signal)
331  m_wbCqiTimers.insert ( std::pair<uint16_t, uint32_t > (params.m_rnti, m_cqiTimersThreshold));
332  }

353  // generate correspondent timer
354  m_wbCqiTimers.insert ( std::pair<uint16_t, uint32_t > (rnti, m_cqiTimersThreshold));
355  }
356  else

358  // update the CQI value
359  (*it).second = params.m_cqiList.at (i).m_wbCqi;
360  // update correspondent timer
361  std::map <uint16_t,uint32_t>::iterator itTimers;
362  itTimers = m_wbCqiTimers.find (rnti);
363  (*itTimers).second = m_cqiTimersThreshold;
364  }
365  }
366  else if ( params.m_cqiList.at (i).m_cqiType == DlCqiInfo::SB )
367  {
368  // subband CQI reporting high layer configured
369  // Not used by RR Scheduler
370  }
371  else
372  {
373  NS_LOG_ERROR (this << " CQI type unknown");
374  }

380 MmWaveFlexTtiMacScheduler::DoSchedUlCqiInfoReq (const struct MmWaveMacSchedSapProvider::SchedUlCqiInfoReqParameters& params)
381 {
382  NS_LOG_FUNCTION (this);
383 
384  uint16_t frameNum = params.m_sfnSf.m_frameNum;
385  uint8_t subframeNum = params.m_sfnSf.m_sfNum;
386  uint8_t slotNum = params.m_sfnSf.m_slotNum;
387  uint8_t symNum = params.m_sfnSf.m_symStart;
388 
389  switch (params.m_ulCqi.m_type)
390  {
391  case UlCqiInfo::PUSCH:
392  {
393  std::map <uint32_t, struct AllocMapElem>::iterator itMap;
394  std::map <uint16_t, struct UlCqiMapElem>::iterator itCqi;
395  itMap = m_ulAllocationMap.find (params.m_sfnSf.Encode ());
396  if (itMap == m_ulAllocationMap.end ())
397  {
398  NS_LOG_INFO (this << " Does not find info on allocation, size : " << m_ulAllocationMap.size ());
399  return;
400  }
401  NS_ASSERT_MSG (itMap->second.m_rntiPerChunk.size () == m_phyMacConfig->GetNumChunks (), "SINR chunk map must cover full BW in TDMA mode");
402  for (unsigned i = 0; i < itMap->second.m_rntiPerChunk.size (); i++)
403  {
404  // convert from fixed point notation Sxxxxxxxxxxx.xxx to double
405  //double sinr = LteFfConverter::fpS11dot3toDouble (params.m_ulCqi.m_sinr.at (i));
406  itCqi = m_ueUlCqi.find (itMap->second.m_rntiPerChunk.at (i));
407  if (itCqi == m_ueUlCqi.end ())
408  {
409  // create a new entry
410  std::vector <double> newCqi;
411  for (uint32_t j = 0; j < m_phyMacConfig->GetNumChunks (); j++)
412  {
413  unsigned chunkInd = i;
414  if (chunkInd == j)
415  {
416  newCqi.push_back (params.m_ulCqi.m_sinr.at (i));
417  NS_LOG_INFO ("UL CQI report for RNTI " << itMap->second.m_rntiPerChunk.at (i) << " chunk " << i << " SINR " << params.m_ulCqi.m_sinr.at (i) << \
418  " frame " << frameNum << " subframe " << (unsigned)subframeNum << " slot " << (unsigned)slotNum << " startSym " << (unsigned)symNum);
419  }
420  else
421  {
422  // initialize with NO_SINR value.
423  newCqi.push_back (30.0);
424  }
425  }
426  m_ueUlCqi.insert (std::pair <uint16_t, struct UlCqiMapElem> (itMap->second.m_rntiPerChunk.at (i),
427  UlCqiMapElem (newCqi, itMap->second.m_numSym, itMap->second.m_tbSize)) );
428  // generate correspondent timer
429  m_ueCqiTimers.insert (std::pair <uint16_t, uint32_t > (itMap->second.m_rntiPerChunk.at (i), m_cqiTimersThreshold));
430  }
431  else
432  {
433  // update the value
434  (*itCqi).second.m_ueUlCqi.at (i) = params.m_ulCqi.m_sinr.at (i);
435  (*itCqi).second.m_numSym = itMap->second.m_numSym;
436  (*itCqi).second.m_tbSize = itMap->second.m_tbSize;
437  // update correspondent timer
438  std::map <uint16_t, uint32_t>::iterator itTimers;
439  itTimers = m_ueCqiTimers.find (itMap->second.m_rntiPerChunk.at (i));
440  (*itTimers).second = m_cqiTimersThreshold;
441 
442  NS_LOG_INFO ("UL CQI report for RNTI " << itMap->second.m_rntiPerChunk.at (i) << " chunk " << i << " SINR " << params.m_ulCqi.m_sinr.at (i) << \
443  " frame " << frameNum << " subframe " << (unsigned)subframeNum << " slot " << (unsigned)slotNum << " startSym " << (unsigned)symNum);
444 
445  }
446 
447  }
448  // remove obsolete info on allocation
449  m_ulAllocationMap.erase (itMap);
450  }
451  break;
452  default:
453  NS_FATAL_ERROR ("Unknown type of UL-CQI");
454  }
455  return;

605 unsigned MmWaveFlexTtiMacScheduler::CalcMinTbSizeNumSym (unsigned mcs, unsigned bufSize, unsigned &tbSize)
606 {
607  // Bisection line search is used to find the minimum number of slots (OFDM symbols)
608  // needed to encode entire buffer.
609  MmWaveMacPduHeader dummyMacHeader;
610  //unsigned macHdrSize = 10; //dummyMacHeader.GetSerializedSize ();

613 
614  int diff = 0;
615  tbSize = (m_amc->GetTbSizeFromMcsSymbols (mcs, numSymHigh) / 8); // start with max value, in number of bytes
616  while ((unsigned)tbSize > bufSize)
617  {
618  diff = abs(numSymHigh-numSymLow)/2;
619  if (diff == 0)
620  {
621  tbSize = (m_amc->GetTbSizeFromMcsSymbols (mcs, numSymHigh) / 8);
622  return numSymHigh;
623  }
624  tbSize = (m_amc->GetTbSizeFromMcsSymbols (mcs, numSymHigh - diff) / 8);
625  if ((unsigned)tbSize >= bufSize)
626  {
627  numSymHigh -= diff;
628  }
629  if ((unsigned) tbSize == bufSize)
630  {
631  return numSymHigh;
632  }
633  while ((unsigned)tbSize < bufSize)
634  {
635  diff = abs(numSymHigh-numSymLow)/2;
636  if (diff == 0)
637  {
638  tbSize = (m_amc->GetTbSizeFromMcsSymbols (mcs, numSymHigh) / 8);
639  return numSymHigh;
640  }
641  tbSize = (m_amc->GetTbSizeFromMcsSymbols (mcs, numSymLow + diff) / 8);
642  if ((unsigned)tbSize <= bufSize)
643  {
644  numSymLow += diff;
645  }
646  if ((unsigned) tbSize == bufSize)
647  {
648  return numSymLow;
649  }
650  }
651  }
652  tbSize = (m_amc->GetTbSizeFromMcsSymbols (mcs, numSymHigh) / 8);
