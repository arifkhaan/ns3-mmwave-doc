65 void randomPos (Ptr<Node> n, int R_max, int z)
66 {
67  Ptr<UniformRandomVariable> rand = CreateObject<UniformRandomVariable> ();
68  // already U[0,1] by default
69  double rho = R_max * std::sqrt (rand->GetValue ());
70  double theta = rand->GetValue ();
71  double x = rho * std::cos (theta);
72  double y = rho * std::sin (theta);
73  Ptr<ConstantPositionMobilityModel> loc = CreateObject<ConstantPositionMobilityModel> ();
74  n->AggregateObject (loc);
75  Vector locVec2 (x, y, z);
76  NS_LOG_UNCOND ("UE in Position " << locVec2);
77  loc->SetPosition (locVec2);
78 }
