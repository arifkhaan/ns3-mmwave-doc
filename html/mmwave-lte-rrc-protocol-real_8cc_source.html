272 void
273 MmWaveLteUeRrcProtocolReal::SetEnbRrcSapProvider ()
274 {
275  uint16_t cellId = m_rrc->GetCellId ();
276 
277  // walk list of all nodes to get the peer eNB
278  Ptr<LteEnbNetDevice> enbDev;
279  Ptr<MmWaveEnbNetDevice> mmWaveEnbDev;
280  NodeList::Iterator listEnd = NodeList::End ();
281  bool found = false;
282  for (NodeList::Iterator i = NodeList::Begin ();
283  (i != listEnd) && (!found);
284  ++i)
285  {
286  Ptr<Node> node = *i;
287  int nDevs = node->GetNDevices ();
288  for (int j = 0;
289  (j < nDevs) && (!found);
290  j++)
291  {
292  enbDev = node->GetDevice (j)->GetObject <LteEnbNetDevice> ();
293  if (enbDev == 0)
294  {
295  // check if it is a MmWave eNB
296  mmWaveEnbDev = node->GetDevice (j)->GetObject <MmWaveEnbNetDevice> ();
297  if (mmWaveEnbDev == 0)
298  {
299  continue;
300  }
301  else
302  {
303  if (mmWaveEnbDev->HasCellId (cellId))
304  {
305  found = true;
306  break;
307  }
308  }
309  }
310  else
311  {
312  if (enbDev->GetCellId () == cellId)
313  {
314  found = true;
315  break;
316  }
317  }
318  }
319  }
320  NS_ASSERT_MSG (found, " Unable to find eNB with CellId =" << cellId);
321  if (enbDev != 0)
322  {
323  m_enbRrcSapProvider = enbDev->GetRrc ()->GetLteEnbRrcSapProvider ();
324  Ptr<MmWaveLteEnbRrcProtocolReal> enbRrcProtocolReal = enbDev->GetRrc ()->GetObject<MmWaveLteEnbRrcProtocolReal> ();
325  enbRrcProtocolReal->SetUeRrcSapProvider (m_rnti, m_ueRrcSapProvider);
326  }
327  else if (mmWaveEnbDev != 0)
328  {
329  m_enbRrcSapProvider = mmWaveEnbDev->GetRrc ()->GetLteEnbRrcSapProvider ();
330  Ptr<MmWaveLteEnbRrcProtocolReal> enbRrcProtocolReal = mmWaveEnbDev->GetRrc ()->GetObject<MmWaveLteEnbRrcProtocolReal> ();
331  enbRrcProtocolReal->SetUeRrcSapProvider (m_rnti, m_ueRrcSapProvider);
332  }
333  else
334  {
335  NS_FATAL_ERROR ("Nor LTE eNB neither MmWave eNB");
336  }
337 }

339 void
340 MmWaveLteUeRrcProtocolReal::DoReceivePdcpPdu (Ptr<Packet> p)
341 {
342  NS_LOG_FUNCTION (this);
343  // Get type of message received
344  RrcDlCcchMessage rrcDlCcchMessage;
345  p->PeekHeader (rrcDlCcchMessage);
346  NS_LOG_INFO (rrcDlCcchMessage);
347 
348 
349  // Declare possible headers to receive
350  RrcConnectionReestablishmentHeader rrcConnectionReestablishmentHeader;
351  RrcConnectionReestablishmentRejectHeader rrcConnectionReestablishmentRejectHeader;
352  RrcConnectionSetupHeader rrcConnectionSetupHeader;
353  RrcConnectionRejectHeader rrcConnectionRejectHeader;
354  RrcConnectToMmWaveHeader rrcConnectToMmWaveHeader;
355 
356  // Declare possible messages
357  LteRrcSap::RrcConnectionReestablishment rrcConnectionReestablishmentMsg;
358  LteRrcSap::RrcConnectionReestablishmentReject rrcConnectionReestablishmentRejectMsg;
359  LteRrcSap::RrcConnectionSetup rrcConnectionSetupMsg;
360  LteRrcSap::RrcConnectionReject rrcConnectionRejectMsg;
361 
362  // Deserialize packet and call member recv function with appropiate structure
363  switch ( rrcDlCcchMessage.GetMessageType () )
364  {
365  case 0:
366  // RrcConnectionReestablishment
367  p->RemoveHeader (rrcConnectionReestablishmentHeader);
368  rrcConnectionReestablishmentMsg = rrcConnectionReestablishmentHeader.GetMessage ();
369  m_ueRrcSapProvider->RecvRrcConnectionReestablishment (rrcConnectionReestablishmentMsg);
370  break;
371  case 1:
372  // RrcConnectionReestablishmentReject
373  p->RemoveHeader (rrcConnectionReestablishmentRejectHeader);
374  rrcConnectionReestablishmentRejectMsg = rrcConnectionReestablishmentRejectHeader.GetMessage ();
375  // m_ueRrcSapProvider->RecvRrcConnectionReestablishmentReject (rrcConnectionReestablishmentRejectMsg);
376  break;
377  case 2:
378  // RrcConnectionReject
379  p->RemoveHeader (rrcConnectionRejectHeader);
380  rrcConnectionRejectMsg = rrcConnectionRejectHeader.GetMessage ();
381  m_ueRrcSapProvider->RecvRrcConnectionReject (rrcConnectionRejectMsg);
382  break;
383  case 3:
384  // RrcConnectionSetup
385  p->RemoveHeader (rrcConnectionSetupHeader);
386  rrcConnectionSetupMsg = rrcConnectionSetupHeader.GetMessage ();
387  m_ueRrcSapProvider->RecvRrcConnectionSetup (rrcConnectionSetupMsg);
388  break;
389  case 4:
390  // RrcConnectToMmWave
391  p->RemoveHeader (rrcConnectToMmWaveHeader);
392  uint16_t mmWaveCellId = rrcConnectToMmWaveHeader.GetMessage ();
393  NS_LOG_INFO ("Recv RRC connect to MmWave cellId " << mmWaveCellId);
394  m_ueRrcSapProvider->RecvRrcConnectToMmWave (mmWaveCellId);
395  break;
396  }
397 }
