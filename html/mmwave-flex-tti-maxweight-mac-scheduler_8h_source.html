214  static bool CompareFlowWeightsEdf (FlowStats* lflow, FlowStats* rflow)
215  {
216  int lRelDeadline = lflow->m_deadlineUs - lflow->m_txQueueHolDelay;
217  int rRelDeadline = rflow->m_deadlineUs - rflow->m_txQueueHolDelay;
218  return (lRelDeadline < rRelDeadline); // earlier deadline = greater weight
219  }
220 
221  static bool CompareFlowWeightsDeliveryDebt (FlowStats* lflow, FlowStats* rflow)
222  {
223  int lflowDebt = (lflow->m_arrivalRate / (1 - lflow->m_probErr)) - lflow->m_grantedRate;
224  int rflowDebt = (rflow->m_arrivalRate / (1 - rflow->m_probErr)) - rflow->m_grantedRate;
225  return (lflowDebt > rflowDebt);
226  }
