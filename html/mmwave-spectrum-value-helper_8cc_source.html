55 Ptr<SpectrumModel>
56 MmWaveSpectrumValueHelper::GetSpectrumModel (Ptr<MmWavePhyMacCommon> ptrConfig)
57 {
58  NS_LOG_FUNCTION (ptrConfig->GetCenterFrequency () << (uint32_t) ptrConfig->GetNumChunks ());
59  uint8_t ccId = ptrConfig->GetCcId ();
60  if (m_model.find (ccId) != m_model.end () )
61  {
62  if (m_model[ccId]->GetNumBands () != 0)
63  {
64  NS_LOG_DEBUG ("CC " << (uint32_t)ptrConfig->GetCcId () << " NumBands " << (uint32_t)m_model[ccId]->GetNumBands () );
65  return m_model[ccId];
66  }
67  }
68 
69  double fc = ptrConfig->GetCenterFrequency ();
70  double f = 0.00;
71 
72  NS_ASSERT_MSG (fc != 0, "The carrier frequency cannot be set to 0");
73 
74  f = fc - (ptrConfig->GetNumChunks () * ptrConfig->GetChunkWidth () / 2.0);
75 
76  Bands rbs; // A vector representing each resource block
77  for (uint32_t numrb = 0; numrb < ptrConfig->GetNumChunks (); ++numrb)
78  {
79  BandInfo rb;
80  rb.fl = f;
81  f += ptrConfig->GetChunkWidth () / 2;
82  rb.fc = f;
83  f += ptrConfig->GetChunkWidth () / 2;
84  rb.fh = f;
85 
86  rbs.push_back (rb);
87  }
88  NS_LOG_DEBUG ("CC " << (uint32_t)ptrConfig->GetCcId () << " rbs size " << (uint32_t)rbs.size () );
89  m_model[ccId] = Create<SpectrumModel> (rbs);
90  return m_model[ccId];
91 }

132 Ptr<SpectrumValue>
133 MmWaveSpectrumValueHelper::CreateNoisePowerSpectralDensity (double noiseFigureDb, Ptr<SpectrumModel> spectrumModel)
134 {
135  NS_LOG_FUNCTION (noiseFigureDb << spectrumModel);
136  const double kT_dBm_Hz = -174.0; // dBm/Hz
137  double kT_W_Hz = std::pow (10.0, (kT_dBm_Hz - 30) / 10.0);
138  double noiseFigureLinear = std::pow (10.0, noiseFigureDb / 10.0);
139  double noisePowerSpectralDensity = kT_W_Hz * noiseFigureLinear;
140 
141  Ptr<SpectrumValue> noisePsd = Create <SpectrumValue> (spectrumModel);
142  (*noisePsd) = noisePowerSpectralDensity;
143  return noisePsd;
144 }
