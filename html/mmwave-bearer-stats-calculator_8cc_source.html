299 MmWaveBearerStatsCalculator::ShowResults (void)
300 {
301 
302  NS_LOG_FUNCTION (this << GetUlOutputFilename ().c_str () << GetDlOutputFilename ().c_str ());
303  NS_LOG_INFO ("Write Rlc Stats in " << GetUlOutputFilename ().c_str () << " and in " << GetDlOutputFilename ().c_str ());
304 
305  std::ofstream ulOutFile;
306  std::ofstream dlOutFile;
307 
308  if (m_firstWrite == true)
309  {
310  ulOutFile.open (GetUlOutputFilename ().c_str ());
311  if (!ulOutFile.is_open ())
312  {
313  NS_LOG_ERROR ("Can't open file " << GetUlOutputFilename ().c_str ());
314  return;
315  }
316 
317  dlOutFile.open (GetDlOutputFilename ().c_str ());
318  if (!dlOutFile.is_open ())
319  {
320  NS_LOG_ERROR ("Can't open file " << GetDlOutputFilename ().c_str ());
321  return;
322  }
323  m_firstWrite = false;
324  ulOutFile << "% start\tend\tCellId\tIMSI\tRNTI\tLCID\tnTxPDUs\tTxBytes\tnRxPDUs\tRxBytes\t";
325  ulOutFile << "delay\tstdDev\tmin\tmax\t";
326  ulOutFile << "PduSize\tstdDev\tmin\tmax";
327  ulOutFile << std::endl;
328  dlOutFile << "% start\tend\tCellId\tIMSI\tRNTI\tLCID\tnTxPDUs\tTxBytes\tnRxPDUs\tRxBytes\t";
329  dlOutFile << "delay\tstdDev\tmin\tmax\t";
330  dlOutFile << "PduSize\tstdDev\tmin\tmax";
331  dlOutFile << std::endl;
332  }
333  else
334  {
335  ulOutFile.open (GetUlOutputFilename ().c_str (), std::ios_base::app);
336  if (!ulOutFile.is_open ())
337  {
338  NS_LOG_ERROR ("Can't open file " << GetUlOutputFilename ().c_str ());
339  return;
340  }
341 
342  dlOutFile.open (GetDlOutputFilename ().c_str (), std::ios_base::app);
343  if (!dlOutFile.is_open ())
344  {
345  NS_LOG_ERROR ("Can't open file " << GetDlOutputFilename ().c_str ());

347  }
348  }
349 
350  WriteUlResults (ulOutFile);

356 void
357 MmWaveBearerStatsCalculator::WriteUlResults (std::ofstream& outFile)
358 {
359  NS_LOG_FUNCTION (this);
360 
361  // Get the unique IMSI / LCID list
362 
363  std::vector < ImsiLcidPair_t > pairVector;
364  for (Uint32Map::iterator it = m_ulTxPackets.begin (); it != m_ulTxPackets.end (); ++it)
365  {
366  if (find (pairVector.begin (), pairVector.end (), (*it).first) == pairVector.end ())
367  {
368  pairVector.push_back ((*it).first);
369  }
370  }
371 
372  Time endTime = m_startTime + m_epochDuration;
373  for (std::vector<ImsiLcidPair_t>::iterator it = pairVector.begin (); it != pairVector.end (); ++it)
374  {
375  ImsiLcidPair_t p = *it;
376  outFile << m_startTime.GetNanoSeconds () / 1.0e9 << "\t";
377  outFile << endTime.GetNanoSeconds () / 1.0e9 << "\t";
378  outFile << GetUlCellId (p.m_imsi, p.m_lcId) << "\t";
379  outFile << p.m_imsi << "\t";
380  outFile << m_flowId[p].m_rnti << "\t";
381  outFile << (uint32_t) m_flowId[p].m_lcId << "\t";
382  outFile << GetUlTxPackets (p.m_imsi, p.m_lcId) << "\t";
383  outFile << GetUlTxData (p.m_imsi, p.m_lcId) << "\t";
384  outFile << GetUlRxPackets (p.m_imsi, p.m_lcId) << "\t";
385  outFile << GetUlRxData (p.m_imsi, p.m_lcId) << "\t";
386  std::vector<double> stats = GetUlDelayStats (p.m_imsi, p.m_lcId);
387  for (std::vector<double>::iterator it = stats.begin (); it != stats.end (); ++it)
388  {
389  outFile << (*it) * 1e-9 << "\t";
390  }
391  stats = GetUlPduSizeStats (p.m_imsi, p.m_lcId);
392  for (std::vector<double>::iterator it = stats.begin (); it != stats.end (); ++it)
393  {
394  outFile << (*it) << "\t";
395  }
396  outFile << std::endl;
397  }
398 
399  outFile.close ();
400 }
