60 void
61 MmWaveAttachmentTestCase::DoRun (void)
62 {
63  // Create two UE-BS pairs, call the method AttachToClosestEnb () and check
64  // if the UEs are correctly attached.
65 
66  // (0,20,1.6)--------------------(100,20,1.6)
67  // UE1-----------------------------UE2
68  // | |
69  // BS1-----------------------------BS2
70  // (0,0,25)----------------------(100,0,25)
71 
72  uint8_t numCc = 2; // number of CCs
73  double freq[] {6.0e9, 28.0e9}; // frequency of the CCs
74 
75  // create the MmWaveHelper
76  Ptr<MmWaveHelper> helper = CreateObject<MmWaveHelper> ();
77  helper->SetAttribute ("UseCa", BooleanValue ((numCc > 1)));
78  helper->SetAttribute ("NumberOfComponentCarriers", UintegerValue (numCc));
79 
80  // create and configure the CCs
81  std::map<uint8_t, MmWaveComponentCarrier> ccMap;
82  for (uint8_t i = 0; i < numCc; i++)
83  {
84  Ptr<MmWavePhyMacCommon> phyMacConfig = CreateObject<MmWavePhyMacCommon> ();
85  phyMacConfig->SetAttribute ("CenterFreq", DoubleValue (freq [i]));
86 
87  Ptr<MmWaveComponentCarrier> cc = CreateObject<MmWaveComponentCarrier> ();
88  cc->SetConfigurationParameters (phyMacConfig);
89  cc->SetAsPrimary ((i == 0));
90  ccMap [i] = *cc;
91  }
92 
93  // assign the map of CCs to the helper
94  helper->SetCcPhyParams (ccMap);
95 
96  // choose the pathloss model to use
97  helper->SetPathlossModelType ("ns3::ThreeGppUmaPropagationLossModel");
98 
99  // choose the channel condition model to use
100  helper->SetChannelConditionModelType ("ns3::ThreeGppUmaChannelConditionModel");
101 
102  // choose the spectrum propagation loss model
103  helper->SetChannelModelType ("ns3::ThreeGppSpectrumPropagationLossModel");
104 
105  // create the BS nodes
106  NodeContainer bsNodes;
107  bsNodes.Create (2);
108 
109  // create a ListPositionAllocator and add the position of the BS
110  Ptr<ListPositionAllocator> bsPositionAlloc = CreateObject<ListPositionAllocator> ();
111  bsPositionAlloc->Add (Vector (0.0, 0.0, 25.0));
112  bsPositionAlloc->Add (Vector (100.0, 0.0, 25.0));
113 
114  // use a MobilityHelper to install the MobilityModel for the BS
115  MobilityHelper bsMobility;
116  bsMobility.SetMobilityModel ("ns3::ConstantPositionMobilityModel");
117  bsMobility.SetPositionAllocator (bsPositionAlloc);
118  bsMobility.Install (bsNodes);
119 
120  // install the BS device on the proper Node
121  NetDeviceContainer bsNetDevs = helper->InstallEnbDevice (bsNodes);
122 
123  // create the UE nodes
124  NodeContainer ueNodes;
125  ueNodes.Create (2);
126 
127  // create a ListPositionAllocator and add the position of the UEs
128  Ptr<ListPositionAllocator> uePositionAlloc = CreateObject<ListPositionAllocator> ();
129  uePositionAlloc->Add (Vector (0.0, 20.0, 1.6));
130  uePositionAlloc->Add (Vector (100.0, 20.0, 1.6));
131 
132  // use a MobilityHelper to install the MobilityModel for the UEs
133  MobilityHelper ueMobility;
134  ueMobility.SetMobilityModel ("ns3::ConstantPositionMobilityModel");
135  ueMobility.SetPositionAllocator (uePositionAlloc);
136  ueMobility.Install (ueNodes);
137 
138  // install the UE device on the proper Node
139  NetDeviceContainer ueNetDevs = helper->InstallUeDevice (ueNodes);
140 
141  // attach the UE
142  helper->AttachToClosestEnb (ueNetDevs, bsNetDevs);
143 
144  // check if the UEs are correctly attached
145  Ptr<MmWaveUeNetDevice> mmWaveUeDev1 = DynamicCast<MmWaveUeNetDevice> (ueNetDevs.Get (0));
146  Ptr<MmWaveEnbNetDevice> targetBs1 = mmWaveUeDev1->GetTargetEnb ();
147  NS_TEST_ASSERT_MSG_EQ (bsNetDevs.Get (0), targetBs1, "UE 1 should be attached to BS 1");
148 
149  Ptr<MmWaveUeNetDevice> mmWaveUeDev2 = DynamicCast<MmWaveUeNetDevice> (ueNetDevs.Get (1));
150  Ptr<MmWaveEnbNetDevice> targetBs2 = mmWaveUeDev2->GetTargetEnb ();
151  NS_TEST_ASSERT_MSG_EQ (bsNetDevs.Get (1), targetBs2, "UE 2 should be attached to BS 2");
152 
153 }
