81 void
82 mmWaveInterference::StartRx (Ptr<const SpectrumValue> rxPsd)
83 {
84  NS_LOG_FUNCTION (this << *rxPsd);
85  if (m_receiving == false)
86  {
87  NS_LOG_LOGIC ("first signal");
88  m_rxSignal = rxPsd->Copy ();
89  m_lastChangeTime = Now ();
90  m_receiving = true;
91  for (std::list<Ptr<mmWaveChunkProcessor> >::const_iterator it = m_PowerChunkProcessorList.begin (); it != m_PowerChunkProcessorList.end (); ++it)
92  {
93  (*it)->Start ();
94  }
95  for (std::list<Ptr<mmWaveChunkProcessor> >::const_iterator it = m_sinrChunkProcessorList.begin (); it != m_sinrChunkProcessorList.end (); ++it)
96  {
97  (*it)->Start ();
98  }
99  }
100  else
101  {
102  NS_LOG_LOGIC ("additional signal" << *m_rxSignal);
103  // receiving multiple simultaneous signals, make sure they are synchronized
104  NS_ASSERT (m_lastChangeTime == Now ());
105  // make sure they use orthogonal resource blocks
106  NS_ASSERT (Sum ((*rxPsd) * (*m_rxSignal)) == 0.0);
107  (*m_rxSignal) += (*rxPsd);
108  }
109 }

113 mmWaveInterference::EndRx ()
114 {
115  NS_LOG_FUNCTION (this);

118  NS_LOG_INFO ("EndRx was already evaluated or RX was aborted");

136 void
137 mmWaveInterference::AddSignal (Ptr<const SpectrumValue> spd, const Time duration)
138 {
139  NS_LOG_FUNCTION (this << *spd << duration);
140  DoAddSignal (spd);
141  uint32_t signalId = ++m_lastSignalId;
142  if (signalId == m_lastSignalIdBeforeReset)
143  {
144  // This happens when m_lastSignalId eventually wraps around. Given that so
145  // many signals have elapsed since the last reset, we hope that by now there is
146  // no stale pending signal (i.e., a signal that was scheduled
147  // for subtraction before the reset). So we just move the
148  // boundary further.
149  m_lastSignalIdBeforeReset += 0x10000000;
150  }
151  Simulator::Schedule (duration, &mmWaveInterference::DoSubtractSignal, this, spd, signalId);
152 }

163 void
164 mmWaveInterference::DoSubtractSignal (Ptr<const SpectrumValue> spd, uint32_t signalId)
165 {
166  NS_LOG_FUNCTION (this << *spd);
167  ConditionallyEvaluateChunk ();
168  int32_t deltaSignalId = signalId - m_lastSignalIdBeforeReset;
169  if (deltaSignalId > 0)
170  {
171  (*m_allSignals) -= (*spd);
172  }
173  else
174  {
175  NS_LOG_INFO ("ignoring signal scheduled for subtraction before last reset");
176  }
177 }

180 void
181 mmWaveInterference::ConditionallyEvaluateChunk ()
182 {
183  NS_LOG_FUNCTION (this);
184  if (m_receiving)
185  {
186  NS_LOG_DEBUG (this << " Receiving");
187  }
188  NS_LOG_DEBUG (this << " now " << Now () << " last " << m_lastChangeTime);
189  if (m_receiving && (Now () > m_lastChangeTime))
190  {
191  NS_LOG_LOGIC (this << " signal = " << *m_rxSignal << " allSignals = " << *m_allSignals << " noise = " << *m_noise);
192  SpectrumValue interf = (*m_allSignals) - (*m_rxSignal) + (*m_noise);
193  SpectrumValue sinr = (*m_rxSignal) / interf;
194  Time duration = Now () - m_lastChangeTime;
195  for (std::list<Ptr<mmWaveChunkProcessor> >::const_iterator it = m_PowerChunkProcessorList.begin (); it != m_PowerChunkProcessorList.end (); ++it)
196  {
197  (*it)->EvaluateChunk (*m_rxSignal, duration);
198  }
199  for (std::list<Ptr<mmWaveChunkProcessor> >::const_iterator it = m_sinrChunkProcessorList.begin (); it != m_sinrChunkProcessorList.end (); ++it)
200  {
201  (*it)->EvaluateChunk (sinr, duration);
202  }
203  m_lastChangeTime = Now ();
204  }
205 }
