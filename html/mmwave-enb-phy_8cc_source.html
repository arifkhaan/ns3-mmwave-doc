269 MmWaveEnbPhy::ApplyFilter (std::vector<double> noisySinr)
270 {
271  std::vector<double> noisySinrdB;
272  for (uint64_t i = 0; i < noisySinr.size (); ++i)
273  {
274  noisySinrdB.push_back (10 * std::log10 (noisySinr.at (i)));
275  }
276 
277  std::vector<double> vectorVar;
278  NS_LOG_DEBUG ("noisySinrdBSize() " << noisySinrdB.size ());
279  for (uint64_t i = 0; i < noisySinrdB.size () - 1; ++i)
280  {
281  std::vector<double> partialSamples;
282  partialSamples.push_back (noisySinrdB.at (i));
283  partialSamples.push_back (noisySinrdB.at (i + 1));
284  double meanValue = MakeAvg (partialSamples);
285  double varValue = MakeVar (partialSamples,meanValue);
286  vectorVar.push_back (varValue);
287  partialSamples.clear ();
288  }
289 
290  uint64_t startFilter = 1e6;
291  uint64_t endFilter = 1e6;
292 
293  bool flagStartFilter = true;
294  bool flagEndFilter = true;
295 
296  uint64_t noisySinrIndex = 0;
297 
298  for (uint64_t varIndex = vectorVar.size () - 1; varIndex > 0; varIndex--) // start filter when variance of the noisy trace is high
299  {
300  NS_LOG_DEBUG ("varIndex " << varIndex);
301  noisySinrIndex = varIndex + 1;
302  NS_LOG_DEBUG ("vectorVar[i] " << vectorVar.at (varIndex));
303  NS_LOG_DEBUG ("vectorVar[i] == NaN " << (std::isnan (vectorVar.at (varIndex))));
304  bool highVariance = (vectorVar.at (varIndex) > 5 || std::isnan (vectorVar.at (varIndex)));
305  bool lowSinr = noisySinr.at (noisySinrIndex) < 10;
306 
307  if (highVariance || (lowSinr && !highVariance)) // filter is applied only for low-SINR regimes [dB]
308  {
309  endFilter = noisySinrIndex;
310  flagEndFilter = false; // a "start" sample has been identified
311  break;
312  }
313  }
314 
315  if (flagEndFilter) // in this case, we can avoid the filtering
316  {
317  endFilter = 0;
318  }
319 
320  /* in this case, consider at least a window of 15 samples, after which we can consider
321  * as we are leaving the blockage phase and we start coming back to LOS PL regimes
322  */
323  uint64_t varIndex = 0;
324  uint64_t numberOfVarWindow = 16;
325  NS_LOG_DEBUG ("VectorVarSize() " << vectorVar.size ());
326  for (uint64_t noisySinrIndex = endFilter; noisySinrIndex > numberOfVarWindow; --noisySinrIndex) // must be at least after the beginnning of the blocakge
327  {
328  NS_LOG_DEBUG ("noisySinrIndex " << noisySinrIndex);
329  varIndex = noisySinrIndex - 1;
330 
331  std::vector<double>::const_iterator first = vectorVar.begin () + varIndex;
332  std::vector<double>::const_iterator last = vectorVar.begin () + varIndex - (numberOfVarWindow - 1); // vectorVar has one sample less than noisySinrdB
333  std::vector<double> prov (last,first);
334 
335  std::vector<double>::const_iterator firstNoisy = noisySinrdB.begin () + noisySinrIndex;
336  std::vector<double>::const_iterator lastNoisy = noisySinrdB.begin () + noisySinrIndex - numberOfVarWindow;
337  std::vector<double> provNoisy (lastNoisy,firstNoisy);
338 
339  NS_LOG_INFO ("provNoisy.size " << provNoisy.size ());
340  for (std::vector<double>::const_iterator h = provNoisy.begin (); h != provNoisy.end (); h++)
341  {
342  NS_LOG_INFO ("h " << *h);
343  NS_LOG_INFO ("i " << noisySinrIndex );
344  NS_LOG_INFO ("hh " << *(noisySinrdB.begin () + noisySinrIndex - numberOfVarWindow));
345  }
346 
347 
348  /* the filtering ends when the variance of the noisy trace is almost the same, so when
349  * the SINR is on sufficiently high values
350  */
351 
352  if (Simulator::Now () > Seconds (2.1) && Simulator::Now () < Seconds (2.3))
353  {
354  NS_LOG_DEBUG ("(std::all_of(prov.begin(),prov.end(), [](double j){return j < 1;})) " << (std::all_of (prov.begin (),prov.end (), [] (double j){
355  return j < 1;
356  })));
357  NS_LOG_DEBUG ("(std::all_of(prov.begin(),prov.end(), [](double j){nan;})) " << (std::all_of (prov.begin (),prov.end (), [] (double k){
358  return !std::isnan (k);
359  })));
360  NS_LOG_DEBUG ("(std::all_of(provNoisy.begin(),provNoisy.end(), [](double p){return p > 10;})) " << (std::all_of (provNoisy.begin (),provNoisy.end (), [] (double p){
361  return p > 10;
362  })));
363  }
364 
365  if (((std::all_of (prov.begin (),prov.end (), [] (double j){
366  return j < 1;
367  }))
368  && (std::all_of (prov.begin (),prov.end (), [] (double k){
369  return !std::isnan (k);
370  })))
371  || (std::all_of (provNoisy.begin (),provNoisy.end (), [] (double p){
372  return p > 10;
373  })))
374  {
375  startFilter = noisySinrIndex;
376  flagStartFilter = false; // a "end" sample has been identified
377  break;
378  }
379 
380  // bool lowVariance = (vectorVar.at(varIndex) < 1 && !std::isnan(vectorVar.at(varIndex)));
381  // bool highSinr = noisySinr.at(noisySinrIndex) > 10;
382 
383  // if (highSinr && lowVariance) // filter is applied only for low-SINR regimes [dB]
384  // {
385  // startFilter = noisySinrIndex;
386  // flagStartFilter = false; // a "start" sample has been identified
387  // break;
388  // }
389 
390  }
391 
392  if (flagStartFilter) // in this case, filter till the end of the trace
393  {
394  startFilter = 0;
395  }
396 
397 
398  std::pair< uint64_t, uint64_t > pairFiltering = std::make_pair (startFilter,endFilter);
399 
400  return pairFiltering;
