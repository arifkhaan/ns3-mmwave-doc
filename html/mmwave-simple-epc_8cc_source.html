52 main (int argc, char *argv[])
53 {
54  uint16_t numEnb = 1;
55  uint16_t numUe = 1;
56  double simTime = 2.0;
57  double interPacketInterval = 100;
58  double minDistance = 10.0; // eNB-UE distance in meters
59  double maxDistance = 150.0; // eNB-UE distance in meters
60  bool harqEnabled = true;
61  bool rlcAmEnabled = false;
62 
63  // Command line arguments
64  CommandLine cmd;
65  cmd.AddValue ("numEnb", "Number of eNBs", numEnb);
66  cmd.AddValue ("numUe", "Number of UEs per eNB", numUe);
67  cmd.AddValue ("simTime", "Total duration of the simulation [s])", simTime);
68  cmd.AddValue ("interPacketInterval", "Inter-packet interval [us])", interPacketInterval);
69  cmd.AddValue ("harq", "Enable Hybrid ARQ", harqEnabled);
70  cmd.AddValue ("rlcAm", "Enable RLC-AM", rlcAmEnabled);
71  cmd.Parse (argc, argv);
72 
73  Config::SetDefault ("ns3::MmWaveHelper::RlcAmEnabled", BooleanValue (rlcAmEnabled));
74  Config::SetDefault ("ns3::MmWaveHelper::HarqEnabled", BooleanValue (harqEnabled));
75  Config::SetDefault ("ns3::MmWaveFlexTtiMacScheduler::HarqEnabled", BooleanValue (harqEnabled));
76  Config::SetDefault ("ns3::LteRlcAm::ReportBufferStatusTimer", TimeValue (MicroSeconds (100.0)));
77  Config::SetDefault ("ns3::LteRlcUmLowLat::ReportBufferStatusTimer", TimeValue (MicroSeconds (100.0)));
78 
79  Ptr<MmWaveHelper> mmwaveHelper = CreateObject<MmWaveHelper> ();
80  mmwaveHelper->SetSchedulerType ("ns3::MmWaveFlexTtiMacScheduler");
81  Ptr<MmWavePointToPointEpcHelper> epcHelper = CreateObject<MmWavePointToPointEpcHelper> ();
82  mmwaveHelper->SetEpcHelper (epcHelper);
83  mmwaveHelper->SetHarqEnabled (harqEnabled);
84 
85  ConfigStore inputConfig;
86  inputConfig.ConfigureDefaults ();
87 
88  // parse again so you can override default values from the command line
89  cmd.Parse (argc, argv);
90 
91  Ptr<Node> pgw = epcHelper->GetPgwNode ();
92 
93  // Create a single RemoteHost
94  NodeContainer remoteHostContainer;
95  remoteHostContainer.Create (1);
96  Ptr<Node> remoteHost = remoteHostContainer.Get (0);
97  InternetStackHelper internet;
98  internet.Install (remoteHostContainer);
99 
100  // Create the Internet
101  PointToPointHelper p2ph;
102  p2ph.SetDeviceAttribute ("DataRate", DataRateValue (DataRate ("100Gb/s")));
103  p2ph.SetDeviceAttribute ("Mtu", UintegerValue (1500));
104  p2ph.SetChannelAttribute ("Delay", TimeValue (Seconds (0.010)));
105  NetDeviceContainer internetDevices = p2ph.Install (pgw, remoteHost);
106  Ipv4AddressHelper ipv4h;
107  ipv4h.SetBase ("1.0.0.0", "255.0.0.0");
108  Ipv4InterfaceContainer internetIpIfaces = ipv4h.Assign (internetDevices);
109  // interface 0 is localhost, 1 is the p2p device
110  Ipv4Address remoteHostAddr = internetIpIfaces.GetAddress (1);
111 
112  Ipv4StaticRoutingHelper ipv4RoutingHelper;
113  Ptr<Ipv4StaticRouting> remoteHostStaticRouting = ipv4RoutingHelper.GetStaticRouting (remoteHost->GetObject<Ipv4> ());
114  remoteHostStaticRouting->AddNetworkRouteTo (Ipv4Address ("7.0.0.0"), Ipv4Mask ("255.0.0.0"), 1);
115 
116  NodeContainer ueNodes;
117  NodeContainer enbNodes;
118  enbNodes.Create (numEnb);
119  ueNodes.Create (numUe);
120 
121  // Install Mobility Model
122  Ptr<ListPositionAllocator> enbPositionAlloc = CreateObject<ListPositionAllocator> ();
123  enbPositionAlloc->Add (Vector (0.0, 0.0, 0.0));
124  MobilityHelper enbmobility;
125  enbmobility.SetMobilityModel ("ns3::ConstantPositionMobilityModel");
126  enbmobility.SetPositionAllocator (enbPositionAlloc);
127  enbmobility.Install (enbNodes);
128 
129  MobilityHelper uemobility;
130  Ptr<ListPositionAllocator> uePositionAlloc = CreateObject<ListPositionAllocator> ();
131  Ptr<UniformRandomVariable> distRv = CreateObject<UniformRandomVariable> ();
132  for (unsigned i = 0; i < numUe; i++)
133  {
134  double dist = distRv->GetValue (minDistance, maxDistance);
135  uePositionAlloc->Add (Vector (dist, 0.0, 0.0));
136  }
137  uemobility.SetMobilityModel ("ns3::ConstantPositionMobilityModel");
138  uemobility.SetPositionAllocator (uePositionAlloc);
139  uemobility.Install (ueNodes);
140 
141  // Install mmWave Devices to the nodes
142  NetDeviceContainer enbmmWaveDevs = mmwaveHelper->InstallEnbDevice (enbNodes);
143  NetDeviceContainer uemmWaveDevs = mmwaveHelper->InstallUeDevice (ueNodes);
144 
145  // Install the IP stack on the UEs
146  internet.Install (ueNodes);
147  Ipv4InterfaceContainer ueIpIface;
148  ueIpIface = epcHelper->AssignUeIpv4Address (NetDeviceContainer (uemmWaveDevs));
149  // Assign IP address to UEs, and install applications
150  for (uint32_t u = 0; u < ueNodes.GetN (); ++u)
151  {
152  Ptr<Node> ueNode = ueNodes.Get (u);
153  // Set the default gateway for the UE
154  Ptr<Ipv4StaticRouting> ueStaticRouting = ipv4RoutingHelper.GetStaticRouting (ueNode->GetObject<Ipv4> ());
155  ueStaticRouting->SetDefaultRoute (epcHelper->GetUeDefaultGatewayAddress (), 1);
156  }
157 
158  mmwaveHelper->AttachToClosestEnb (uemmWaveDevs, enbmmWaveDevs);
159 
160 
161  // Install and start applications on UEs and remote host
162  uint16_t dlPort = 1234;
163  uint16_t ulPort = 2000;
164  uint16_t otherPort = 3000;
165  ApplicationContainer clientApps;
166  ApplicationContainer serverApps;
167  for (uint32_t u = 0; u < ueNodes.GetN (); ++u)
168  {
169  ++ulPort;
170  ++otherPort;
171  PacketSinkHelper dlPacketSinkHelper ("ns3::UdpSocketFactory", InetSocketAddress (Ipv4Address::GetAny (), dlPort));
172  PacketSinkHelper ulPacketSinkHelper ("ns3::UdpSocketFactory", InetSocketAddress (Ipv4Address::GetAny (), ulPort));
173  PacketSinkHelper packetSinkHelper ("ns3::UdpSocketFactory", InetSocketAddress (Ipv4Address::GetAny (), otherPort));
174  serverApps.Add (dlPacketSinkHelper.Install (ueNodes.Get (u)));
175  serverApps.Add (ulPacketSinkHelper.Install (remoteHost));
176  serverApps.Add (packetSinkHelper.Install (ueNodes.Get (u)));
177 
178  UdpClientHelper dlClient (ueIpIface.GetAddress (u), dlPort);
179  dlClient.SetAttribute ("Interval", TimeValue (MicroSeconds (interPacketInterval)));
180  dlClient.SetAttribute ("MaxPackets", UintegerValue (1000000));
181 
182  UdpClientHelper ulClient (remoteHostAddr, ulPort);
183  ulClient.SetAttribute ("Interval", TimeValue (MicroSeconds (interPacketInterval)));
184  ulClient.SetAttribute ("MaxPackets", UintegerValue (1000000));
185

189 
190  clientApps.Add (dlClient.Install (remoteHost));
191  clientApps.Add (ulClient.Install (ueNodes.Get(u)));

200  }
201  serverApps.Start (Seconds (0.1));
202  clientApps.Start (Seconds (0.1));
203  mmwaveHelper->EnableTraces ();
204  // Uncomment to enable PCAP tracing
205  p2ph.EnablePcapAll ("mmwave-epc-simple");
