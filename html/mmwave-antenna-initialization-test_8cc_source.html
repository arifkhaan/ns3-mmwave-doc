62 void
63 MmwaveThreeGppAntennaInitializationTestCase::DoRun (void)
64 {
65  uint8_t numCc = 2; // number of CCs
66  double freq[] {6.0e9, 28.0e9}; // frequency of the CCs
67 
68  // create the MmWaveHelper
69  Ptr<MmWaveHelper> helper = CreateObject<MmWaveHelper> ();
70  helper->SetAttribute ("UseCa", BooleanValue ((numCc > 1)));
71  helper->SetAttribute ("NumberOfComponentCarriers", UintegerValue (numCc));
72 
73  // create and configure the CCs
74  std::map<uint8_t, MmWaveComponentCarrier> ccMap;
75  for (uint8_t i = 0; i < numCc; i++)
76  {
77  Ptr<MmWavePhyMacCommon> phyMacConfig = CreateObject<MmWavePhyMacCommon> ();
78  phyMacConfig->SetAttribute ("CenterFreq", DoubleValue (freq [i]));
79 
80  Ptr<MmWaveComponentCarrier> cc = CreateObject<MmWaveComponentCarrier> ();
81  cc->SetConfigurationParameters (phyMacConfig);
82  cc->SetAsPrimary ((i == 0));
83  ccMap [i] = *cc;
84  }
85 
86  // assign the map of CCs to the helper
87  helper->SetCcPhyParams (ccMap);
88 
89  // choose the pathloss model to use
90  helper->SetPathlossModelType ("ns3::ThreeGppUmaPropagationLossModel");
91 
92  // choose the channel condition model to use
93  helper->SetChannelConditionModelType ("ns3::ThreeGppUmaChannelConditionModel");
94 
95  // choose the spectrum propagation loss model
96  helper->SetChannelModelType ("ns3::ThreeGppSpectrumPropagationLossModel");
97 
98  // create the BS node
99  NodeContainer bsNodes;
100  bsNodes.Create (1);
101 
102  // create a ListPositionAllocator and add the position of the BS
103  Ptr<ListPositionAllocator> bsPositionAlloc = CreateObject<ListPositionAllocator> ();
104  bsPositionAlloc->Add (Vector (0.0, 0.0, 10.0));
105 
106  // use a MobilityHelper to install the MobilityModel for the BS
107  MobilityHelper bsMobility;
108  bsMobility.SetMobilityModel ("ns3::ConstantPositionMobilityModel");
109  bsMobility.SetPositionAllocator (bsPositionAlloc);
110  bsMobility.Install (bsNodes);
111 
112  // install the BS device on the proper Node
113  NetDeviceContainer bsNetDev = helper->InstallEnbDevice (bsNodes);
114 
115  // retrive the MmWaveEnbNetDevice
116  Ptr<MmWaveEnbNetDevice> mmWaveEnbDev = DynamicCast<MmWaveEnbNetDevice> (bsNetDev.Get (0));
117 
118  // iterate through the CCs
119  for (auto ccIndex : mmWaveEnbDev->GetCcMap ())
120  {
121  // retrieve the MmWaveEnbPhy
122  Ptr<MmWaveEnbPhy> mmWaveEnbPhy = mmWaveEnbDev->GetPhy (ccIndex.first);
123 
124  // retrieve the MmWaveSpectrumPhy
125  Ptr<MmWaveSpectrumPhy> mmWaveSpectrumPhy = mmWaveEnbPhy->GetDlSpectrumPhy ();
126 
127  // retrieve the MmWaveBeamformingModel
128  Ptr<MmWaveBeamformingModel> mmwaveBfModel = mmWaveSpectrumPhy->GetBeamformingModel ();
129 
130  // retrieve the ThreeGppAntennaArrayModel
131  Ptr<ThreeGppAntennaArrayModel> antenna = mmwaveBfModel->GetAntenna ();
132 
133  // check if the number of antenna elements has been configured
134  NS_TEST_ASSERT_MSG_EQ (antenna->GetNumberOfElements (), mmWaveEnbDev->GetAntennaNum (),"The number of antenna elements was not properly configured");
135  }
136 
137  // create the UE node
138  NodeContainer ueNodes;
139  ueNodes.Create (1);
140 
141  // create a ListPositionAllocator and add the position of the UE
142  Ptr<ListPositionAllocator> uePositionAlloc = CreateObject<ListPositionAllocator> ();
143  uePositionAlloc->Add (Vector (0.0, 0.0, 10.0));
144 
145  // use a MobilityHelper to install the MobilityModel for the UE
146  MobilityHelper ueMobility;
147  ueMobility.SetMobilityModel ("ns3::ConstantPositionMobilityModel");
148  ueMobility.SetPositionAllocator (uePositionAlloc);
149  ueMobility.Install (ueNodes);
150 
151  // install the UE device on the proper Node
152  NetDeviceContainer ueNetDev = helper->InstallUeDevice (ueNodes);
153 
154  // retrieve the MmWaveUeNetDevice
155  Ptr<MmWaveUeNetDevice> mmWaveUeDev = DynamicCast<MmWaveUeNetDevice> (ueNetDev.Get (0));
156 
157  for (auto ccIndex : mmWaveUeDev->GetCcMap ())
158  {
159  // retrieve the MmWaveUePhy
160  Ptr<MmWaveUePhy> mmWaveUePhy = mmWaveUeDev->GetPhy (ccIndex.first);
161 
162  // retrieve the MmWaveSpectrumPhy
163  Ptr<MmWaveSpectrumPhy> mmWaveSpectrumPhy = mmWaveUePhy->GetDlSpectrumPhy ();
164 
165  // retrieve the MmWaveBeamformingModel
166  Ptr<MmWaveBeamformingModel> mmwaveBfModel = mmWaveSpectrumPhy->GetBeamformingModel ();
167 
168  // retrieve the ThreeGppAntennaArrayModel
169  Ptr<ThreeGppAntennaArrayModel> antenna = mmwaveBfModel->GetAntenna ();
170 
171  // check if the number of antenna elements has been configured
172  NS_TEST_ASSERT_MSG_EQ (antenna->GetNumberOfElements (), mmWaveUeDev->GetAntennaNum (),"The number of antenna elements was not properly configured");
173  }
