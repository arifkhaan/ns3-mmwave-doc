192 MmWavePhyTrace::ReportUlPhyTransmissionCallback (Ptr<MmWavePhyTrace> phyStats, PhyTransmissionTraceParams param)
193 {
194  if (!m_ulPhyTraceFile.is_open ())
195  {
196  m_ulPhyTraceFile.open (m_ulPhyTraceFilename.c_str ());
197  if (!m_ulPhyTraceFile.is_open ())
198  {
199  NS_FATAL_ERROR ("Could not open tracefile");
200  }
201  m_ulPhyTraceFile << "frame\tsubF\tslot\trnti\tfirstSym\tnumSym\ttype\ttddMode\tretxNum\tccId" << std::endl;
202  }
203 
204  // Trace the UL PHY transmission info
205  m_ulPhyTraceFile << +param.m_frameNum << "\t" << +param.m_sfNum << "\t"
206  << +param.m_slotNum << "\t" << +param.m_rnti << "\t"
207  << +param.m_symStart << "\t" << +param.m_numSym << "\t"
208  << +param.m_ttiType << "\t" << +param.m_tddMode << "\t"
209  << +param.m_rv << "\t" << +param.m_ccId << std::endl;

234 MmWavePhyTrace::RxPacketTraceUeCallback (Ptr<MmWavePhyTrace> phyStats, std::string path, RxPacketTraceParams params)
235 {
236  if (!m_rxPacketTraceFile.is_open ())
237  {
238  m_rxPacketTraceFile.open (m_rxPacketTraceFilename.c_str ());
239  m_rxPacketTraceFile << "DL/UL\ttime\tframe\tsubF\tslot\t1stSym\tsymbol#\tcellId\trnti\tccId\ttbSize\tmcs\trv\tSINR(dB)\tcorrupt\tTBler" << std::endl;
240  if (!m_rxPacketTraceFile.is_open ())
241  {
242  NS_FATAL_ERROR ("Could not open tracefile");
243  }
244  }
245  m_rxPacketTraceFile << "DL\t" << Simulator::Now ().GetSeconds () << "\t"
246  << params.m_frameNum << "\t" << +params.m_sfNum << "\t"
247  << +params.m_slotNum << "\t" << +params.m_symStart << "\t"
248  << +params.m_numSym << "\t" << params.m_cellId << "\t"
249  << params.m_rnti << "\t" << +params.m_ccId << "\t"
250  << params.m_tbSize << "\t" << +params.m_mcs << "\t"
251  << +params.m_rv << "\t" << 10 * std::log10 (params.m_sinr) << "\t"
252  << params.m_corrupt << "\t" << params.m_tbler << std::endl;
253 
254  if (params.m_corrupt)
255  {
256  NS_LOG_DEBUG ("DL TB error\t" << params.m_frameNum << "\t" << +params.m_sfNum << "\t"
257  << +params.m_slotNum << "\t" << +params.m_symStart << "\t"
258  << +params.m_numSym << "\t" << params.m_rnti << "\t"
259  << +params.m_ccId << "\t" << params.m_tbSize << "\t"
260  << +params.m_mcs << "\t" << +params.m_rv << "\t"
261  << 10 * std::log10 (params.m_sinr) << "\t" << params.m_tbler << "\t"
262  << params.m_corrupt);
263  }
