68 double
69 MmWaveHarqPhy::GetAccumulatedMiDl (uint16_t rnti, uint8_t harqId)
70 {
71  NS_LOG_FUNCTION (this << (uint16_t)rnti << (uint16_t)harqId);
72  std::map <uint16_t, std::vector <MmWaveHarqProcessInfoList_t> >::iterator it;
73  it = m_miDlHarqProcessesInfoMap.find (rnti);
74  NS_ASSERT_MSG (it != m_miDlHarqProcessesInfoMap.end (), " Does not find MI for RNTI");
75  MmWaveHarqProcessInfoList_t list = (*it).second.at (harqId);
76  double mi = 0.0;
77  for (uint8_t i = 0; i < list.size (); i++)
78  {
79  mi += list.at (i).m_mi;
80  }
81  return (mi);
82 }

144 void
145 MmWaveHarqPhy::UpdateDlHarqProcessStatus (uint16_t rnti, uint8_t harqId, double mi, uint32_t infoBytes, uint32_t codeBytes)
146 {
147  NS_LOG_FUNCTION (this << (uint16_t) harqId << mi);
148  std::map <uint16_t, std::vector <MmWaveHarqProcessInfoList_t> >::iterator it;
149  it = m_miDlHarqProcessesInfoMap.find (rnti);
150  if (it == m_miDlHarqProcessesInfoMap.end ())
151  {
152  // new entry
153  std::vector <MmWaveHarqProcessInfoList_t> harqList;
154  harqList.resize (m_harqNum);
155  MmWaveHarqProcessInfoElement_t el;
156  el.m_mi = mi;
157  el.m_rv = 0;
158  el.m_infoBits = infoBytes * 8;
159  el.m_codeBits = codeBytes * 8;
160  harqList.at (harqId).push_back (el);
161  m_miDlHarqProcessesInfoMap.insert (std::pair <uint16_t, std::vector <MmWaveHarqProcessInfoList_t> > (rnti, harqList));
162  }
163  else
164  {
165  if ((*it).second.at (harqId).size () == 3) // MAX HARQ RETX
166  {
167  // HARQ should be disabled -> discard info
168  return;
169  }
170  MmWaveHarqProcessInfoElement_t el;
171  el.m_mi = mi;
172  if (!(*it).second.at (harqId).empty ())
173  {
174  el.m_rv = (*it).second.at (harqId).back ().m_rv + 1;
175  }
176  else
177  {
178  el.m_rv = 0;
179  }
180  el.m_infoBits = infoBytes * 8;
181  el.m_codeBits = codeBytes * 8;
182  (*it).second.at (harqId).push_back (el);
183  }
184 }
